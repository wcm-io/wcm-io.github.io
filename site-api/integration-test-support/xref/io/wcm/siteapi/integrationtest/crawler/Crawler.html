20  package io.wcm.siteapi.integrationtest.crawler;
21  
22  import java.util.Collection;
23  import java.util.Collections;
24  import java.util.HashSet;
25  import java.util.LinkedHashSet;
26  import java.util.List;
27  import java.util.Set;
28  
29  import org.jetbrains.annotations.NotNull;
30  import org.slf4j.Logger;
31  import org.slf4j.LoggerFactory;
32  
33  import io.wcm.siteapi.integrationtest.IntegrationTestContext;
34  import io.wcm.siteapi.integrationtest.linkextractor.LinkExtractor;
35  
36  /**
37   * Site API crawler.
38   */
39  public final class Crawler {
40  
41    private final IntegrationTestContext context;
42    private final Collection<LinkExtractor> linkExtractors;
43    private final Set<String> visitedUrls = new HashSet<>();
44    private final Set<String> failedUrls = new LinkedHashSet<>();
45    private final UrlParser urlParser;
46  
47    private static final Logger log = LoggerFactory.getLogger(Crawler.class.getSimpleName());
48  
49    /**
50     * @param context Integration test context
51     * @param linkExtractors Link extractors.
52     */
53    public Crawler(@NotNull IntegrationTestContext context,
54        @NotNull List<LinkExtractor> linkExtractors) {
55      this.context = context;
56      this.linkExtractors = Collections.unmodifiableCollection(linkExtractors);
57      this.urlParser = new UrlParser(context.getPublishUrl(),
58          context.getSelector(), context.getApiVersion(), context.getExtension());
59    }
60  
61    /**
62     * Start API crawling.
63     * @param url API index URL. URL is automatically prefixed with Publish URL if not present.
64     */
65    public void start(@NotNull String url) {
66      CrawlerItem item = new CrawlerItem(this, context, url);
67      item.fetch();
68    }
69  
70    /**
71     * Is called when a URL should be visited.
72     * @param url Url to visit
73     * @return true if the page was not already visited
74     */
75    public boolean visitUrl(@NotNull String url) {
76      boolean doVisit = visitedUrls.add(url);
77      if (doVisit) {
78        log.info("Visit: {}", url);
79      }
80      else {
81        log.debug("Skip: {}", url);
82      }
83      return doVisit;
84    }
85  
86    /**
87     * Log a failed visit.
88     * @param url url
89     * @param message Message
90     * @param fullResponse Full response
91     */
92    public void logFailedVisitUrl(@NotNull String url, @NotNull String message, @NotNull String fullResponse) {
93      log.error("Validation FAILED: {}\n{}\n\n{}\n", url, message, fullResponse);
94      failedUrls.add(url);
95    }
96  
97    /**
98     * @return Total number of visited URLs.
99     */
100   public int numberOfVisits() {
101     return visitedUrls.size();
102   }
103 
104   /**
105    * @return Number of failed visits.
106    */
107   public int numberOfFailedVisits() {
108     return failedUrls.size();
109   }
110 
111   /**
112    * @return URLs of failed visits.
113    */
114   public @NotNull Collection<String> failedVisitUrls() {
115     return Collections.unmodifiableCollection(failedUrls);
116   }
117 
118   @NotNull
119   Collection<LinkExtractor> getLinkExtractors() {
120     return this.linkExtractors;
121   }
122 
123   @NotNull
124   String parseSuffix(@NotNull String url) {
125     return urlParser.parseSuffix(url);
126   }
127 
128 }
