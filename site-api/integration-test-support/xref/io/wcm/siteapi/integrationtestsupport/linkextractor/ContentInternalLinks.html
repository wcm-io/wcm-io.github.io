20  package io.wcm.siteapi.integrationtestsupport.linkextractor;
21  
22  import static com.jayway.jsonpath.Criteria.where;
23  import static com.jayway.jsonpath.Filter.filter;
24  
25  import java.util.List;
26  import java.util.Map;
27  import java.util.Set;
28  import java.util.stream.Stream;
29  
30  import com.jayway.jsonpath.DocumentContext;
31  import com.jayway.jsonpath.JsonPath;
32  
33  /**
34   * Generic extractor that is able to detect any "internal" links produced by
35   * <a href="https://wcm.io/handler/link/">wcm.io Link Handler</a>.
36   *
37   * <p>
38   * It used very basic heuristic and looks for any JSON elements with a properties "url" property,
39   * and a "type=internal" property.
40   * </p>
41   */
42  public final class ContentInternalLinks implements LinkExtractor {
43  
44    private static final Set<String> SUFFIXES = Set.of("content", "navigation", "config");
45    private static final JsonPath JSON_PATH = JsonPath.compile("$..[?]",
46        filter(where("url").exists(true).and("type").is("internal")));
47  
48    @Override
49    public boolean accept(String suffix) {
50      return SUFFIXES.contains(suffix);
51    }
52  
53    @Override
54    public Stream<String> getLinks(DocumentContext jsonPathContext) {
55      final List<Map<String, String>> contentLinkObjects = jsonPathContext.read(JSON_PATH);
56      return contentLinkObjects
57          .stream()
58          .map(item -> item.get("url"));
59    }
60  
61  }
