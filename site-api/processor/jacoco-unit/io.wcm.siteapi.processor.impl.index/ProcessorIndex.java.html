package io.wcm.siteapi.processor.impl.index;

/**
 * Pojo representing an index entry for a processor.
 */
class ProcessorIndex {

  private final String suffix;
  private final String url;

  ProcessorIndex(String suffix, String url) {
    this.suffix = suffix;
    this.url = url;
  }

  public String getSuffix() {
    return this.suffix;
  }

  public String getUrl() {
    return this.url;
  }

}
