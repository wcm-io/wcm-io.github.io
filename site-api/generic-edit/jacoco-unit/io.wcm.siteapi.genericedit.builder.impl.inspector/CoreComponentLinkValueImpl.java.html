package io.wcm.siteapi.genericedit.builder.impl.inspector;

import java.util.Map;

import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import com.adobe.cq.wcm.core.components.commons.link.Link;

import io.wcm.siteapi.genericedit.component.value.LinkValue;

class CoreComponentLinkValueImpl implements LinkValue {

  private final Link<?> link;

  CoreComponentLinkValueImpl(@NotNull Link<?> link) {
    this.link = link;
  }

  @Override
  public @Nullable String getUrl() {
    return link.getURL();
  }

  @Override
  public @Nullable Map<String, String> getHtmlAttributes() {
    return link.getHtmlAttributes();
  }

  @Override
  public boolean isValid() {
    return link.isValid();
  }

}
