package io.wcm.siteapi.genericedit.handler.impl.inspector;

import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import io.wcm.handler.media.Asset;
import io.wcm.handler.media.Media;
import io.wcm.siteapi.genericedit.component.value.MediaValue;

class MediaValueImpl implements MediaValue {

  private final Media media;
  private final String assetPath;

  MediaValueImpl(@NotNull Media media) {
    this.media = media;
    this.assetPath = getAssetPath(media);
  }

  @Override
  public String getUrl() {
    return media.getUrl();
  }

  @Override
  public String getMarkup() {
    return media.getMarkup();
  }

  @Override
  public String getAssetPath() {
    return this.assetPath;
  }

  @Override
  public boolean isValid() {
    return media.isValid();
  }

  private static @Nullable String getAssetPath(Media media) {
    Asset asset = media.getAsset();
    if (asset != null) {
      return asset.getPath();
    }
    return null;
  }

}
