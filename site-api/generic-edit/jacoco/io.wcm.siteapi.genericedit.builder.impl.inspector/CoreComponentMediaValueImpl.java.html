package io.wcm.siteapi.genericedit.builder.impl.inspector;

import org.apache.sling.xss.XSSAPI;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import io.wcm.siteapi.genericedit.component.value.MediaValue;

class CoreComponentMediaValueImpl implements MediaValue {

  private final String imageUrl;
  private final String markup;

  CoreComponentMediaValueImpl(@NotNull String imageUrl, @NotNull XSSAPI xssApi) {
    this.imageUrl = imageUrl;
    markup = "<img src=\"" + xssApi.encodeForHTMLAttr(imageUrl) + "\"/>";
  }

  @Override
  public @Nullable String getUrl() {
    return imageUrl;
  }

  @Override
  public @Nullable String getMarkup() {
    return markup;
  }

  @Override
  public @Nullable String getAssetPath() {
    return null;
  }

  @Override
  public boolean isValid() {
    return true;
  }

  @Override
  public String toString() {
    return imageUrl;
  }

}
