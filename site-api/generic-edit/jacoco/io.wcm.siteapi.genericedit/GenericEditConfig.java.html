package io.wcm.siteapi.genericedit;

import org.jetbrains.annotations.NotNull;
import org.osgi.service.component.annotations.Activate;
import org.osgi.service.component.annotations.Component;
import org.osgi.service.metatype.annotations.AttributeDefinition;
import org.osgi.service.metatype.annotations.Designate;
import org.osgi.service.metatype.annotations.ObjectClassDefinition;

/**
 * Generic Edit Configuration.
 */
@Component(service = GenericEditConfig.class, immediate = true)
@Designate(ocd = GenericEditConfig.Config.class)
public class GenericEditConfig {

  @ObjectClassDefinition(
      name = "wcm.io Site API Generic Edit Configuration",
      description = "Configures the generic edit mode.")
  @interface Config {

    @AttributeDefinition(
        name = "Selector",
        description = "Selector name to enable generic edit mode.")
    String selector() default "generic-edit";

  }

  private String selector;

  @Activate
  private void activate(Config config) {
    this.selector = config.selector();
  }

  /**
   * @return Selector name to enable generic edit mode
   */
  public @NotNull String getSelector() {
    return this.selector;
  }

}
