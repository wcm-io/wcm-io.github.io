128   @PostConstruct
129   private void activate() {
130     ValueMap properties = resource.getValueMap();
131 
132     // read basic properties
133     displayPopupTitle = properties.get(PN_DISPLAY_POPUP_TITLE, currentStyle.get(PN_DISPLAY_POPUP_TITLE, true));
134     enableLazyLoading = !currentStyle.get(PN_DESIGN_LAZY_LOADING_ENABLED, true);
135     lazyThreshold = currentStyle.get(PN_DESIGN_LAZY_THRESHOLD, 0);
136     isDecorative = properties.get(PN_IS_DECORATIVE, currentStyle.get(PN_IS_DECORATIVE, false));
137     boolean altFromAsset = properties.get(PN_ALT_VALUE_FROM_DAM, currentStyle.get(PN_ALT_VALUE_FROM_DAM, true));
138 
139     // resolve media and properties from DAM asset
140     media = HandlerUnwrapper.get(mediaHandler, resource)
141         // disable dynamic media support as it is not compatible with the "src-pattern" concept
142         .dynamicMediaDisabled(true)
143         .decorative(isDecorative)
144         .forceAltValueFromAsset(altFromAsset)
145         .build();
146     if (media.isValid() && !media.getRendition().isImage()) {
147       // no image asset selected (cannot be rendered) - set to invalid
148       media = mediaHandler.invalid();
149     }
150     if (media.isValid()) {
151       initPropertiesFromDamAsset(properties);
152       widths = buildRenditionWidths(media.getRendition());
153       noScriptWidth = getNoScriptWidth();
154       srcPattern = buildSrcPattern(media.getUrl());
155       areas = ImageAreaV1Impl.convertMap(media.getMap());
156     }
157 
158     // resolve link - decorative images have no link and no alt text by definition
159     if (isDecorative) {
160       link = linkHandler.invalid();
161     }
162     else {
163       link = HandlerUnwrapper.get(linkHandler, resource).build();
164     }
165   }

167   /**
168    * Checks if the resolved media is a DAM asset, and initializes properties from it.
169    * @param properties Resource properties
170    */
171   private void initPropertiesFromDamAsset(ValueMap properties) {
172     Asset asset = media.getAsset();
173     if (asset != null) {
174       com.day.cq.dam.api.Asset damAsset = asset.adaptTo(com.day.cq.dam.api.Asset.class);
175       if (damAsset != null) {
176         boolean titleFromAsset = properties.get(PN_TITLE_VALUE_FROM_DAM, currentStyle.get(PN_TITLE_VALUE_FROM_DAM, true));
177         boolean uuidDisabled = currentStyle.get(PN_UUID_DISABLED, false);
178 
179         fileReference = damAsset.getPath();
180         alt = asset.getAltText();
181 
182         if (!uuidDisabled) {
183           uuid = damAsset.getID();
184         }
185 
186         if (titleFromAsset) {
187           String assetTitle = asset.getTitle();
188           if (StringUtils.isNotEmpty(assetTitle)) {
189             title = assetTitle;
190           }
191         }
192       }
193     }
194   }

287   /**
288    * Validates the configured list of widths, removes those that are bigger than the original rendition,
289    * and returns them sorted by size.
290    * @param rendition Primary rendition
291    * @return Widths
292    */
293   private List<Long> buildRenditionWidths(Rendition rendition) {
294     long maxWidth = rendition.getWidth();
295     String[] configuredWidths = currentStyle.get(PN_DESIGN_ALLOWED_RENDITION_WIDTHS, new String[0]);
296     return Arrays.stream(configuredWidths)
297         .map(width -> NumberUtils.toLong(width))
298         .filter(width -> width > 0 && width <= maxWidth)
299         .sorted()
300         .collect(Collectors.toList());
301   }
302 
303   /**
304    * Picks a "medium" widths from the set of available widths.
305    * @return Picked width
306    */
307   private long getNoScriptWidth() {
308     if (widths.isEmpty()) {
309       return 0;
310     }
311     return widths.get((int)Math.round(widths.size() / 2d - 0.5d));
312   }

314   /**
315    * Build image url pattern based in ImageWidthServlet for dynamic rendition selection.
316    * @param mediaUrl Media Url
317    * @return Url pattern
318    */
319   private String buildSrcPattern(String mediaUrl) {
320     String extension = StringUtils.substringAfterLast(media.getUrl(), ".");
321     String fileName = StringUtils.substringAfterLast(media.getUrl(), "/");
322 
323     // build suffix containing file name and crop/rotation parameters as "cache killers"
324     StringBuilder suffix = new StringBuilder();
325     if (StringUtils.isNotEmpty(imageCrop)) {
326       suffix.append(imageCrop).append("/");
327     }
328     if (StringUtils.isNotEmpty(imageRotate)) {
329       suffix.append(imageRotate).append("/");
330     }
331     suffix.append(fileName);
332 
333     String url = urlHandler.get(resource)
334         .selectors(ImageWidthProxyServlet.SELECTOR)
335         .extension(extension)
336         .suffix(suffix.toString())
337         .buildExternalResourceUrl();
338 
339     // insert {.width} placeholder for rendition selection
340     return StringUtils.replace(url, "." + ImageWidthProxyServlet.SELECTOR + ".",
341         "." + ImageWidthProxyServlet.SELECTOR + WIDTH_PLACEHOLDER + ".");
342   }

346   @Override
347   @JsonIgnore
348   @SuppressWarnings("null")
349   public @NotNull ImageData getComponentData() {
350     return DataLayerBuilder.extending(super.getComponentData()).asImageComponent()
351         .withTitle(this::getTitle)
352         .withLinkUrl(this::getLink)
353         .withAssetData(() -> Optional.of(media)
354             .filter(Media::isValid)
355             .map(Media::getAsset)
356             .map(asset -> asset.adaptTo(com.day.cq.dam.api.Asset.class))
357             .map(DataLayerBuilder::forAsset)
358             .map(AssetDataBuilder::build)
359             .orElse(null))
360         .build();
361   }
