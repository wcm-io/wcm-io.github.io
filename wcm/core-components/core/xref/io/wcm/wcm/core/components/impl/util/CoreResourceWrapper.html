20  package io.wcm.wcm.core.components.impl.util;
21  
22  import java.util.HashMap;
23  import java.util.Map;
24  import java.util.Set;
25  
26  import org.apache.commons.lang3.StringUtils;
27  import org.apache.sling.api.resource.Resource;
28  import org.apache.sling.api.resource.ResourceResolver;
29  import org.apache.sling.api.resource.ResourceWrapper;
30  import org.apache.sling.api.resource.ValueMap;
31  import org.apache.sling.api.wrappers.ValueMapDecorator;
32  import org.apache.sling.models.annotations.Exporter;
33  import org.jetbrains.annotations.NotNull;
34  import org.jetbrains.annotations.Nullable;
35  
36  import com.adobe.cq.export.json.ExporterConstants;
37  
38  /**
39   * Resource wrapper that overwrites the resource type of the target resource.
40   * Additionally supports overwriting and hiding some properties.
41   */
42  @Exporter(name = ExporterConstants.SLING_MODEL_EXPORTER_NAME, extensions = ExporterConstants.SLING_MODEL_EXTENSION)
43  public class CoreResourceWrapper extends ResourceWrapper {
44  
45    private ValueMap valueMap;
46    private String overriddenResourceType;
47  
48    /**
49     * @param resource Target resource
50     * @param overriddenResourceType New resource type
51     */
52    public CoreResourceWrapper(@NotNull Resource resource, @NotNull String overriddenResourceType) {
53      this(resource, overriddenResourceType, null, null);
54    }
55  
56    /**
57     * @param resource Target resource
58     * @param overriddenResourceType New resource type
59     * @param overriddenProperties Properties to add/overwrite in value map
60     * @param hiddenProperties Properties to hide from value map
61     */
62    public CoreResourceWrapper(@NotNull Resource resource, @NotNull String overriddenResourceType,
63        @Nullable Map<String, Object> overriddenProperties, @Nullable Set<String> hiddenProperties) {
64      super(resource);
65      if (StringUtils.isEmpty(overriddenResourceType)) {
66        throw new IllegalArgumentException("The " + CoreResourceWrapper.class.getName() + " needs to override the resource type of " +
67            "the wrapped resource, but the resourceType argument was null or empty.");
68      }
69      this.overriddenResourceType = overriddenResourceType;
70      valueMap = new ValueMapDecorator(new HashMap<>(resource.getValueMap()));
71      valueMap.put(ResourceResolver.PROPERTY_RESOURCE_TYPE, overriddenResourceType);
72      if (overriddenProperties != null) {
73        valueMap.putAll(overriddenProperties);
74      }
75      if (hiddenProperties != null) {
76        hiddenProperties.forEach(valueMap::remove);
77      }
78    }
79  
80    @Override
81    @SuppressWarnings("unchecked")
82    public <AdapterType> AdapterType adaptTo(Class<AdapterType> type) {
83      if (type == ValueMap.class) {
84        return (AdapterType)valueMap;
85      }
86      return super.adaptTo(type);
87    }
88  
89    @Override
90    @NotNull
91    public ValueMap getValueMap() {
92      return valueMap;
93    }
94  
95    @Override
96    @NotNull
97    public String getResourceType() {
98      return overriddenResourceType;
99    }
100 
101   @Override
102   public boolean isResourceType(String resourceType) {
103     return this.getResourceResolver().isResourceType(this, resourceType);
104   }
105 
106 }
