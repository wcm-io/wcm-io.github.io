53  /**
54   * Model for customized columnview Granite UI component for path field.
55   * Logic derived from <code>/libs/granite/ui/components/coral/foundation/columnview/columnview.jsp</code>
56   */
57  @Model(adaptables = SlingHttpServletRequest.class)
58  @ProviderType
59  public final class ColumnView {
60  
61    private static final String FALLBACK_ROOT_RESOURCE = "/";
62    private static final long DEFAULT_PAGINATION_LIMIT = 1000;
63  
64    private static final String NN_DATASOURCE = "datasource";
65    private static final String PN_SIZE = "size";
66    private static final String PN_LIMIT = "limit";
67    private static final String PN_ITEM_RESOURCE_TYPE = "itemResourceType";
68    private static final String PN_SHOW_ROOT = "showRoot";
69    private static final String PN_LOAD_ANCESTORS = "loadAncestors";
70    private static final String PN_PATH = "path";
71  
72    @SlingObject
73    private Resource componentResource;
74    @SlingObject
75    private SlingHttpServletRequest request;
76    @SlingObject
77    private SlingHttpServletResponse response;
78    @SlingObject
79    private ResourceResolver resourceResolver;
80  
81    private Resource currentResource;
82    private final List<Column> columns = new ArrayList<>();
83  
84    @PostConstruct
85    @SuppressWarnings("null")
86    private void activate() {
87      ComponentHelper cmp = new ComponentHelper(new DummyPageContext(request, response));
88      Config cfg = cmp.getConfig();
89      ExpressionHelper ex = cmp.getExpressionHelper();
90  
91      Integer size = ex.get(cfg.get(PN_SIZE, String.class), Integer.class);
92      Long limit = ex.get(cfg.get(PN_LIMIT, String.class), Long.class);
93      String itemResourceType = cfg.get(PN_ITEM_RESOURCE_TYPE, String.class);
94      boolean showRoot = cfg.get(PN_SHOW_ROOT, false);
95      boolean loadAncestors = cfg.get(PN_LOAD_ANCESTORS, false);
96  
97      // make sure we always have a valid root resource
98      Resource rootResource = resourceResolver.getResource(ex.getString(cfg.get("rootPath", FALLBACK_ROOT_RESOURCE)));
99      if (rootResource == null) {
100       rootResource = resourceResolver.getResource(FALLBACK_ROOT_RESOURCE);
101     }
102 
103     // if current resource is invalid or not same or descendant of root resource, set it to root resource
104     String path = ex.getString(cfg.get(PN_PATH, rootResource.getPath()));
105     currentResource = resourceResolver.getResource(path);
106     if (currentResource == null || !isSameResourceOrChild(rootResource, currentResource)) {
107       currentResource = rootResource;
108     }
109 
110     // generate column for root
111     if (showRoot && (StringUtils.equals(currentResource.getPath(), rootResource.getPath()) || loadAncestors)) {
112       columns.add(getRootColumn(rootResource, itemResourceType));
113     }
114 
115     // generate columns for ancestors
116     if (loadAncestors) {
117       columns.addAll(getAncestorColumns(currentResource, rootResource));
118     }
119 
120     // calculate total number of items to return
121     // NOTE: i assume the logic was intended in a different way, using DEFAULT_PAGINATION_LIMIT as max cap and not
122     // as minimum - but the logic in columnview.jsp is exactly like this
123     long totalSize = DEFAULT_PAGINATION_LIMIT;
124     if (limit != null) {
125       totalSize = Math.max(totalSize, limit);
126     }
127     if (size != null) {
128       totalSize = Math.max(totalSize, size);
129     }
130 
131     // check if a limit value is defined for the data source
132     Long limitFromDataSource = null;
133     Resource datasourceResource = componentResource.getChild(NN_DATASOURCE);
134     if (datasourceResource != null) {
135       limitFromDataSource = ex.get(datasourceResource.getValueMap().get(PN_LIMIT, String.class), Long.class);
136     }
137 
138     DataSource dataSource = null;
139     if (size != null && size >= 20 && datasourceResource != null && limitFromDataSource != null) {
140       // if a limit is configured for the data source or size is at least 20,
141       // calculate a new limit for the data source and overwrite it (synthetic) in the data source definition
142       long newLimit = limitFromDataSource + totalSize - size + 1;
143       dataSource = getDataSource(cmp, currentResource, datasourceResource, newLimit);
144     }
145     else {
146       dataSource = getDataSource(cmp, currentResource);
147       if (size != null) {
148         totalSize = size;
149       }
150     }
151 
152     // generate columns for items
153     columns.add(getCurrentResourceColumn(dataSource, totalSize, currentResource, itemResourceType));
154   }
155 
156   private boolean isSameResourceOrChild(Resource rootResource, Resource resource) {
157     if (StringUtils.equals(rootResource.getPath(), resource.getPath())) {
158       return true;
159     }
160     else {
161       return StringUtils.startsWith(resource.getPath(), rootResource.getPath() + "/");
162     }
163   }
164 
165   public Resource getCurrentResource() {
166     return this.currentResource;
167   }
168 
169   public List<Column> getColumns() {
170     return this.columns;
171   }

173   /**
174    * Get data source to list children of given resource.
175    * @param cmp Component helper
176    * @param resource Given resource
177    * @return Data source
178    */
179   private DataSource getDataSource(ComponentHelper cmp, Resource resource) {
180     return getDataSource(cmp, resource, null, null);
181   }

184   /**
185    * Get data source to list children of given resource.
186    * @param cmp Component helper
187    * @param resource Resource pointing to current path
188    * @param dataSourceResource Data source resource
189    * @param newLimit Set limit defined in data source to this new value
190    * @return Data source
191    */
192   @SuppressWarnings("java:S112") // allow generic exception
193   private DataSource getDataSource(@NotNull ComponentHelper cmp, @NotNull Resource resource,
194       @Nullable Resource dataSourceResource, @Nullable Long newLimit) {
195     try {
196       /*
197        * by default the path is read from request "path" parameter
198        * here we overwrite it via a synthetic resource because the path may be overwritten by validation logic
199        * to ensure the path is not beyond the configured root path
200        */
201       ValueMap overwriteProperties = new ValueMapDecorator(Map.of(PN_PATH, resource.getPath()));
202       Resource resourceWrapper = GraniteUiSyntheticResource.wrapMerge(componentResource, overwriteProperties);
203 
204       if (dataSourceResource != null && newLimit != null) {
205         // overwrite limit property in data source definition
206         ValueMap overwriteDataSourceProperties = new ValueMapDecorator(Map.of(PN_LIMIT, newLimit));
207         Resource dataSourceResourceWrapper = GraniteUiSyntheticResource.child(resourceWrapper, NN_DATASOURCE,
208             dataSourceResource.getResourceType(),
209             new CompositeValueMap(overwriteDataSourceProperties, dataSourceResource.getValueMap()));
210         return cmp.asDataSource(dataSourceResourceWrapper, resourceWrapper);
211       }
212       else {
213         return cmp.getItemDataSource(resourceWrapper);
214       }
215     }
216     catch (ServletException | IOException ex) {
217       throw new RuntimeException("Unable to get data source.", ex);
218     }
219   }

221   /**
222    * Generate column for data source items for current resource.
223    * @param dataSource Data source
224    * @param totalSize Size limit
225    * @param currentResource Current resource
226    * @param itemResourceType Item resource type
227    * @return Column
228    */
229   private static Column getCurrentResourceColumn(DataSource dataSource, long totalSize,
230       Resource currentResource, String itemResourceType) {
231 
232     Iterator<Resource> items = dataSource.iterator();
233 
234     boolean hasMore = false;
235     List<Resource> list = new ArrayList<>();
236     while (items.hasNext() && list.size() < totalSize) {
237       list.add(items.next());
238     }
239     hasMore = items.hasNext();
240     items = list.iterator();
241 
242     Column column = new Column()
243         .isCurrentResource(true)
244         .columnId(currentResource.getPath())
245         .hasMore(hasMore)
246         .metaElement(true);
247     while (items.hasNext()) {
248       Resource item = items.next();
249       column.addItem(new ColumnItem(item)
250           .resourceType(itemResourceType));
251     }
252 
253     return column;
254   }

256   /**
257    * Generate extra column representing the root resource.
258    * @param rootResource Root resource
259    * @param itemResourceType Item resource type
260    * @return Column
261    */
262   private static Column getRootColumn(Resource rootResource, String itemResourceType) {
263     /*
264      * Put a special path for columnId to avoid having the same columnId with the next column to avoid breaking the contract of columnId.
265      * The contract of columnId is that it should be a path of the current column, i.e. the path should be a path representing a parent.
266      * e.g. When columnId = "/", then the column will show the children of this path, such as "/a", "/b".
267      * So for showRoot scenario, if we want to show the item with path = "/", we need to generate the column having a columnId with value of the parent of "/".
268      * Since the cannot have a parent of "/", then we decide to just use a special convention ("parentof:<path>") to indicate this.
269      * Other component (e.g. `.granite-collection-navigator`) reading the columnId can then understand this convention and handle it accordingly.
270      */
271     String columnId = "parentof:" + rootResource.getPath();
272 
273     Column column = new Column()
274         .columnId(columnId)
275         .hasMore(false);
276     column.addItem(new ColumnItem(rootResource)
277         .resourceType(itemResourceType)
278         .active(true));
279     return column;
280   }

282   /**
283    * Generate column for each ancestor.
284    * @param currentResource Current resource
285    * @param rootResource Root resource
286    * @return Columns
287    */
288   private static List<Column> getAncestorColumns(Resource currentResource, Resource rootResource) {
289     List<Column> columns = new ArrayList<>();
290     List<Resource> ancestors = getAncestors(currentResource, rootResource);
291     for (int i = 0; i < ancestors.size(); i++) {
292       Resource r = ancestors.get(i);
293 
294       String activeId;
295       if (i < ancestors.size() - 1) {
296         activeId = ancestors.get(i + 1).getPath();
297       }
298       else {
299         activeId = currentResource.getPath();
300       }
301 
302       Column column = new Column()
303           .columnId(r.getPath())
304           .lazy(true)
305           .activeId(activeId);
306       columns.add(column);
307     }
308     return columns;
309   }

311   /**
312    * Returns the ancestors of the current resource (inclusive) up to the root.
313    * The result is ordered with the root as the first item.
314    */
315   private static List<Resource> getAncestors(Resource currentResource, Resource rootResource) {
316     List<Resource> results = new ArrayList<>();
317 
318     if (currentResource == null || rootResource == null || StringUtils.equals(currentResource.getPath(), rootResource.getPath())) {
319       return results;
320     }
321 
322     Resource parent = currentResource.getParent();
323     while (parent != null) {
324       results.add(0, parent);
325       if (parent.getPath().equals(rootResource.getPath())) {
326         break;
327       }
328       parent = parent.getParent();
329     }
330     return results;
331   }
