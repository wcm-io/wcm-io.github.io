20  package io.wcm.wcm.ui.granite.pathfield.impl.predicate;
21  
22  import java.util.Set;
23  import java.util.regex.Pattern;
24  
25  import org.apache.commons.collections.Predicate;
26  import org.apache.commons.lang3.StringUtils;
27  import org.apache.sling.api.resource.Resource;
28  
29  import com.google.common.collect.ImmutableSet;
30  
31  /**
32   * Hide certain "AEM-internal" content paths when listing resource children.
33   */
34  public class HideInternalContentPathsPredicate implements Predicate {
35  
36    /**
37     * List of paths that are hidden by default.
38     * Please note: it's allowed to see them if they are selected as root path, but not visible as children.
39     */
40    private static final Set<String> HIDE_CONTENT_PATHS = ImmutableSet.<String>builder()
41        .add("/content/catalogs")
42        .add("/content/cq:tags")
43        .add("/content/campaigns")
44        .add("/content/communities")
45        .add("/content/community")
46        .add("/content/community-components")
47        .add("/content/community-templates")
48        .add("/content/dam")
49        .add("/content/dam/catalogs")
50        .add("/content/dam/collections")
51        .add("/content/dam/formsanddocuments")
52        .add("/content/dam/formsanddocuments-themes")
53        .add("/content/dam/templates")
54        .add("/content/entities")
55        .add("/content/experience-fragments")
56        .add("/content/forms")
57        .add("/content/launches")
58        .add("/content/mac")
59        .add("/content/mobileapps")
60        .add("/content/publications")
61        .add("/content/usergenerated")
62        .add("/content/phonegap")
63        .add("/content/projects")
64        .add("/content/screens")
65        .add("/content/sites")
66        .add("/content/versionhistory")
67        .build();
68  
69    private static final Pattern FIRST_LEVEL_PATH = Pattern.compile("^/[^/]+$");
70    private static final String CONTENT_ROOT_PATH = "/content";
71  
72    @Override
73    public boolean evaluate(Object object) {
74      // if resource is a resource on the first level (below root node), allow only /content
75      String path = ((Resource)object).getPath();
76      if (FIRST_LEVEL_PATH.matcher(path).matches()) {
77        return StringUtils.equals(path, CONTENT_ROOT_PATH);
78      }
79      // allow all other paths except those from the list above
80      return !HIDE_CONTENT_PATHS.contains(path);
81    }
82  
83  }
