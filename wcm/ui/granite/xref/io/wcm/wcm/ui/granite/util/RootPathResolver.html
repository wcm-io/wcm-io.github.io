20  package io.wcm.wcm.ui.granite.util;
21  
22  import java.util.HashMap;
23  import java.util.Map;
24  
25  import org.apache.commons.lang3.StringUtils;
26  import org.apache.sling.api.SlingHttpServletRequest;
27  import org.apache.sling.api.resource.ResourceResolver;
28  import org.jetbrains.annotations.NotNull;
29  import org.osgi.annotation.versioning.ProviderType;
30  
31  import com.adobe.granite.ui.components.ComponentHelper;
32  import com.adobe.granite.ui.components.Config;
33  import com.adobe.granite.ui.components.ExpressionHelper;
34  import com.day.text.Text;
35  
36  /**
37   * Helper class for path-based GraniteUI components to resolve the root path.
38   * <p>
39   * Resolution order for root path detection:
40   * </p>
41   * <ul>
42   * <li>Reads configured root path from <code>rootPath</code> property</li>
43   * <li>Calls the provided root path detector implementation to detect root path from context</li>
44   * <li>Reads fallback root path from <code>fallbackRootPath</code> property</li>
45   * <li>Uses fallback root path provided for this instance</li>
46   * <li>Fallback to "/"</li>
47   * </ul>
48   * <p>
49   * Additionally the root path is modified:
50   * </p>
51   * <ul>
52   * <li>If an <code>appendPath</code> property is configured it is appended to the detected root path</li>
53   * <li>Than it is checked if the root path is valid - if not the next-valid parent path is returned</li>
54   * </ul>
55   */
56  @ProviderType
57  public final class RootPathResolver {
58  
59    static final String PN_ROOT_PATH = "rootPath";
60    static final String PN_APPEND_PATH = "appendPath";
61    static final String PN_FALLBACK_PATH = "fallbackRootPath";
62    static final String DEFAULT_FALLBACK_ROOT_PATH = "/";
63  
64    private final ComponentHelper cmp;
65    private final Config cfg;
66    private final ExpressionHelper ex;
67    private final SlingHttpServletRequest request;
68    private final ResourceResolver resourceResolver;
69  
70    private RootPathDetector rootPathDetector;
71    private String fallbackRootPath = DEFAULT_FALLBACK_ROOT_PATH;
72  
73    /**
74     * Creates a new RootPathResolver instance.
75     * @param cmp Component helper
76     * @param request Request
77     */
78    public RootPathResolver(@NotNull ComponentHelper cmp, @NotNull SlingHttpServletRequest request) {
79      this.cmp = cmp;
80      this.cfg = cmp.getConfig();
81      this.ex = cmp.getExpressionHelper();
82      this.request = request;
83      this.resourceResolver = request.getResourceResolver();
84    }
85  
86    /**
87     * Sets the root path detector implementation to detect root path from context.
88     * @param rootPathDetector For detecting root path from context
89     */
90    public void setRootPathDetector(@NotNull RootPathDetector rootPathDetector) {
91      this.rootPathDetector = rootPathDetector;
92    }
93  
94    /**
95     * Sets the fallback root path to be used if none is configured.
96     * @param fallbackRootPath Fallback root path that is used if none is configured
97     */
98    public void setFallbackRootPath(@NotNull String fallbackRootPath) {
99      this.fallbackRootPath = fallbackRootPath;
100   }
101 
102   /**
103    * Get the resolved and validated root path.
104    * @return Root path.
105    */
106   public @NotNull String get() {
107     // get configured or detected or fallback root path
108     String rootPath = getRootPath();
109 
110     // append path if configured
111     rootPath = appendPath(rootPath);
112 
113     // resolve to existing path
114     return getExistingPath(rootPath);
115   }
116 
117   /**
118    * Get map of override properties for super component based on the wcm.io Granite UI Extensions PathField.
119    * @return Path properties
120    */
121   public Map<String, Object> getOverrideProperties() {
122     Map<String, Object> props = new HashMap<>();
123     props.put(PN_ROOT_PATH, get());
124     props.put(PN_APPEND_PATH, "");
125     props.put(PN_FALLBACK_PATH, "");
126     return props;
127   }
128 
129   /**
130    * @return Configured or detected root path or fallback path
131    */
132   private @NotNull String getRootPath() {
133 
134     // check for configured root path
135     String rootPath = ex.getString(cfg.get(PN_ROOT_PATH, String.class));
136     if (StringUtils.isNotBlank(rootPath)) {
137       return rootPath;
138     }
139 
140     // call root path detector
141     if (rootPathDetector != null) {
142       rootPath = rootPathDetector.detectRootPath(cmp, request);
143       if (rootPath != null && StringUtils.isNotBlank(rootPath)) {
144         return rootPath;
145       }
146     }
147 
148     // check for configured fallback path
149     rootPath = ex.getString(cfg.get(PN_FALLBACK_PATH, String.class));
150     if (StringUtils.isNotBlank(rootPath)) {
151       return rootPath;
152     }
153 
154     // fallback to default fallback path
155     return fallbackRootPath;
156   }
157 
158   /**
159    * Appends the "appendPath" if configured.
160    * @param rootPath Root path
161    * @return Path with appendix
162    */
163   private @NotNull String appendPath(@NotNull String rootPath) {
164     String appendPath = ex.getString(cfg.get(PN_APPEND_PATH, String.class));
165     if (StringUtils.isBlank(appendPath)) {
166       return rootPath;
167     }
168     StringBuilder combinedPath = new StringBuilder(rootPath);
169     if (!StringUtils.startsWith(appendPath, "/")) {
170       combinedPath.append("/");
171     }
172     combinedPath.append(appendPath);
173     return combinedPath.toString();
174   }
175 
176   /**
177    * Make sure the root path exists. If it does not exist go up to parent hierarchy until it returns an
178    * existing resource path.
179    */
180   @NotNull
181   String getExistingPath(@NotNull String rootPath) {
182     if (resourceResolver.getResource(rootPath) == null) {
183       String parentPath = Text.getRelativeParent(rootPath, 1);
184       if (StringUtils.isBlank(parentPath)) {
185         return DEFAULT_FALLBACK_ROOT_PATH;
186       }
187       return getExistingPath(parentPath);
188     }
189     else {
190       return rootPath;
191     }
192   }
193 
194 }
