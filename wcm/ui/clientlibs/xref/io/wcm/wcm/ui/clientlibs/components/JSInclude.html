29  import org.apache.commons.lang3.StringUtils;

45  /**
46   * Include JavaScript client libraries with optional attributes for script tag.
47   */
48  @Model(adaptables = SlingHttpServletRequest.class)
49  @ProviderType
50  public class JSInclude {
51  
52    private static final Set<String> CROSSORIGIN_ALLOWED_VALUES = Set.of(
53        "anonymous", "use-credentials");
54    private static final Set<String> REFERRERPOLICY_ALLOWED_VALUES = Set.of(
55        "no-referrer", "no-referrer-when-downgrade", "origin", "origin-when-cross-origin",
56        "same-origin", "strict-origin", "strict-origin-when-cross-origin", "unsafe-url");
57    private static final Set<String> TYPE_ALLOWED_VALUES = Set.of(
58        "text/javascript", "module");
59  
60    @SlingObject
61    private SlingHttpServletRequest request;
62    @SlingObject
63    private ResourceResolver resourceResolver;
64    @OSGiService
65    private HtmlLibraryManager htmlLibraryManager;
66    @OSGiService
67    private XSSAPI xssApi;
68  
69    @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
70    private Object categories;
71    @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
72    private boolean async;
73    @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
74    private String crossorigin;
75    @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
76    private boolean defer;
77    @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
78    private String integrity;
79    @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
80    private boolean nomodule;
81    @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
82    private String nonce;
83    @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
84    private String referrerpolicy;
85    @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
86    private String type;
87    @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
88    private Object customAttributes;
89  
90    private String include;
91  
92    @PostConstruct
93    private void activate() {
94      // build include string
95      String[] categoryArray = IncludeUtil.toArray(categories);
96      if (categoryArray != null) {
97        List<String> libraryPaths = IncludeUtil.getLibraryUrls(htmlLibraryManager, resourceResolver,
98            categoryArray, LibraryType.JS);
99        if (!libraryPaths.isEmpty()) {
100         Map<String, String> attrs = validateAndBuildAttributes();
101         Map<String, String> customAttrs = IncludeUtil.getCustomAttributes(customAttributes);
102         this.include = buildIncludeString(libraryPaths, attrs, customAttrs);
103       }
104     }
105   }
106 
107   /**
108    * Validate attribute values from HTL script, escape them properly and build a map with all attributes
109    * for the resulting script tag(s).
110    * @return Map with attribute for script tag
111    */
112   private @NotNull Map<String, String> validateAndBuildAttributes() {
113     Map<String, String> attrs = new HashMap<>();
114     if (async) {
115       attrs.put("async", null);
116     }
117     if (crossorigin != null && CROSSORIGIN_ALLOWED_VALUES.contains(crossorigin)) {
118       attrs.put("crossorigin", crossorigin);
119     }
120     if (defer) {
121       attrs.put("defer", null);
122     }
123     if (StringUtils.isNotEmpty(integrity)) {
124       attrs.put("integrity", xssApi.encodeForHTMLAttr(integrity));
125     }
126     if (nomodule) {
127       attrs.put("nomodule", null);
128     }
129     if (StringUtils.isNotEmpty(nonce)) {
130       attrs.put("nonce", xssApi.encodeForHTMLAttr(nonce));
131     }
132     if (referrerpolicy != null && REFERRERPOLICY_ALLOWED_VALUES.contains(referrerpolicy)) {
133       attrs.put("referrerpolicy", referrerpolicy);
134     }
135     if (type != null && TYPE_ALLOWED_VALUES.contains(type)) {
136       attrs.put("type", type);
137     }
138     return attrs;
139   }
140 
141   /**
142    * Build script tags for all client libraries with the defined custom script tag attributes set.
143    * @param libraryPaths Library paths
144    * @param attrs HTML attributes for script tag
145    * @param customAttrs Custom HTML attributes for script tag
146    * @return HTML markup with script tags
147    */
148   private @NotNull String buildIncludeString(@NotNull List<String> libraryPaths, @NotNull Map<String, String> attrs,
149       @NotNull Map<String, String> customAttrs) {
150     StringBuilder markup = new StringBuilder();
151     for (String libraryPath : libraryPaths) {
152       HtmlTagBuilder builder = new HtmlTagBuilder("script", true, xssApi);
153       builder.setAttrs(attrs);
154       builder.setAttrs(customAttrs);
155       builder.setAttr("src", IncludeUtil.appendRequestPath(libraryPath, request));
156       markup.append(builder.build());
157     }
158     return markup.toString();
159   }
160 
161   /**
162    * @return HTML markup with script tags
163    */
164   public @Nullable String getInclude() {
165     return include;
166   }
167 
168 }
