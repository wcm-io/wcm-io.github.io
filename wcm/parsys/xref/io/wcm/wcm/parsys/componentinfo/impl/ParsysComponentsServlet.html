53  /**
54   * Handles AJAX calls for updateComponentListHandler JS method to update list of allowed component lists dynamically.
55   */
56  @Component(service = Servlet.class, immediate = true, property = {
57      "sling.servlet.extensions=" + FileExtension.JSON,
58      "sling.servlet.selectors=" + ParsysComponentsServlet.SELECTOR,
59      "sling.servlet.resourceTypes=sling/servlet/default",
60      "sling.servlet.methods=" + HttpConstants.METHOD_GET
61  })
62  public final class ParsysComponentsServlet extends SlingSafeMethodsServlet {
63    private static final long serialVersionUID = 1L;
64  
65    static final String RP_PATH = "path";
66    static final String RP_RESOURCE_TYPE = "resourceType";
67    static final String SELECTOR = "wcmio-parsys-components";
68  
69    private static final Logger log = LoggerFactory.getLogger(ParsysComponentsServlet.class);
70  
71    @Reference
72    private AllowedComponentsProvider allowedComponentsProvider;
73  
74    @Reference
75    private InstanceTypeService instanceTypeService;
76  
77    private boolean enabled;
78  
79    @Activate
80    protected void activate() {
81      // Activate only in author mode
82      enabled = instanceTypeService.isAuthor();
83    }
84  
85    @SuppressWarnings({ "null", "deprecation" })
86    @Override
87    protected void doGet(SlingHttpServletRequest request, SlingHttpServletResponse response) throws ServletException, IOException {
88      if (!enabled) {
89        response.sendError(HttpServletResponse.SC_NOT_FOUND);
90        return;
91      }
92  
93      ResourceResolver resolver = request.getResourceResolver();
94      PageManager pageManager = AdaptTo.notNull(resolver, PageManager.class);
95      Page currentPage = pageManager.getContainingPage(request.getResource());
96  
97      if (currentPage == null) {
98        response.sendError(HttpServletResponse.SC_NOT_FOUND);
99        return;
100     }
101 
102     long startTime = 0;
103     if (log.isDebugEnabled()) {
104       startTime = System.currentTimeMillis();
105     }
106 
107     response.setContentType(ContentType.JSON);
108 
109     org.apache.sling.commons.json.JSONArray allowedComponents = new org.apache.sling.commons.json.JSONArray();
110 
111     String relativePath = RequestParam.get(request, RP_PATH);
112     String resourceType = RequestParam.get(request, RP_RESOURCE_TYPE);
113     if (StringUtils.isNotEmpty(relativePath)) {
114       // get resource from paragraph system
115       Set<String> allowed = allowedComponentsProvider.getAllowedComponents(currentPage, relativePath, resourceType, resolver);
116 
117       // create set with relative resource type paths
118       Set<String> allowedComponentsRelative = new TreeSet<>();
119       for (String allowedResourceType : allowed) {
120         allowedComponentsRelative.add(ResourceType.makeAbsolute(allowedResourceType, resolver));
121       }
122 
123       allowedComponents = new org.apache.sling.commons.json.JSONArray(allowedComponentsRelative);
124     }
125 
126     response.getWriter().write(allowedComponents.toString());
127 
128     // output profiling info in DEBUG mode
129     if (log.isDebugEnabled()) {
130       long endTime = System.currentTimeMillis();
131       long duration = endTime - startTime;
132       log.debug("ParsysComponentsServlet for {}/{} took {}ms", currentPage.getPath(), relativePath, duration);
133     }
134   }
