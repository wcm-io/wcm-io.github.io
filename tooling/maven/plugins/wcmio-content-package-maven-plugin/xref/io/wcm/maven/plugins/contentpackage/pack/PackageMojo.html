218    * the packaging will fail.
219    * <p>
220    * Possible values
221    * </p>
222    * <ul>
223    * <li><code>ignore</code>: Ignores the packaged access control and leaves the target unchanged.</li>
224    * <li><code>overwrite</code>: Applies the access control provided with the package to the target. this also removes
225    * existing access control.</li>
226    * <li><code>merge</code>: Merge access control provided with the package with the one in the content by replacing the

229    * <li><code>merge_preserve</code>: Merge access control in the content with the one provided with the package by
230    * adding the access control entries of principals not present in the content (i.e. content first). It never alters

399   @Override
400   @SuppressFBWarnings({ "RV_RETURN_VALUE_IGNORED_BAD_PRACTICE", "REC_CATCH_EXCEPTION" })
401   public void execute() throws MojoExecutionException, MojoFailureException {
402 
403     // show deprecation warning
404     getLog().warn("Please switch to Jackrabbit filevault-package-maven-plugin - see https://wcm-io.atlassian.net/wiki/x/GYB1BQ");
405 
406     try {
407       final File finalFile = new File(outputDirectory, finalName + PACKAGE_EXT);
408       final File vaultFolder = new File(workDirectory, META_DIR);
409       final File vaultDefinitionFolder = new File(vaultFolder, DEFINITION_FOLDER);
410       vaultFolder.mkdirs();
411       vaultDefinitionFolder.mkdirs();
412 
413       JcrContentPackageArchiver jcrContentPackageArchiver = new JcrContentPackageArchiver();
414       Map<String, File> additionalFiles = new HashMap<>();
415 
416       // get filter definition from file system
417       DefaultWorkspaceFilter filter = null;
418       if (filterSource != null && filterSource.exists() && !filterSource.isDirectory()) {
419         filter = loadFilter(filterSource);
420       }
421       else {
422         filter = loadFilterInFolder(vaultFolder);
423       }
424 
425       // merge with filters applied via plugin properties
426       if (this.filters != null) {
427         this.filters.merge(filter);
428       }
429 
430       obtainEmbeddedBundles(embeddeds, additionalFiles, filter);
431       obtainNestedPackages(subPackages, additionalFiles, filter);
432 
433       File filterFile = new File(vaultFolder, FILTER_XML);
434       // Reset the Filter to make sure the output is adjusted to any changes
435       filter.resetSource();
436       FileUtils.fileWrite(filterFile.getAbsolutePath(), filter.getSourceAsString());
437 
438       writePropertiesFile(vaultFolder);
439       checkAndCopy(vaultFolder, CONFIG_XML);
440       checkAndCopy(vaultFolder, SETTINGS_XML);
441       checkAndCopy(vaultFolder, PACKAGE_DEFINITION_XML);
442       if (thumbnailImage != null && thumbnailImage.exists()) {
443         FileUtils.copyFile(thumbnailImage, new File(vaultDefinitionFolder, THUMBNAIL_FILE));
444       }
445       jcrContentPackageArchiver.addDirectory(workDirectory);
446 
447       // add content from builtContentDirectory
448       if (builtContentDirectory.exists()) {
449         jcrContentPackageArchiver.addDirectory(builtContentDirectory, FileUtils.normalize(JCR_ROOT));
450       }
451 
452       // ensure that empty directories are included
453       jcrContentPackageArchiver.setIncludeEmptyDirs(true);
454       // Add additional files
455       for (Map.Entry<String, File> entry : additionalFiles.entrySet()) {
456         jcrContentPackageArchiver.addFile(entry.getValue(), entry.getKey());
457       }
458 
459       MavenArchiver mavenArchiver = new MavenArchiver();
460       mavenArchiver.setArchiver(jcrContentPackageArchiver);
461       mavenArchiver.setOutputFile(finalFile);
462       mavenArchiver.createArchive(null, project, getMavenArchiveConfiguration());
463 
464       final Artifact projectArtifact = project.getArtifact();
465       projectArtifact.setFile(finalFile);
466       projectArtifact.setArtifactHandler(artifactHandlerManager.getArtifactHandler(PACKAGE_TYPE));
467     }
468     /*CHECKSTYLE:OFF*/catch (Exception ex) { /*CHECKSTYLE:ON*/
469       throw new MojoExecutionException(ex.getMessage(), ex);
470     }
471   }

473   @SuppressWarnings("PMD.UseStringBufferForStringAppends")
474   private void obtainEmbeddedBundles(List<EmbeddedBundle> embeddedBundleList, Map<String, File> resultList, DefaultWorkspaceFilter filter)
475       throws MojoFailureException {
476     for (EmbeddedBundle embeddedBundle : embeddedBundleList) {
477       final List<Artifact> artifacts = embeddedBundle.getMatchingArtifacts(project);
478       if (artifacts.isEmpty()) {
479         if (failOnMissingEmbed) {
480           throw new MojoFailureException(
481               "Embedded artifact: '" + embeddedBundle + "' but no dependency artifact found. Add the missing dependency or adjust the embedded definition.");
482         }
483         else {
484           getLog().warn("No matching artifacts found for '" + embeddedBundle + "'");
485           continue;
486         }
487       }
488       if (embeddedBundle.getDestFileName() != null && artifacts.size() > 1) {
489         getLog().warn("Destination File Name defined but several artifacts match the '" + embeddedBundle + "'");
490       }
491 
492       String embeddedTargetPath = embeddedBundle.getTarget();
493       if (embeddedTargetPath == null) {
494         embeddedTargetPath = embeddedTarget;
495         if (embeddedTargetPath == null) {
496           embeddedTargetPath = "/apps/bundles/install/";
497           getLog().info(
498               "No target path set for '" + embeddedBundle + "'. Using default '" + embeddedTargetPath + "'");
499         }
500       }
501       embeddedTargetPath = resolvePath(embeddedTargetPath);
502 
503       embeddedTargetPath = JCR_ROOT + embeddedTargetPath;
504       embeddedTargetPath = FileUtils.normalize(embeddedTargetPath);
505 
506       getLog().info("Embedding Bundle '" + embeddedBundle + "'");
507       for (final Artifact artifact : artifacts) {
508         final File source = artifact.getFile();
509         String destinationFileName = embeddedBundle.getDestFileName();
510         if (destinationFileName == null) {
511           destinationFileName = source.getName();
512         }
513         final String targetPathName = embeddedTargetPath + destinationFileName;
514         resultList.put(targetPathName, source);
515         final String targetNodePathName = targetPathName.substring(JCR_ROOT.length() - 1);
516         if (embeddedBundle.isGenerateFilter()) {
517           filter.add(new PathFilterSet(targetNodePathName));
518         }
519       }
520     }
521   }

523   private void obtainNestedPackages(List<NestedPackage> nestedPackageList, Map<String, File> resultList, DefaultWorkspaceFilter filter) throws IOException {
524     for (NestedPackage nestedPackage : nestedPackageList) {
525       final List<Artifact> artifacts = nestedPackage.getMatchingArtifacts(project);
526       if (artifacts.isEmpty()) {
527         getLog().warn("No matching artifacts for nested package: '" + nestedPackage + "'");
528         continue;
529       }
530 
531       // get the package path
532       getLog().info("Adding Nested Package '" + nestedPackage + "'");
533       for (Artifact artifact : artifacts) {
534         final File source = artifact.getFile();
535 
536         // load properties
537         ZipFile zipFile = null;
538         InputStream in = null;
539         Properties props = new Properties();
540         try {
541           zipFile = new ZipFile(source, ZipFile.OPEN_READ);
542           ZipEntry zipEntry = zipFile.getEntry("META-INF/vault/properties.xml");
543           if (zipEntry == null) {
544             getLog().error("Package is invalid as it does not contain properties.xml");
545             throw new IOException("properties.xml missing in nested package: " + source.getName());
546           }
547           in = zipFile.getInputStream(zipEntry);
548           props.loadFromXML(in);
549         }
550         finally {
551           IOUtils.closeQuietly(in);
552           if (zipFile != null) {
553             zipFile.close();
554           }
555         }
556         PackageId pid = new PackageId(
557             props.getProperty("group"),
558             props.getProperty("name"),
559             props.getProperty("version"));
560         final String targetNodePathName = pid.getInstallationPath() + ".zip";
561         final String targetPathName = "jcr_root" + targetNodePathName;
562 
563         resultList.put(targetPathName, source);
564         getLog().info("Embedding " + artifact.getId() + " -> " + targetPathName);
565         if (nestedPackage.isGenerateFilter()) {
566           filter.add(new PathFilterSet(targetNodePathName));
567         }
568       }
569     }
570   }

572   /**
573    * Build Package Properties XML file.
574    * @param vaultFolder Folder in where the properties.xml file is written to
575    */
576   @SuppressFBWarnings("WMI_WRONG_MAP_ITERATOR")
577   private void writePropertiesFile(File vaultFolder)
578       throws IOException {
579     final DateFormat dateFormat = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss.SSSZ", Locale.US);
580     final Properties vaultProperties = new Properties();
581 
582     String description = project.getDescription();
583     if (description == null) {
584       description = project.getName();
585       if (description == null) {
586         description = project.getArtifactId();
587       }
588     }
589     vaultProperties.put("description", description);
590 
591     // Add User Values first
592     for (Object propertyKey : properties.keySet()) {
593       if (properties.get(propertyKey) == null) {
594         properties.put(propertyKey, "");
595       }
596     }
597 
598     vaultProperties.putAll(properties);
599 
600     // Package Descriptions
601     vaultProperties.put(PROPERTY_GROUP, group);
602     vaultProperties.put(PROPERTY_NAME, name);
603     vaultProperties.put(PROPERTY_VERSION, version);
604 
605     // Artifact Description
606     vaultProperties.put(PROPERTY_GROUP_ID, project.getGroupId());
607     vaultProperties.put(PROPERTY_ARTIFACT_ID, project.getArtifactId());
608 
609     // Package Dependencies
610     if (!dependencies.isEmpty()) {
611       vaultProperties.put(PROPERTY_DEPENDENCIES, Dependency.toString(dependencies));
612     }
613 
614     // User and Timestamp
615     if (!vaultProperties.containsKey(PROPERTY_CREATED_BY)) {
616       vaultProperties.put(PROPERTY_CREATED_BY, System.getProperty("user.name"));
617     }
618     vaultProperties.put(PROPERTY_CREATED, dateFormat.format(new Date()));
619 
620     // configurable properties
621     vaultProperties.put(PROPERTY_REQUIRES_ROOT, String.valueOf(requiresRoot));
622     vaultProperties.put(PROPERTY_ALLOW_INDEX_DEFINITIONS, String.valueOf(allowIndexDefinitions));
623     vaultProperties.put(PROPERTY_PATH, ETC_PACKAGES + "/" + group + "/" + name + PACKAGE_EXT);
624     vaultProperties.put(PROPERTY_AC_HANDLING, StringUtils.lowerCase(acHandling));
625 
626     try (FileOutputStream fos = new FileOutputStream(new File(vaultFolder, PROPERTIES_XML));
627         BufferedOutputStream bos = new BufferedOutputStream(fos)) {
628       vaultProperties.storeToXML(bos, project.getName());
629     }
630   }

632   private String resolvePath(final String path) {
633     String answer = path;
634     if (!answer.startsWith("/")) {
635       answer = "/" + path;
636       getLog().info("Relative path resolved to " + answer);
637     }
638 
639     return answer;
640   }

642   private File getFileFromFolder(File folder, final String fileName) {
643     File answer = null;
644     File[] files = folder.listFiles(
645         new FilenameFilter() {
646           @Override
647           public boolean accept(File dir, String itemName) {
648             return itemName.equals(fileName);
649           }
650         });
651     if (files != null && files.length > 0) {
652       answer = files[0];
653     }
654     return answer;
655   }

657   private void checkAndCopy(File vaultFolder, String fileName) throws IOException {
658     if (getFileFromFolder(vaultFolder, fileName) == null) {
659       InputStream ios = getClass().getResourceAsStream("/vault-file-templates/" + fileName);
660       if (ios != null) {
661         FileOutputStream fos = new FileOutputStream(new File(vaultFolder, fileName));
662         IOUtils.copy(ios, fos);
663         IOUtils.closeQuietly(ios);
664         IOUtils.closeQuietly(fos);
665       }
666     }
667   }

669   private DefaultWorkspaceFilter loadFilterInFolder(File vaultFolder)
670       throws IOException, ConfigurationException {
671     return loadFilter(
672         vaultFolder != null && vaultFolder.exists() && vaultFolder.isDirectory() ? new File(vaultFolder, FILTER_XML) : null);
673   }
674 
675   private DefaultWorkspaceFilter loadFilter(File filterFile)
676       throws IOException, ConfigurationException {
677     DefaultWorkspaceFilter answer = null;
678     InputStream in = null;
679     try {
680       if (filterFile != null && filterFile.exists() && !filterFile.isDirectory()) {
681         in = new FileInputStream(filterFile);
682       }
683       if (in != null) {
684         answer = new DefaultWorkspaceFilter();
685         answer.load(in);
686       }
687     }
688     finally {
689       if (in != null) {
690         IOUtils.closeQuietly(in);
691       }
692     }
693     if (answer == null) {
694       answer = new DefaultWorkspaceFilter();
695     }
696     return answer;
697   }
