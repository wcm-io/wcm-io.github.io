124   /**
125    * Bundle status JSON URL. If an URL is configured the activation status of all bundles in the system is checked
126    * before it is tried to upload and install a new package and after each upload.
127    *
128    * <p>
129    * If not all bundles are activated the upload is delayed up to {@link #bundleStatusWaitLimit} seconds,
130    * every 5 seconds the activation status is checked anew.
131    * </p>
132    *
133    * <p>
134    * Expected is an URL like: http://localhost:4502/system/console/bundles/.json
135    * </p>
136    *
137    * <p>
138    * If the URL is not set it is derived from serviceURL. If set to "-" the status check is disabled.
139    * </p>
140    */
141   @Parameter(property = "vault.bundleStatusURL", required = false)
142   private String bundleStatusURL;

144   /**

146    * If this limit is reached and the bundle status is still not positive the install of the package proceeds anyway.
147    */
148   @Parameter(property = "vault.bundleStatusWaitLimit", defaultValue = "360")
149   private int bundleStatusWaitLimit;

151   /**
152    * Package Manager install status JSON URL. If an URL is configured the installation status of packages and

155    * <p>

157    * seconds, every 5 seconds the installation status is checked anew.
158    * </p>
159    *
160    * <p>
161    * Expected is an URL like: http://localhost:4502/crx/packmgr/installstatus.jsp
162    * </p>
163    *
164    * <p>
165    * If the URL is not set it is derived from serviceURL. If set to "-" the status check is disabled.
166    * </p>

168   @Parameter(property = "vault.packageManagerInstallStatusURL", required = false)
169   private String packageManagerInstallStatusURL;

174    * anyway.

176   @Parameter(property = "vault.packageManagerInstallStatusWaitLimit", defaultValue = "360")
177   private int packageManagerInstallStatusWaitLimit;

179   /**
180    * Patterns for symbolic names of bundles that are expected to be not present in bundle list.

195           + "^org\\.apache\\.sling\\.jcr\\.webdav$,"
196           + "^org\\.apache\\.sling\\.jcr\\.davex$")

266   /**
267    * @return Package manager properties
268    * @throws MojoExecutionException If configuration is invalid
269    */
270   protected PackageManagerProperties getPackageManagerProperties() throws MojoExecutionException {
271     PackageManagerProperties props = new PackageManagerProperties();
272 
273     props.setPackageManagerUrl(buildPackageManagerUrl());
274     props.setUserId(this.userId);
275     props.setPassword(this.password);
276     props.setOAuth2AccessToken(this.oauth2AccessToken);
277     props.setConsoleUserId(this.consoleUserId);
278     props.setConsolePassword(this.consolePassword);
279     props.setConsoleOAuth2AccessToken(this.consoleOauth2AccessToken);
280     props.setRetryCount(this.retryCount);
281     props.setRetryDelaySec(this.retryDelay);
282     props.setBundleStatusUrl(buildBundleStatusUrl());
283     props.setBundleStatusWaitLimitSec(this.bundleStatusWaitLimit);
284     props.setBundleStatusBlacklistBundleNames(Arrays.asList(this.bundleStatusBlacklistBundleNames));
285     props.setBundleStatusWhitelistBundleNames(Arrays.asList(this.bundleStatusWhitelistBundleNames));
286     props.setSystemReadyUrl(buildSystemReadyUrl());
287     props.setSystemReadyWaitLimitSec(this.systemReadyWaitLimitSec);
288     props.setPackageManagerInstallStatusURL(buildPackageManagerInstallStatusUrl());
289     props.setPackageManagerInstallStatusWaitLimitSec(this.packageManagerInstallStatusWaitLimit);
290     props.setRelaxedSSLCheck(this.relaxedSSLCheck);
291     props.setHttpConnectTimeoutSec(this.httpConnectTimeoutSec);
292     props.setHttpSocketTimeoutSec(this.httpSocketTimeout);
293     props.setProxies(ProxySupport.getMavenProxies(session, decrypter));
294     props.setPackageManagerOutputLogLevel(this.packageManagerOutputLogLevel);
295 
296     return props;
297   }

299   private String buildPackageManagerUrl() throws MojoExecutionException {
300     String serviceUrl = this.serviceURL;
301     switch (VendorInstallerFactory.identify(serviceUrl)) {
302       case CRX:
303         serviceUrl = VendorInstallerFactory.getBaseUrl(serviceUrl) + CRX_URL;
304         break;
305       case COMPOSUM:
306         serviceUrl = VendorInstallerFactory.getBaseUrl(serviceUrl) + COMPOSUM_URL;
307         break;
308       default:
309         throw new MojoExecutionException("Unsupported service URL: " + serviceUrl);
310     }
311     return serviceUrl;
312   }
313 
314   private String buildBundleStatusUrl() throws MojoExecutionException {
315     if (Strings.CS.equals(this.bundleStatusURL, "-")) {
316       return null;
317     }
318     if (this.bundleStatusURL != null) {
319       return this.bundleStatusURL;
320     }
321     // if not set use hostname from serviceURL and add default path to bundle status
322     String baseUrl = VendorInstallerFactory.getBaseUrl(buildPackageManagerUrl());
323     return baseUrl + "/system/console/bundles/.json";
324   }
325 
326   private String buildSystemReadyUrl() throws MojoExecutionException {
327     if (Strings.CS.equals(this.systemReadyURL, "-")) {
328       return null;
329     }
330     if (this.systemReadyURL != null) {
331       return this.systemReadyURL;
332     }
333     // if not set use hostname from serviceURL and add default path to bundle status
334     String baseUrl = VendorInstallerFactory.getBaseUrl(buildPackageManagerUrl());
335     return baseUrl + "/systemready";
336   }
337 
338   private String buildPackageManagerInstallStatusUrl() throws MojoExecutionException {
339     if (Strings.CS.equals(this.packageManagerInstallStatusURL, "-")
340         || VendorInstallerFactory.identify(this.serviceURL) != Service.CRX) {
341       return null;
342     }
343     if (this.packageManagerInstallStatusURL != null) {
344       return this.packageManagerInstallStatusURL;
345     }
346     // if not set use hostname from serviceURL and add default path to bundle status
347     String baseUrl = VendorInstallerFactory.getBaseUrl(buildPackageManagerUrl());
348     return baseUrl + "/crx/packmgr/installstatus.jsp";
349   }
