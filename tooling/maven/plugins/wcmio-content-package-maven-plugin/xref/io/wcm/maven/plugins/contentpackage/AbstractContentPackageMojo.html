45    /**
46     * The name of the content package file to install on the target system.
47     * If not set, the primary artifact of the project is considered the content package to be installed.
48     */
49    @Parameter(property = "vault.file", defaultValue = "${project.build.directory}/${project.build.finalName}.zip")
50    private File packageFile;

52    /**
53     * The URL of the HTTP service API of the CRX package manager.
54     *
55     * <p>
56     * See <a href=
57     * "https://experienceleague.adobe.com/docs/experience-manager-65/administering/operations/curl.html?lang=en#package-management"
58     * >Package Manager HTTP API</a> for details on this interface.
59     * </p>
60     */
61    @Parameter(property = "vault.serviceURL", required = true, defaultValue = "http://localhost:4502/crx/packmgr/service")
62    private String serviceURL;

68    private String userId;

70    /**
71     * The password to authenticate against the remote CRX system (package manager).

73    @Parameter(property = "vault.password", required = true, defaultValue = "admin")
74    private String password;

76    /**

123   /**
124    * Bundle status JSON URL. If an URL is configured the activation status of all bundles in the system is checked
125    * before it is tried to upload and install a new package and after each upload.
126    *
127    * <p>
128    * If not all bundles are activated the upload is delayed up to {@link #bundleStatusWaitLimit} seconds,
129    * every 5 seconds the activation status is checked anew.
130    * </p>
131    *
132    * <p>
133    * Expected is an URL like: http://localhost:4502/system/console/bundles/.json
134    * </p>
135    *
136    * <p>
137    * If the URL is not set it is derived from serviceURL. If set to "-" the status check is disabled.
138    * </p>
139    */
140   @Parameter(property = "vault.bundleStatusURL", required = false)
141   private String bundleStatusURL;
142 
143   /**
144    * Number of seconds to wait as maximum for a positive bundle status check.
145    * If this limit is reached and the bundle status is still not positive the install of the package proceeds anyway.
146    */
147   @Parameter(property = "vault.bundleStatusWaitLimit", defaultValue = "360")
148   private int bundleStatusWaitLimit;
149 
150   /**
151    * Package Manager install status JSON URL. If an URL is configured the installation status of packages and
152    * embedded packages is checked before it is tried to upload and install a new package and after each upload.
153    *
154    * <p>
155    * If not all packages are installed the upload is delayed up to {@link #packageManagerInstallStatusWaitLimit}
156    * seconds, every 5 seconds the installation status is checked anew.
157    * </p>
158    *
159    * <p>
160    * Expected is an URL like: http://localhost:4502/crx/packmgr/installstatus.jsp
161    * </p>
162    *
163    * <p>
164    * If the URL is not set it is derived from serviceURL. If set to "-" the status check is disabled.
165    * </p>
166    */
167   @Parameter(property = "vault.packageManagerInstallStatusURL", required = false)
168   private String packageManagerInstallStatusURL;
169 
170   /**
171    * Number of seconds to wait as maximum for a positive package manager install status check.
172    * If this limit is reached and the package manager status is still not positive the install of the package proceeds
173    * anyway.
174    */
175   @Parameter(property = "vault.packageManagerInstallStatusWaitLimit", defaultValue = "360")
176   private int packageManagerInstallStatusWaitLimit;
177 
178   /**
179    * Patterns for symbolic names of bundles that are expected to be not present in bundle list.
180    * If any of these bundles are found in the bundle list, this system is assumed as not ready for installing further
181    * packages because a previous installation (e.g. of AEM service pack) is still in progress.
182    */
183   @Parameter(property = "vault.bundleStatusBlacklistBundleNames", defaultValue = "^updater\\.aem.*$")
184   private String[] bundleStatusBlacklistBundleNames;
185 
186   /**
187    * Patterns for symbolic names of bundles that are ignored in bundle list.
188    * The state of these bundles has no effect on the bundle status check.
189    */
190   @Parameter(property = "vault.bundleStatusWhitelistBundleNames",
191       defaultValue = "^com\\.day\\.crx\\.crxde-support$,"
192           + "^com\\.adobe\\.granite\\.crx-explorer$,"
193           + "^com\\.adobe\\.granite\\.crxde-lite$,"
194           + "^org\\.apache\\.sling\\.jcr\\.webdav$,"
195           + "^org\\.apache\\.sling\\.jcr\\.davex$")
196   private String[] bundleStatusWhitelistBundleNames;
197 
198   /**
199    * If set to true also self-signed certificates are accepted.

201   @Parameter(property = "vault.relaxedSSLCheck", defaultValue = "false")
202   private boolean relaxedSSLCheck;
203

237   protected PackageManagerProperties getPackageManagerProperties() throws MojoExecutionException {
238     PackageManagerProperties props = new PackageManagerProperties();
239 
240     props.setPackageManagerUrl(buildPackageManagerUrl());
241     props.setUserId(this.userId);
242     props.setPassword(this.password);
243     props.setOAuth2AccessToken(this.oauth2AccessToken);
244     props.setConsoleUserId(this.consoleUserId);
245     props.setConsolePassword(this.consolePassword);
246     props.setConsoleOAuth2AccessToken(this.consoleOauth2AccessToken);
247     props.setRetryCount(this.retryCount);
248     props.setRetryDelaySec(this.retryDelay);
249     props.setBundleStatusUrl(buildBundleStatusUrl());
250     props.setBundleStatusWaitLimitSec(this.bundleStatusWaitLimit);
251     props.setBundleStatusBlacklistBundleNames(Arrays.asList(this.bundleStatusBlacklistBundleNames));
252     props.setBundleStatusWhitelistBundleNames(Arrays.asList(this.bundleStatusWhitelistBundleNames));
253     props.setPackageManagerInstallStatusURL(buildPackageManagerInstallStatusUrl());
254     props.setPackageManagerInstallStatusWaitLimitSec(this.packageManagerInstallStatusWaitLimit);
255     props.setRelaxedSSLCheck(this.relaxedSSLCheck);
256     props.setHttpConnectTimeoutSec(this.httpConnectTimeoutSec);
257     props.setHttpSocketTimeoutSec(this.httpSocketTimeout);
258     props.setProxies(ProxySupport.getMavenProxies(session, decrypter));
259     props.setPackageManagerOutputLogLevel(this.packageManagerOutputLogLevel);
260 
261     return props;
262   }

264   private String buildPackageManagerUrl() throws MojoExecutionException {
265     String serviceUrl = this.serviceURL;
266     switch (VendorInstallerFactory.identify(serviceUrl)) {
267       case CRX:
268         serviceUrl = VendorInstallerFactory.getBaseUrl(serviceUrl) + CRX_URL;
269         break;
270       case COMPOSUM:
271         serviceUrl = VendorInstallerFactory.getBaseUrl(serviceUrl) + COMPOSUM_URL;
272         break;
273       default:
274         throw new MojoExecutionException("Unsupported service URL: " + serviceUrl);
275     }
276     return serviceUrl;
277   }
278 
279   private String buildBundleStatusUrl() throws MojoExecutionException {
280     if (StringUtils.equals(this.bundleStatusURL, "-")) {
281       return null;
282     }
283     if (this.bundleStatusURL != null) {
284       return this.bundleStatusURL;
285     }
286     // if not set use hostname from serviceURL and add default path to bundle status
287     String baseUrl = VendorInstallerFactory.getBaseUrl(buildPackageManagerUrl());
288     return baseUrl + "/system/console/bundles/.json";
289   }
290 
291   private String buildPackageManagerInstallStatusUrl() throws MojoExecutionException {
292     if (StringUtils.equals(this.packageManagerInstallStatusURL, "-")
293         || VendorInstallerFactory.identify(this.serviceURL) != Service.CRX) {
294       return null;
295     }
296     if (this.packageManagerInstallStatusURL != null) {
297       return this.packageManagerInstallStatusURL;
298     }
299     // if not set use hostname from serviceURL and add default path to bundle status
300     String baseUrl = VendorInstallerFactory.getBaseUrl(buildPackageManagerUrl());
301     return baseUrl + "/crx/packmgr/installstatus.jsp";
302   }
