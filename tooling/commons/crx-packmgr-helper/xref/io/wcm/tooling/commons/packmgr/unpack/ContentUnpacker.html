102   /**
103    * @param properties Configuration properties
104    */
105   public ContentUnpacker(ContentUnpackerProperties properties) {
106     this.excludeFiles = toPatternArray(properties.getExcludeFiles());
107     this.excludeNodes = toPatternArray(properties.getExcludeNodes());
108     this.excludeProperties = toPatternArray(properties.getExcludeProperties());
109     this.excludeMixins = toPatternArray(properties.getExcludeMixins());
110     this.markReplicationActivated = properties.isMarkReplicationActivated();
111     this.markReplicationActivatedIncludeNodes = toPatternArray(properties.getMarkReplicationActivatedIncludeNodes());
112 
113     if (StringUtils.isNotBlank(properties.getDateLastReplicated())) {
114       this.dateLastReplicated = properties.getDateLastReplicated();
115     }
116     else {
117       // set to current date
118       Calendar cal = Calendar.getInstance();
119       cal.set(Calendar.HOUR_OF_DAY, 0);
120       cal.set(Calendar.MINUTE, 0);
121       cal.set(Calendar.SECOND, 0);
122       cal.set(Calendar.MILLISECOND, 0);
123       this.dateLastReplicated = ISO8601.format(cal);
124     }
125   }

127   private static Pattern[] toPatternArray(String[] patternStrings) {
128     if (patternStrings == null) {
129       return new Pattern[0];
130     }
131     Pattern[] patterns = new Pattern[patternStrings.length];
132     for (int i = 0; i < patternStrings.length; i++) {
133       try {
134         patterns[i] = Pattern.compile(patternStrings[i]);
135       }
136       catch (PatternSyntaxException ex) {
137         throw new PackageManagerException("Invalid regexp pattern: " + patternStrings[i], ex);
138       }
139     }
140     return patterns;
141   }

143   private static boolean matches(String name, Pattern[] patterns, boolean defaultIfNotPatternsDefined) {
144     if (patterns.length == 0) {
145       return defaultIfNotPatternsDefined;
146     }
147     for (Pattern pattern : patterns) {
148       if (pattern.matcher(name).matches()) {
149         return true;
150       }
151     }
152     return false;
153   }

155   private boolean applyXmlExcludes(String name) {
156     if (this.excludeNodes.length == 0 && this.excludeProperties.length == 0) {
157       return false;
158     }
159     return isXmlFile(name);
160   }

162   private boolean isXmlFile(String name) {
163     return StringUtils.equalsIgnoreCase(FilenameUtils.getExtension(name), "xml");
164   }

166   /**
167    * Unpacks file
168    * @param file File
169    * @param outputDirectory Output directory
170    */
171   public void unpack(File file, File outputDirectory) {
172     try (ZipFile zipFile = new ZipFile.Builder().setFile(file).get()) {
173       Enumeration<ZipArchiveEntry> entries = zipFile.getEntries();
174       while (entries.hasMoreElements()) {
175         ZipArchiveEntry entry = entries.nextElement();
176         if (!matches(entry.getName(), excludeFiles, false)) {
177           unpackEntry(zipFile, entry, outputDirectory);
178         }
179       }
180     }
181     catch (IOException ex) {
182       throw new PackageManagerException("Error reading content package " + file.getAbsolutePath(), ex);
183     }
184   }
185 
186   @SuppressFBWarnings("RV_RETURN_VALUE_IGNORED_BAD_PRACTICE")
187   private void unpackEntry(ZipFile zipFile, ZipArchiveEntry entry, File outputDirectory) throws IOException {
188     if (entry.isDirectory()) {
189       File directory = FileUtils.getFile(outputDirectory, entry.getName());
190       directory.mkdirs();
191     }
192     else {
193       Set<String> namespacePrefixes = null;
194       if (applyXmlExcludes(entry.getName())) {
195         namespacePrefixes = getNamespacePrefixes(zipFile, entry);
196       }
197 
198       try (InputStream entryStream = zipFile.getInputStream(entry)) {
199         File outputFile = FileUtils.getFile(outputDirectory, entry.getName());
200         if (outputFile.exists()) {
201           outputFile.delete();
202         }
203         File directory = outputFile.getParentFile();
204         directory.mkdirs();
205 
206         try (FileOutputStream fos = new FileOutputStream(outputFile)) {
207           if (applyXmlExcludes(entry.getName()) && namespacePrefixes != null) {
208             // write file with XML filtering
209             try {
210               writeXmlWithExcludes(entry, entryStream, fos, namespacePrefixes);
211             }
212             catch (JDOMException ex) {
213               throw new PackageManagerException("Unable to parse XML file: " + entry.getName(), ex);
214             }
215           }
216           else {
217             // write file directly without XML filtering
218             IOUtils.copy(entryStream, fos);
219           }
220         }
221         if (isXmlFile(outputFile.getName())) {
222           // format output file using DocView format
223           try {
224             DOCVIEWFORMAT.format(outputFile, false);
225           }
226           catch (IOException ex) {
227             throw new IOException("Unable to apply DocView format to file: " + outputFile.getAbsolutePath(), ex);
228           }
229         }
230       }
231     }
232   }

234   /**
235    * Parses XML file with namespace-aware SAX parser to get defined namespaces prefixes in order of appearance
236    * (to keep the same order when outputting the XML file again).
237    * @param zipFile ZIP file
238    * @param entry ZIP entry
239    * @return Ordered set with namespace prefixes in correct order.
240    *         Returns null if given XML file does not contain FileVault XML content.
241    */
242   private Set<String> getNamespacePrefixes(ZipFile zipFile, ZipArchiveEntry entry) throws IOException {
243     try (InputStream entryStream = zipFile.getInputStream(entry)) {
244       SAXParser parser = SAX_PARSER_FACTORY.newSAXParser();
245       final Set<String> prefixes = new LinkedHashSet<>();
246 
247       final AtomicBoolean foundRootElement = new AtomicBoolean(false);
248       DefaultHandler handler = new DefaultHandler() {
249         @Override
250         public void startElement(String uri, String localName, String qName, Attributes attributes) throws SAXException {
251           // validate that XML file contains FileVault XML content
252           if (StringUtils.equals(uri, JCR_NAMESPACE.getURI()) && StringUtils.equals(localName, "root")) {
253             foundRootElement.set(true);
254           }
255         }
256         @Override
257         public void startPrefixMapping(String prefix, String uri) throws SAXException {
258           if (StringUtils.isNotBlank(prefix)) {
259             prefixes.add(prefix);
260           }
261         }
262       };
263       parser.parse(entryStream, handler);
264 
265       if (!foundRootElement.get()) {
266         return null;
267       }
268       else {
269         return prefixes;
270       }
271     }
272     catch (IOException | SAXException | ParserConfigurationException ex) {
273       throw new IOException("Error parsing " + entry.getName(), ex);
274     }
275   }

277   private void writeXmlWithExcludes(ZipArchiveEntry entry, InputStream inputStream, OutputStream outputStream, Set<String> namespacePrefixes)
278       throws IOException, JDOMException {
279     SAXBuilder saxBuilder = new SAXBuilder();
280     saxBuilder.setProperty(XMLConstants.ACCESS_EXTERNAL_DTD, "");
281     saxBuilder.setProperty(XMLConstants.ACCESS_EXTERNAL_SCHEMA, "");
282     Document doc = saxBuilder.build(inputStream);
283 
284     Set<String> namespacePrefixesActuallyUsed = new HashSet<>();
285 
286     // check for namespace prefix in file name
287     String namespacePrefix = getNamespacePrefix(entry.getName());
288     if (namespacePrefix != null) {
289       namespacePrefixesActuallyUsed.add(namespacePrefix);
290     }
291 
292     applyXmlExcludes(doc.getRootElement(), getParentPath(entry), namespacePrefixesActuallyUsed, false);
293 
294     XMLOutputter outputter = new XMLOutputter(Format.getPrettyFormat()
295         .setIndent("    ")
296         .setLineSeparator(LineSeparator.UNIX));
297     outputter.setXMLOutputProcessor(new NamspaceOrderedXmlProcessor(namespacePrefixes, namespacePrefixesActuallyUsed));
298     outputter.output(doc, outputStream);
299     outputStream.flush();
300   }

302   static String getNamespacePrefix(String path) {
303     String fileName = FilenameUtils.getName(path);
304     if (StringUtils.equals(DOT_CONTENT_XML, fileName)) {
305       String parentFolderName = FilenameUtils.getName(FilenameUtils.getPathNoEndSeparator(path));
306       if (parentFolderName != null) {
307         String nodeName = PlatformNameFormat.getRepositoryName(parentFolderName);
308         Matcher matcher = FILENAME_NAMESPACE_PATTERN.matcher(nodeName);
309         if (matcher.matches()) {
310           return matcher.group(1);
311         }
312       }
313     }
314     return null;
315   }

317   private String getParentPath(ZipArchiveEntry entry) {
318     return StringUtils.removeEnd(StringUtils.removeStart(entry.getName(), ROOT_DIR), "/" + DOT_CONTENT_XML);
319   }

321   private String buildElementPath(Element element, String parentPath) {
322     StringBuilder path = new StringBuilder(parentPath);
323     if (!StringUtils.equals(element.getQualifiedName(), "jcr:root")) {
324       path.append("/").append(element.getQualifiedName());
325     }
326     return path.toString();
327   }

329   @SuppressWarnings("PMD.EmptyControlStatement")
330   private void applyXmlExcludes(Element element, String parentPath, Set<String> namespacePrefixesActuallyUsed,
331       boolean insideReplicationElement) {
332     String path = buildElementPath(element, parentPath);
333     if (matches(path, this.excludeNodes, false)) {
334       element.detach();
335       return;
336     }
337     collectNamespacePrefix(namespacePrefixesActuallyUsed, element.getNamespacePrefix());
338 
339     String jcrPrimaryType = element.getAttributeValue("primaryType", JCR_NAMESPACE);
340     boolean isRepositoryUserGroup = StringUtils.equals(jcrPrimaryType, "rep:User") || StringUtils.equals(jcrPrimaryType, "rep:Group");
341     boolean isReplicationElement = StringUtils.equals(jcrPrimaryType, "cq:Page")
342         || StringUtils.equals(jcrPrimaryType, "dam:Asset")
343         || StringUtils.equals(jcrPrimaryType, "cq:Template");
344     boolean isContent = insideReplicationElement && StringUtils.equals(element.getQualifiedName(), "jcr:content");
345     boolean setReplicationAttributes = isContent && markReplicationActivated;
346 
347     List<Attribute> attributes = new ArrayList<>(element.getAttributes());
348     for (Attribute attribute : attributes) {
349       boolean excluded = false;
350       if (matches(attribute.getQualifiedName(), this.excludeProperties, false)) {
351         if (isRepositoryUserGroup && StringUtils.equals(attribute.getQualifiedName(), JcrConstants.JCR_UUID)) {
352           // keep jcr:uuid property for groups and users, otherwise they cannot be imported again
353         }
354         else {
355           attribute.detach();
356           excluded = true;
357         }
358       }
359       else if (StringUtils.equals(attribute.getQualifiedName(), PRIMARYTYPE_PROPERTY)) {
360         String namespacePrefix = StringUtils.substringBefore(attribute.getValue(), ":");
361         collectNamespacePrefix(namespacePrefixesActuallyUsed, namespacePrefix);
362       }
363       else if (StringUtils.equals(attribute.getQualifiedName(), MIXINS_PROPERTY)) {
364         String filteredValue = filterMixinsPropertyValue(attribute.getValue(), namespacePrefixesActuallyUsed);
365         if (StringUtils.isBlank(filteredValue)) {
366           attribute.detach();
367         }
368         else {
369           attribute.setValue(filteredValue);
370         }
371       }
372       else if (StringUtils.startsWith(attribute.getValue(), "{Name}")) {
373         collectNamespacePrefixNameArray(namespacePrefixesActuallyUsed, attribute.getValue());
374         // alphabetically sort name values
375         attribute.setValue(sortReferenceValues(attribute.getValue(), PropertyType.NAME));
376       }
377       else if (StringUtils.startsWith(attribute.getValue(), "{WeakReference}")) {
378         // alphabetically sort weak reference values
379         attribute.setValue(sortReferenceValues(attribute.getValue(), PropertyType.WEAKREFERENCE));
380       }
381       if (!excluded) {
382         collectNamespacePrefix(namespacePrefixesActuallyUsed, attribute.getNamespacePrefix());
383       }
384     }
385 
386     // set replication status for jcr:content nodes inside cq:Page nodes
387     if (setReplicationAttributes && matches(path, markReplicationActivatedIncludeNodes, true)) {
388       addMixin(element, "cq:ReplicationStatus");
389       element.setAttribute("lastReplicated", "{Date}" + dateLastReplicated, CQ_NAMESPACE);
390       element.setAttribute("lastReplicationAction", "Activate", CQ_NAMESPACE);
391       collectNamespacePrefix(namespacePrefixesActuallyUsed, CQ_NAMESPACE.getPrefix());
392     }
393 
394     // if current element is a replication element, but the jcr:content node to set the replication attributes to is missing, add it
395     if (isReplicationElement && element.getChild("content", JCR_NAMESPACE) == null
396         && matches(path + "/jcr:content", markReplicationActivatedIncludeNodes, true)) {
397       Element contentNode = new Element("content", JCR_NAMESPACE);
398       String jcrContentPrimaryType = StringUtils.equals(jcrPrimaryType, "cq:Template") ? "cq:PageContent" : jcrPrimaryType + "Content";
399       contentNode.setAttribute("primaryType", jcrContentPrimaryType, JCR_NAMESPACE);
400       element.addContent(contentNode);
401     }
402 
403     List<Element> children = new ArrayList<>(element.getChildren());
404     for (Element child : children) {
405       applyXmlExcludes(child, path, namespacePrefixesActuallyUsed, (insideReplicationElement || isReplicationElement) && !isContent);
406     }
407   }

409   private String filterMixinsPropertyValue(String value, Set<String> namespacePrefixesActuallyUsed) {
410     if (this.excludeMixins.length == 0 || StringUtils.isBlank(value)) {
411       return value;
412     }
413 
414     List<String> mixins = new ArrayList<>();
415     for (String mixin : DocViewUtil.parseValues(value)) {
416       if (!matches(mixin, this.excludeMixins, false)) {
417         String namespacePrefix = StringUtils.substringBefore(mixin, ":");
418         collectNamespacePrefix(namespacePrefixesActuallyUsed, namespacePrefix);
419         mixins.add(mixin);
420       }
421     }
422 
423     if (mixins.isEmpty()) {
424       return null;
425     }
426 
427     return DocViewUtil.formatValues(mixins);
428   }
429 
430   private void addMixin(Element element, String mixin) {
431     String mixinsString = element.getAttributeValue("mixinTypes", JCR_NAMESPACE);
432 
433     List<String> mixins = new ArrayList<>();
434     if (!StringUtils.isBlank(mixinsString)) {
435       for (String item : DocViewUtil.parseValues(mixinsString)) {
436         mixins.add(item);
437       }
438     }
439     if (!mixins.contains(mixin)) {
440       mixins.add(mixin);
441     }
442 
443     element.setAttribute("mixinTypes", DocViewUtil.formatValues(mixins), JCR_NAMESPACE);
444   }
445 
446   private void collectNamespacePrefix(Set<String> prefixes, String prefix) {
447     if (StringUtils.isNotBlank(prefix)) {
448       prefixes.add(prefix);
449     }
450   }
451 
452   private void collectNamespacePrefixNameArray(Set<String> prefixes, String value) {
453     for (String item : DocViewUtil.parseValues(value)) {
454       String namespacePrefix = StringUtils.substringBefore(item, ":");
455       collectNamespacePrefix(prefixes, namespacePrefix);
456     }
457   }

459   /**
460    * Sort weak reference values alphabetically to ensure consistent ordering.
461    * @param value Property value
462    * @param propertyType Property type from {@link PropertyType}
463    * @return Property value with sorted references
464    */
465   private String sortReferenceValues(String value, int propertyType) {
466     Set<String> refs = new TreeSet<>();
467     for (String item : DocViewUtil.parseValues(value)) {
468       refs.add(item);
469     }
470     return DocViewUtil.formatValues(new ArrayList<>(refs), propertyType);
471   }
