20  package io.wcm.tooling.commons.packmgr.install.crx;
21  
22  import java.util.HashMap;
23  import java.util.Map;
24  
25  import org.apache.commons.lang3.StringUtils;
26  import org.apache.commons.lang3.Strings;
27  import org.json.JSONArray;
28  import org.json.JSONException;
29  import org.json.JSONObject;
30  
31  import io.wcm.tooling.commons.packmgr.PackageManagerException;
32  
33  /**
34   * Check if given package and version is already installed by inspecting the JSON result of
35   * <code>/crx/packmgr/list.jsp</code>.
36   */
37  final class PackageInstalledChecker {
38  
39    private final JSONArray results;
40  
41    static final String PACKMGR_LIST_URL = "/crx/packmgr/list.jsp";
42  
43    private static final long NOT_FOUND_DATE = -1;
44    private static final long NOT_UNPACKED_DATE = 0;
45  
46    PackageInstalledChecker(JSONObject result) {
47      try {
48        this.results = result.getJSONArray("results");
49      }
50      catch (JSONException ex) {
51        throw new PackageManagerException("JSON response from " + PACKMGR_LIST_URL + " does not contain 'results' array.", ex);
52      }
53    }
54  
55    public PackageInstalledStatus getStatus(String group, String name, String version) {
56      Map<String, Long> map = getVersionUnpackedDates(group, name);
57  
58      if (map.isEmpty()) {
59        return PackageInstalledStatus.NOT_FOUND;
60      }
61  
62      long versionUnpackDate = getVersionUnpackDate(map, version);
63      if (versionUnpackDate == NOT_FOUND_DATE) {
64        return PackageInstalledStatus.NOT_FOUND;
65      }
66      if (versionUnpackDate == NOT_UNPACKED_DATE) {
67        return PackageInstalledStatus.UPLOADED;
68      }
69  
70      long lastUnpackDate = getLastUnpackDate(map);
71      if (lastUnpackDate > versionUnpackDate) {
72        return PackageInstalledStatus.INSTALLED_OTHER_VERSION;
73      }
74      else {
75        return PackageInstalledStatus.INSTALLED;
76      }
77    }
78  
79    private Map<String, Long> getVersionUnpackedDates(String group, String name) {
80      Map<String, Long> map = new HashMap<>();
81      for (int i = 0; i < results.length(); i++) {
82        JSONObject item = results.getJSONObject(i);
83        String itemGroup = item.optString("group");
84        String itemName = item.optString("name");
85        String itemVersion = item.optString("version");
86        long itemLastUnpacked = item.optLong("lastUnpacked", NOT_UNPACKED_DATE);
87        if (Strings.CS.equals(group, itemGroup) && Strings.CS.equals(name, itemName) && StringUtils.isNotBlank(itemVersion)) {
88          map.put(itemVersion, itemLastUnpacked);
89        }
90      }
91      return map;
92    }
93  
94    private long getVersionUnpackDate(Map<String, Long> map, String version) {
95      Long value = map.get(version);
96      if (value == null) {
97        return NOT_FOUND_DATE;
98      }
99      else {
100       return value;
101     }
102   }
103 
104   private long getLastUnpackDate(Map<String, Long> map) {
105     long last = 0;
106     for (Long date : map.values()) {
107       if (date > last) {
108         last = date;
109       }
110     }
111     return last;
112   }
113 
114 }
