20  package io.wcm.tooling.commons.packmgr.download.crx;
21  
22  import static io.wcm.tooling.commons.packmgr.PackageManagerHelper.CRX_PACKAGE_EXISTS_ERROR_MESSAGE_PREFIX;
23  
24  import java.io.File;
25  
26  import org.apache.commons.lang3.StringUtils;
27  import org.apache.commons.lang3.Strings;
28  import org.apache.http.client.methods.HttpPost;
29  import org.apache.http.entity.mime.MultipartEntityBuilder;
30  import org.json.JSONObject;
31  
32  import io.wcm.tooling.commons.packmgr.PackageManagerException;
33  import io.wcm.tooling.commons.packmgr.PackageManagerHelper;
34  import io.wcm.tooling.commons.packmgr.download.VendorPackageDownloader;
35  import io.wcm.tooling.commons.packmgr.install.VendorInstallerFactory;
36  
37  /**
38   * Vendor Downloader for AEM's CRX Package Manager
39   */
40  public class CrxPackageDownloader implements VendorPackageDownloader {
41  
42    @Override
43    public String uploadPackageDefinition(String packageManagerUrl, File file, PackageManagerHelper pkgmgr) {
44      HttpPost post = new HttpPost(packageManagerUrl + "/.json?cmd=upload");
45      MultipartEntityBuilder entity = MultipartEntityBuilder.create()
46          .addBinaryBody("package", file)
47          .addTextBody("force", "true");
48      post.setEntity(entity.build());
49      JSONObject jsonResponse = pkgmgr.executePackageManagerMethodJson(pkgmgr.getHttpClient(), pkgmgr.getPackageManagerHttpClientContext(), post);
50      boolean success = jsonResponse.optBoolean("success", false);
51      String msg = jsonResponse.optString("msg", null);
52      String packagePath = jsonResponse.optString("path", null);
53      // package already exists - get path from error message and continue
54      if (!success && Strings.CS.startsWith(msg, CRX_PACKAGE_EXISTS_ERROR_MESSAGE_PREFIX) && StringUtils.isEmpty(packagePath)) {
55        packagePath = StringUtils.substringAfter(msg, CRX_PACKAGE_EXISTS_ERROR_MESSAGE_PREFIX);
56        success = true;
57      }
58      if (!success) {
59        throw new PackageManagerException("Package path detection failed: " + msg);
60      }
61  
62      return packagePath;
63    }
64  
65    @Override
66    public HttpPost createRebuildMethod(String packagePath, String packageManagerUrl) {
67      return new HttpPost(packageManagerUrl + "/console.html" + packagePath + "?cmd=build");
68    }
69  
70    @Override
71    public String createDownloadZipBaseUrl(String packageManagerUrl) {
72      return VendorInstallerFactory.getBaseUrl(packageManagerUrl);
73    }
74  
75  }
