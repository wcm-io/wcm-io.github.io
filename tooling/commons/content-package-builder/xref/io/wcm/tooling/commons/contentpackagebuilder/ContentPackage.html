63  /**
64   * Represents an AEM content package.
65   * Content like structured JCR data and binary files can be added.
66   * This class is not thread-safe.
67   */
68  public final class ContentPackage implements Closeable {
69  
70    private final PackageMetadata metadata;
71    private final ZipOutputStream zip;
72    private final Transformer transformer;
73    private final XmlContentBuilder xmlContentBuilder;
74  
75    private static final String CONTENT_TYPE_CHARSET_EXTENSION = ";charset=";
76    private static final String DOT_DIR_FOLDER = ".dir";
77  
78    /**
79     * @param os Output stream
80     */
81    ContentPackage(PackageMetadata metadata, OutputStream os) throws IOException {
82      this.metadata = metadata;
83      this.zip = new ZipOutputStream(os);
84  
85      TransformerFactory transformerFactory = TransformerFactory.newInstance();
86      transformerFactory.setAttribute(XMLConstants.ACCESS_EXTERNAL_DTD, "");
87      transformerFactory.setAttribute(XMLConstants.ACCESS_EXTERNAL_STYLESHEET, "");
88      try {
89        transformerFactory.setAttribute("indent-number", 2);
90      }
91      catch (IllegalArgumentException ex) {
92        // Implementation does not support configuration property. Ignore.
93      }
94      try {
95        this.transformer = transformerFactory.newTransformer();
96        try {
97          this.transformer.setOutputProperty(OutputKeys.INDENT, "yes");
98        }
99        catch (IllegalArgumentException ex) {
100         // Implementation does not support output property. Ignore.
101       }
102       try {
103         this.transformer.setOutputProperty("{http://xml.apache.org/xslt}indent-amount", "2");
104       }
105       catch (IllegalArgumentException ex) {
106         // Implementation does not support output property. Ignore.
107       }
108     }
109     catch (TransformerException ex) {
110       throw new RuntimeException("Failed to set up XML transformer: " + ex.getMessage(), ex);
111     }
112 
113     this.xmlContentBuilder = new XmlContentBuilder(metadata.getXmlNamespaces());
114 
115     buildPackageMetadata();
116   }

118   /**
119    * Adds a page with given content. The "cq:Page/cq:PageContent envelope" is added automatically.
120    * @param path Full content path of page.
121    * @param content Hierarchy of content elements.
122    * @throws IOException I/O exception
123    */
124   public void addPage(String path, ContentElement content) throws IOException {
125     String fullPath = buildJcrPathForZip(path) + "/" + DOT_CONTENT_XML;
126     Document doc = xmlContentBuilder.buildPage(content);
127     writeXmlDocument(fullPath, doc);
128   }

144   /**
145    * Add some JCR content structure directly to the package.
146    * @param path Full content path of content root node.
147    * @param content Hierarchy of content elements.
148    * @throws IOException I/O exception
149    */
150   public void addContent(String path, ContentElement content) throws IOException {
151     String basePath = buildJcrPathForZip(path);
152     List<ContentPart> parts = ContentFolderSplitter.split(ContentElementConverter.toMap(content));
153     for (ContentPart part : parts) {
154       String fullPath = basePath + part.getPath() + "/" + DOT_CONTENT_XML;
155       Document doc = xmlContentBuilder.buildContent(part.getContent());
156       writeXmlDocument(fullPath, doc);
157     }
158   }

218   /**
219    * Adds a binary file with explicit mime type.
220    * @param path Full content path and file name of file
221    * @param inputStream Input stream with binary data
222    * @param contentType Mime type, optionally with ";charset=XYZ" extension
223    * @throws IOException I/O exception
224    */
225   public void addFile(String path, InputStream inputStream, String contentType) throws IOException {
226     String fullPath = buildJcrPathForZip(path);
227     writeBinaryFile(fullPath, inputStream);
228 
229     if (StringUtils.isNotEmpty(contentType)) {
230       String mimeType = StringUtils.substringBefore(contentType, CONTENT_TYPE_CHARSET_EXTENSION);
231       String encoding = StringUtils.substringAfter(contentType, CONTENT_TYPE_CHARSET_EXTENSION);
232 
233       String fullPathMetadata = fullPath + DOT_DIR_FOLDER + "/" + DOT_CONTENT_XML;
234       Document doc = xmlContentBuilder.buildNtFile(mimeType, encoding);
235       writeXmlDocument(fullPathMetadata, doc);
236     }
237   }

239   /**
240    * If path parts contain namespace definitions they need to be escaped for the ZIP file.
241    * Example: oak:index -> jcr_root/_oak_index
242    * @param path Path
243    * @return Safe path
244    */
245   @SuppressWarnings("PMD.UseStringBufferForStringAppends")
246   static String buildJcrPathForZip(final String path) {
247     String normalizedPath = StringUtils.defaultString(path);
248     if (!normalizedPath.startsWith("/")) {
249       normalizedPath = "/" + normalizedPath;
250     }
251     ensureValidPath(path);
252     return ROOT_DIR + PlatformNameFormat.getPlatformPath(normalizedPath);
253   }

288   /**
289    * Get root path of the package. This does only work if there is only one filter of the package.
290    * If they are more filters use {@link #getFilters()} instead.
291    * @return Root path of package
292    */
293   public String getRootPath() {
294     if (metadata.getFilters().size() == 1) {
295       return metadata.getFilters().get(0).getRootPath();
296     }
297     else {
298       throw new IllegalStateException("Content package has more than one package filter - please use getFilters().");
299     }
300   }

310   /**
311    * Build all package metadata files based on templates.
312    * @throws IOException I/O exception
313    */
314   private void buildPackageMetadata() throws IOException {
315     metadata.validate();
316     buildTemplatedMetadataFile(META_DIR + "/" + CONFIG_XML);
317     buildPropertiesFile(META_DIR + "/" + PROPERTIES_XML);
318     buildTemplatedMetadataFile(META_DIR + "/" + SETTINGS_XML);
319     buildTemplatedMetadataFile(META_DIR + "/" + PACKAGE_DEFINITION_XML);
320     writeXmlDocument(META_DIR + "/" + FILTER_XML, xmlContentBuilder.buildFilter(metadata.getFilters()));
321 
322     // package thumbnail
323     byte[] thumbnailImage = metadata.getThumbnailImage();
324     if (thumbnailImage != null) {
325       zip.putNextEntry(new ZipEntry(META_DIR + "/definition/thumbnail.png"));
326       try {
327         zip.write(thumbnailImage);
328       }
329       finally {
330         zip.closeEntry();
331       }
332     }
333   }

335   /**
336    * Read template file from classpath, replace variables and store it in the zip stream.
337    * @param path Path
338    * @throws IOException I/O exception
339    */
340   @SuppressWarnings("deprecation")
341   private void buildTemplatedMetadataFile(String path) throws IOException {
342     try (InputStream is = getClass().getResourceAsStream("/content-package-template/" + path)) {
343       String xmlContent = IOUtils.toString(is, StandardCharsets.UTF_8);
344       for (Map.Entry<String, Object> entry : metadata.getVars().entrySet()) {
345         xmlContent = StringUtils.replace(xmlContent, "{{" + entry.getKey() + "}}",
346             org.apache.commons.lang3.StringEscapeUtils.escapeXml10(entry.getValue().toString()));
347       }
348       zip.putNextEntry(new ZipEntry(path));
349       try {
350         zip.write(xmlContent.getBytes(StandardCharsets.UTF_8));
351       }
352       finally {
353         zip.closeEntry();
354       }
355     }
356   }

358   /**
359    * Build java Properties XML file.
360    * @param path Path
361    * @throws IOException I/O exception
362    */
363   private void buildPropertiesFile(String path) throws IOException {
364     Properties properties = new Properties();
365     properties.put(PackageProperties.NAME_REQUIRES_ROOT, Boolean.toString(false));
366     properties.put("allowIndexDefinitions", Boolean.toString(false));
367 
368     for (Map.Entry<String, Object> entry : metadata.getVars().entrySet()) {
369       String value = Objects.toString(entry.getValue());
370       if (StringUtils.isNotEmpty(value)) {
371         properties.put(entry.getKey(), value);
372       }
373     }
374 
375     zip.putNextEntry(new ZipEntry(path));
376     try {
377       properties.storeToXML(zip, null);
378     }
379     finally {
380       zip.closeEntry();
381     }
382   }

384   /**
385    * Writes an XML document as binary file entry to the ZIP output stream.
386    * @param path Content path
387    * @param doc XML content
388    * @throws IOException I/O exception
389    */
390   private void writeXmlDocument(String path, Document doc) throws IOException {
391     zip.putNextEntry(new ZipEntry(path));
392     try {
393       DOMSource source = new DOMSource(doc);
394       StreamResult result = new StreamResult(zip);
395       transformer.transform(source, result);
396     }
397     catch (TransformerException ex) {
398       throw new IOException("Failed to generate XML: " + ex.getMessage(), ex);
399     }
400     finally {
401       zip.closeEntry();
402     }
403   }

405   /**
406    * Writes an binary file entry to the ZIP output stream.
407    * @param path Content path
408    * @param is Input stream with binary data
409    * @throws IOException I/O exception
410    */
411   private void writeBinaryFile(String path, InputStream is) throws IOException {
412     zip.putNextEntry(new ZipEntry(path));
413     try {
414       IOUtils.copy(is, zip);
415     }
416     finally {
417       zip.closeEntry();
418     }
419   }
