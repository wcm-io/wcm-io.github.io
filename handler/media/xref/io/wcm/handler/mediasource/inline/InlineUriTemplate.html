20  package io.wcm.handler.mediasource.inline;
21  
22  import static io.wcm.handler.media.MediaNameConstants.URI_TEMPLATE_PLACEHOLDER_HEIGHT;
23  import static io.wcm.handler.media.MediaNameConstants.URI_TEMPLATE_PLACEHOLDER_WIDTH;
24  
25  import org.apache.commons.lang3.StringUtils;
26  import org.apache.sling.api.adapter.Adaptable;
27  import org.apache.sling.api.resource.Resource;
28  import org.jetbrains.annotations.NotNull;
29  import org.jetbrains.annotations.Nullable;
30  
31  import io.wcm.handler.media.CropDimension;
32  import io.wcm.handler.media.Dimension;
33  import io.wcm.handler.media.MediaArgs;
34  import io.wcm.handler.media.UriTemplate;
35  import io.wcm.handler.media.UriTemplateType;
36  import io.wcm.handler.media.impl.ImageFileServlet;
37  import io.wcm.handler.media.impl.ImageFileServletSelector;
38  import io.wcm.handler.media.impl.JcrBinary;
39  import io.wcm.handler.media.impl.MediaFileServlet;
40  import io.wcm.handler.url.UrlHandler;
41  import io.wcm.sling.commons.adapter.AdaptTo;
42  
43  final class InlineUriTemplate implements UriTemplate {
44  
45    private final String uriTemplate;
46    private final UriTemplateType type;
47    private final Dimension dimension;
48  
49    @SuppressWarnings("java:S107") // allow more than 7 params
50    InlineUriTemplate(@NotNull UriTemplateType type, @NotNull Dimension dimension,
51        @NotNull Resource resource, @NotNull String fileName,
52        @Nullable CropDimension cropDimension, @Nullable Integer rotation,
53        @NotNull MediaArgs mediaArgs, @NotNull Adaptable adaptable) {
54      this.uriTemplate = buildUriTemplate(type, resource, fileName, cropDimension, rotation, mediaArgs, adaptable);
55      this.type = type;
56      this.dimension = dimension;
57    }
58  
59    private static String buildUriTemplate(@NotNull UriTemplateType type, @NotNull Resource resource,
60        @NotNull String fileName, @Nullable CropDimension cropDimension, @Nullable Integer rotation,
61        @NotNull MediaArgs mediaArgs, @NotNull Adaptable adaptable) {
62      String resourcePath = resource.getPath();
63  
64      // if parent resource is a nt:file resource, use this one as path for scaled image
65      Resource parentResource = resource.getParent();
66      if (parentResource != null && JcrBinary.isNtFile(parentResource)) {
67        resourcePath = parentResource.getPath();
68      }
69  
70      // URL to render scaled image via {@link InlineRenditionServlet}
71      final long DUMMY_WIDTH = 999991;
72      final long DUMMY_HEIGHT = 999992;
73      String path = resourcePath
74          + "." + ImageFileServletSelector.build(DUMMY_WIDTH, DUMMY_HEIGHT, cropDimension, rotation,
75              mediaArgs.getImageQualityPercentage(), false)
76          + "." + MediaFileServlet.EXTENSION
77          // replace extension based on the format supported by ImageFileServlet for rendering for this rendition
78          + "/" + ImageFileServlet.getImageFileName(fileName, mediaArgs.getEnforceOutputFileExtension());
79  
80      // build externalized URL
81      UrlHandler urlHandler = AdaptTo.notNull(adaptable, UrlHandler.class);
82      String url = urlHandler.get(path).urlMode(mediaArgs.getUrlMode()).buildExternalResourceUrl(resource);
83  
84      switch (type) {
85        case CROP_CENTER:
86          url = StringUtils.replace(url, Long.toString(DUMMY_WIDTH), URI_TEMPLATE_PLACEHOLDER_WIDTH);
87          url = StringUtils.replace(url, Long.toString(DUMMY_HEIGHT), URI_TEMPLATE_PLACEHOLDER_HEIGHT);
88          break;
89        case SCALE_WIDTH:
90          url = StringUtils.replace(url, Long.toString(DUMMY_WIDTH), URI_TEMPLATE_PLACEHOLDER_WIDTH);
91          url = StringUtils.replace(url, Long.toString(DUMMY_HEIGHT), "0");
92          break;
93        case SCALE_HEIGHT:
94          url = StringUtils.replace(url, Long.toString(DUMMY_WIDTH), "0");
95          url = StringUtils.replace(url, Long.toString(DUMMY_HEIGHT), URI_TEMPLATE_PLACEHOLDER_HEIGHT);
96          break;
97        default:
98          throw new IllegalArgumentException("Unsupported type: " + type);
99      }
100     return url;
101   }
102 
103   @Override
104   public @NotNull UriTemplateType getType() {
105     return type;
106   }
107 
108   @Override
109   public @NotNull String getUriTemplate() {
110     return uriTemplate;
111   }
112 
113   @Override
114   public long getMaxWidth() {
115     return dimension.getWidth();
116   }
117 
118   @Override
119   public long getMaxHeight() {
120     return dimension.getHeight();
121   }
122 
123   @Override
124   public String toString() {
125     return uriTemplate;
126   }
127 
128 }
