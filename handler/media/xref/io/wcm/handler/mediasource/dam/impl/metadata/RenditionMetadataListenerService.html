62  /**
63   * Background service that extracts additional metadata like width and height for DAM renditions.
64   */
65  @Component(service = EventHandler.class, immediate = true,
66      // define explicit PID which was the old location of the implementation class
67      configurationPid = "io.wcm.handler.mediasource.dam.impl.DamRenditionMetadataService",
68      property = {
69          EventConstants.EVENT_TOPIC + "=" + DamEvent.EVENT_TOPIC
70      })
71  @Designate(ocd = RenditionMetadataListenerService.Config.class)
72  public final class RenditionMetadataListenerService implements EventHandler {
73  
74    @ObjectClassDefinition(
75        name = "wcm.io Media Handler Rendition Metadata Service",
76        description = "Extracts additional metadata like width and height for AEM asset renditions.")
77    @interface Config {
78  
79      @AttributeDefinition(name = "Enabled", description = "Switch to enable or disable this service.")
80      boolean enabled() default true;
81  
82      @AttributeDefinition(name = "Allowed run mode(s)", description = "Run modes this service is allowed on. "
83          + "The service is disabled if the run mode does not match.")
84      String[] allowedRunMode() default RunMode.AUTHOR;
85  
86      @AttributeDefinition(name = "Thread pool size", description = "Size of threads in pool that is used to process "
87          + "asset rendition events asynchronously. "
88          + "Setting it to 0 disables asynchronous processing completely (not recommended for production environments).")
89      int threadPoolSize() default 10;
90  
91    }
92  
93    private static final int REMOVE_EVENT_EXECUTION_DELAY_SECONDS = 10;
94    private static final int MAX_RETRY_COUNT = 3;
95    private static final int RETRY_DELAY_SECONDS = 5;
96    private static final String SERVICEUSER_SUBSERVICE = "dam-rendition-metadata";
97  
98    private static final EnumSet<DamEvent.Type> SUPPORTED_EVENT_TYPES = EnumSet.of(DamEvent.Type.RENDITION_UPDATED, DamEvent.Type.RENDITION_REMOVED);
99    private static final Logger log = LoggerFactory.getLogger(RenditionMetadataListenerService.class);
100 
101   private boolean enabled;
102   private boolean synchronousProcessing;
103 
104   @Reference
105   private ResourceResolverFactory resourceResolverFactory;
106   @Reference
107   private InstanceTypeService instanceTypeService;
108   @Reference
109   private AssetSynchonizationService assetSynchronizationService;
110   @Reference
111   private AssetStore assetStore;
112 
113   private ScheduledExecutorService executorService;
114 
115   @Activate
116   @SuppressWarnings("deprecation")
117   private void activate(ComponentContext componentContext, Config config) {
118     if (config.enabled()) {
119       if (config.allowedRunMode() != null && config.allowedRunMode().length > 0) {
120         // Activate only if configured run modes are met
121         this.enabled = !RunMode.disableIfNoRunModeActive(instanceTypeService.getRunModes(), config.allowedRunMode(), componentContext, log);
122       }
123       else {
124         this.enabled = true;
125       }
126     }
127     else {
128       this.enabled = false;
129     }
130     this.synchronousProcessing = config.threadPoolSize() <= 0;
131     if (this.enabled && !this.synchronousProcessing) {
132       this.executorService = Executors.newScheduledThreadPool(config.threadPoolSize(),
133           new ThreadFactoryBuilder().setNameFormat(getClass().getSimpleName() + "-%d").build());
134     }
135   }
136 
137   @Deactivate
138   private void deactivate() {
139     this.enabled = false;
140     if (executorService != null) {
141       executorService.shutdown();
142       try {
143         executorService.awaitTermination(10, TimeUnit.SECONDS);
144       }
145       catch (InterruptedException ex) {
146         // ignore
147       }
148       executorService = null;
149     }
150   }
151 
152   @Override
153   public void handleEvent(Event event) {
154     if (!enabled || !StringUtils.equals(event.getTopic(), DamEvent.EVENT_TOPIC)) {
155       return;
156     }
157     DamEvent damEvent = DamEvent.fromEvent(event);
158     // process only rendition-related events
159     if (SUPPORTED_EVENT_TYPES.contains(damEvent.getType())) {
160       handleDamEvent(damEvent);
161     }
162   }
163 
164   /**
165    * Handle dam event if certain conditions are fulfilled.
166    * @param event DAM event
167    */
168   private void handleDamEvent(DamEvent event) {
169     // make sure rendition file extension is an image extensions
170     String renditionPath = event.getAdditionalInfo();
171     String renditionNodeName = Text.getName(renditionPath);
172     boolean isOriginal = StringUtils.equals(renditionNodeName, ORIGINAL_FILE);
173     String fileExtension = FilenameUtils.getExtension(renditionNodeName);
174     if (!(isOriginal || MediaFileType.isImage(fileExtension))) {
175       return;
176     }
177 
178     RenditionMetadataEvent runnable = new RenditionMetadataEvent(event.getAssetPath(),
179         renditionPath, event.getType());
180     if (synchronousProcessing) {
181       // execute directly in synchronous mode (e.g. for unit tests)
182       runnable.run();
183     }
184     else {
185       // decouple event processing from listener to avoid timeouts
186       executorService.schedule(runnable, runnable.getDelaySeconds(), TimeUnit.SECONDS);
187     }
188   }
189 
190   private final class RenditionMetadataEvent implements Runnable {
191 
192     private final String assetPath;
193     private final String renditionPath;
194     private final DamEvent.Type eventType;
195 
196     private int retryCount;
197 
198     RenditionMetadataEvent(String assetPath, String renditionPath, Type eventType) {
199       this.assetPath = assetPath;
200       this.renditionPath = renditionPath;
201       this.eventType = eventType;
202     }
203 
204     private int getDelaySeconds() {
205       if (eventType == DamEvent.Type.RENDITION_REMOVED) {
206         // delay event handling in case of removed event for some time to avoid repository conflicts
207         // e.g. when new packages with sample content are installed remove and udpate events
208         // are quickly fired after another
209         return REMOVE_EVENT_EXECUTION_DELAY_SECONDS;
210       }
211       else {
212         return 0;
213       }
214     }
215 
216     @Override
217     public void run() {
218       // process event synchronized per asset path
219       Lock lock = assetSynchronizationService.getLock(assetPath);
220       lock.lock();
221 
222       ResourceResolver serviceResourceResolver = null;
223       try {
224         // open service user session for reading/writing rendition metadata
225         serviceResourceResolver = resourceResolverFactory
226             .getServiceResourceResolver(Map.of(ResourceResolverFactory.SUBSERVICE, SERVICEUSER_SUBSERVICE));
227 
228         // make sure asset resource exists
229         Resource assetResource = serviceResourceResolver.getResource(assetPath);
230         if (assetResource == null) {
231           log.debug("Unable to read asset at {} with user {}", assetPath, serviceResourceResolver.getUserID());
232           return;
233         }
234 
235         if (eventType == DamEvent.Type.RENDITION_UPDATED) {
236           renditionAddedOrUpdated(serviceResourceResolver);
237         }
238         else if (eventType == DamEvent.Type.RENDITION_REMOVED) {
239           renditionRemoved(serviceResourceResolver);
240         }
241       }
242       catch (PersistenceException ex) {
243         // in case of persistence exception retry execution some times later
244         this.retryCount++;
245         if (this.retryCount >= MAX_RETRY_COUNT) {
246           // retried too often - log as error
247           log.error("Failed after {} attempts: {}", this.retryCount, ex.getMessage(), ex);
248         }
249         else {
250           log.debug("Failed {} attempt(s), retry: {}", this.retryCount, ex.getMessage(), ex);
251           executorService.schedule(this, RETRY_DELAY_SECONDS, TimeUnit.SECONDS);
252         }
253       }
254       catch (LoginException ex) {
255         log.error("Missing service user mapping for 'io.wcm.handler.media:dam-rendition-metadata' - see https://wcm.io/handler/media/configuration.html", ex);
256       }
257       finally {
258         lock.unlock();
259         if (serviceResourceResolver != null) {
260           serviceResourceResolver.close();
261         }
262       }
263     }
264 
265     /**
266      * Create or update rendition metadata if rendition is created or updated.
267      * @throws PersistenceException Persistence exception
268      */
269     private void renditionAddedOrUpdated(ResourceResolver resolver) throws PersistenceException {
270       log.trace("Process rendition added/updated event: {}", renditionPath);
271       RenditionMetadataGenerator generator = new RenditionMetadataGenerator(resolver, assetStore);
272       generator.renditionAddedOrUpdated(renditionPath);
273     }
274 
275     /**
276      * Remove rendition metadata node if rendition is removed.
277      * @throws PersistenceException Persistence exception
278      */
279     private void renditionRemoved(ResourceResolver resolver) throws PersistenceException {
280       log.trace("Process rendition removed event: {}", renditionPath);
281       RenditionMetadataGenerator generator = new RenditionMetadataGenerator(resolver, assetStore);
282       generator.renditionRemoved(renditionPath);
283     }
284 
285   }
