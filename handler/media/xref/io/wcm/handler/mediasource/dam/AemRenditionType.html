20  package io.wcm.handler.mediasource.dam;
21  
22  import static com.day.cq.dam.api.DamConstants.PREFIX_ASSET_THUMBNAIL;
23  import static com.day.cq.dam.api.DamConstants.PREFIX_ASSET_WEB;
24  
25  import java.util.regex.Pattern;
26  import java.util.stream.Stream;
27  
28  import org.jetbrains.annotations.NotNull;
29  import org.jetbrains.annotations.Nullable;
30  
31  import com.day.cq.dam.api.Rendition;
32  
33  /**
34   * Defines the different types of renditions generated automatically by AEM.
35   */
36  public enum AemRenditionType {
37  
38    /**
39     * Thumbnail rendition (with <code>cq5dam.thumbnail.</code> prefix).
40     */
41    THUMBNAIL_RENDITION(Pattern.compile("^" + Pattern.quote(PREFIX_ASSET_THUMBNAIL) + "\\..*$")),
42  
43    /**
44     * Web rendition for the image editor/cropping (with <code>cq5dam.web.</code> prefix).
45     */
46    WEB_RENDITION(Pattern.compile("^" + Pattern.quote(PREFIX_ASSET_WEB) + "\\..*$")),
47  
48    /**
49     * Video rendition (with <code>cq5dam.video.</code> prefix).
50     */
51    VIDEO_RENDITION(Pattern.compile("^cq5dam\\.video\\..*$")),
52  
53    /**
54     * Any other rendition generated by AEM (with <code>cq5dam.</code> or <code>cqdam.</code> prefix).
55     */
56    OTHER_RENDITION(Pattern.compile("^(cq5dam|cqdam)\\..*$"));
57  
58    private final Pattern namePattern;
59  
60    AemRenditionType(@NotNull Pattern namePattern) {
61      this.namePattern = namePattern;
62    }
63  
64    /**
65     * @param renditionName Rendition name
66     * @return true if Rendition name matches with this type
67     */
68    public boolean matches(@NotNull String renditionName) {
69      return namePattern.matcher(renditionName).matches();
70    }
71  
72    /**
73     * @param rendition Rendition
74     * @return true if Rendition name matches with this type
75     */
76    public boolean matches(@NotNull Rendition rendition) {
77      return matches(rendition.getName());
78    }
79  
80    /**
81     * @param renditionName Rendition name
82     * @return Matching AEM rendition type or null if no match
83     */
84    @SuppressWarnings("null")
85    public static @Nullable AemRenditionType forRendition(@NotNull String renditionName) {
86      return Stream.of(values())
87          .filter(type -> type.matches(renditionName))
88          .findFirst()
89          .orElse(null);
90    }
91  
92    /**
93     * @param rendition Rendition
94     * @return Matching AEM rendition type or null if no match
95     */
96    public static @Nullable AemRenditionType forRendition(@NotNull Rendition rendition) {
97      return forRendition(rendition.getName());
98    }
99  
100 }
