20  package io.wcm.handler.mediasource.dam.impl;
21  
22  import java.io.InputStream;
23  
24  import org.apache.commons.lang3.builder.EqualsBuilder;
25  import org.apache.commons.lang3.builder.HashCodeBuilder;
26  import org.jetbrains.annotations.NotNull;
27  import org.jetbrains.annotations.Nullable;
28  
29  import com.day.cq.dam.api.Rendition;
30  import com.day.image.Layer;
31  
32  import io.wcm.handler.media.CropDimension;
33  import io.wcm.handler.media.impl.ImageFileServlet;
34  import io.wcm.handler.media.impl.MediaFileServlet;
35  import io.wcm.handler.mediasource.dam.impl.dynamicmedia.DynamicMediaPath;
36  
37  /**
38   * Virtual rendition that is cropping and/or rotating and downscaling from an existing rendition.
39   */
40  class VirtualTransformedRenditionMetadata extends RenditionMetadata {
41  
42    private final long width;
43    private final long height;
44    private final CropDimension cropDimension;
45    private final Integer rotation;
46  
47    VirtualTransformedRenditionMetadata(Rendition rendition, long width, long height,
48        CropDimension cropDimension, Integer rotation) {
49      super(rendition);
50      this.width = width;
51      this.height = height;
52      this.cropDimension = cropDimension;
53      this.rotation = rotation;
54    }
55  
56    @Override
57    public String getFileName(boolean contentDispositionAttachment) {
58      // replace extension based on the format supported by ImageFileServlet for rendering for this rendition
59      return ImageFileServlet.getImageFileName(super.getFileName(contentDispositionAttachment));
60    }
61  
62    @Override
63    public long getFileSize() {
64      // no size for virutal renditions
65      return 0L;
66    }
67  
68    @Override
69    public long getWidth() {
70      return this.width;
71    }
72  
73    @Override
74    public long getHeight() {
75      return this.height;
76    }
77  
78    public CropDimension getCropDimension() {
79      return this.cropDimension;
80    }
81  
82    public Integer getRotation() {
83      return this.rotation;
84    }
85  
86    @Override
87    public @NotNull String getMediaPath(boolean contentDispositionAttachment) {
88      return RenditionMetadata.buildMediaPath(getRendition().getPath()
89          + "." + ImageFileServlet.buildSelectorString(getWidth(), getHeight(),
90              this.cropDimension, this.rotation, contentDispositionAttachment)
91          + "." + MediaFileServlet.EXTENSION, getFileName(contentDispositionAttachment));
92    }
93  
94    @Override
95    public @Nullable String getDynamicMediaPath(boolean contentDispositionAttachment, DamContext damContext) {
96      if (contentDispositionAttachment) {
97        // do not use dynamic media for request forced with content disposition attachment
98        return null;
99      }
100     else {
101       // render virtual rendition with dynamic media
102       return DynamicMediaPath.buildImage(damContext, getWidth(), getHeight(), this.cropDimension, this.rotation);
103     }
104   }
105 
106   @Override
107   protected Layer getLayer() {
108     Layer layer = super.getLayer();
109     if (layer != null) {
110       if (cropDimension != null) {
111         layer.crop(cropDimension.getRectangle());
112         if (width <= layer.getWidth() && height <= layer.getHeight()) {
113           layer.resize((int)width, (int)height);
114         }
115       }
116       if (rotation != null) {
117         layer.rotate(rotation);
118       }
119     }
120     return layer;
121   }
122 
123   @Override
124   protected InputStream getInputStream() {
125     // currently not supported for virtual renditions
126     return null;
127   }
128 
129   @Override
130   public int hashCode() {
131     return new HashCodeBuilder()
132         .append(this.getRendition().getPath())
133         .append(width)
134         .append(height)
135         .append(cropDimension)
136         .append(rotation)
137         .hashCode();
138   }
139 
140   @Override
141   public boolean equals(Object obj) {
142     if (obj == null || obj.getClass() != this.getClass()) {
143       return false;
144     }
145     VirtualTransformedRenditionMetadata other = (VirtualTransformedRenditionMetadata)obj;
146     return new EqualsBuilder()
147         .append(this.getRendition().getPath(), other.getRendition().getPath())
148         .append(this.width, other.width)
149         .append(this.height, other.height)
150         .append(this.cropDimension, other.cropDimension)
151         .append(this.rotation, other.rotation)
152         .build();
153   }
154 
155   @Override
156   public String toString() {
157     StringBuilder sb = new StringBuilder(super.toString());
158     sb.append(" -> ").append(Long.toString(this.width)).append("x").append(Long.toString(this.height));
159     if (cropDimension != null) {
160       sb.append(", ").append(cropDimension.toString());
161     }
162     if (rotation != null) {
163       sb.append(", rotation:").append(Integer.toString(rotation));
164     }
165     return sb.toString();
166   }
167 
168 }
