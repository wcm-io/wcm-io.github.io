48  /**
49   * Implements {@link DynamicMediaSupportService}.
50   */
51  @Component(service = DynamicMediaSupportService.class, immediate = true)
52  public class DynamicMediaSupportServiceImpl implements DynamicMediaSupportService {
53  
54    @Reference
55    private Features featureFlagService;
56    @Reference
57    private PublishUtils dynamicMediaPublishUtils;
58    @Reference
59    private ResourceResolverFactory resourceResolverFactory;
60  
61    private static final String SERVICEUSER_SUBSERVICE = "dynamic-media-support";
62  
63    private static final Logger log = LoggerFactory.getLogger(DynamicMediaSupportServiceImpl.class);
64  
65    @Override
66    public boolean isDynamicMediaEnabled() {
67      return featureFlagService.isEnabled(EntitlementConstants.ASSETS_SCENE7_FEATURE_FLAG_PID);
68    }
69  
70    @Override
71    public @Nullable ImageProfile getImageProfile(@NotNull String profilePath) {
72      try (ResourceResolver resourceResolver = resourceResolverFactory
73          .getServiceResourceResolver(ImmutableMap.of(ResourceResolverFactory.SUBSERVICE, SERVICEUSER_SUBSERVICE))) {
74        Resource profileResource = resourceResolver.getResource(profilePath);
75        if (profileResource != null) {
76          return new ImageProfileImpl(profileResource);
77        }
78      }
79      catch (LoginException ex) {
80        log.error("Missing service user mapping for 'io.wcm.handler.media:dynamic-media-support' - see https://wcm.io/handler/media/configuration.html", ex);
81      }
82      return null;
83    }
84  
85    @Override
86    @SuppressWarnings("null")
87    public @Nullable ImageProfile getImageProfileForAsset(@NotNull Asset asset) {
88      Resource assetResource = AdaptTo.notNull(asset, Resource.class);
89      Resource folderResource = assetResource.getParent();
90      if (folderResource != null) {
91        Resource folderContentResource = folderResource.getChild(JCR_CONTENT);
92        if (folderContentResource != null) {
93          InheritanceValueMap inheritanceValueMap = new HierarchyNodeInheritanceValueMap(folderContentResource);
94          String imageProfilePath = inheritanceValueMap.getInherited(DamConstants.IMAGE_PROFILE, String.class);
95          if (imageProfilePath != null) {
96            return getImageProfile(imageProfilePath);
97          }
98        }
99      }
100     return null;
101   }
102 
103   @Override
104   public @Nullable String getDynamicMediaProductionAssetUrl(@NotNull Asset asset) {
105     Resource assetResource = AdaptTo.notNull(asset, Resource.class);
106     try {
107       String[] productionAssetUrls = dynamicMediaPublishUtils.externalizeImageDeliveryAsset(assetResource);
108       if (productionAssetUrls != null && productionAssetUrls.length > 0) {
109         return productionAssetUrls[0];
110       }
111     }
112     catch (RepositoryException ex) {
113       log.warn("Unable to get dynamic media production asset URLs for {}", assetResource.getPath(), ex);
114     }
115     return null;
116   }
117 
118 }
