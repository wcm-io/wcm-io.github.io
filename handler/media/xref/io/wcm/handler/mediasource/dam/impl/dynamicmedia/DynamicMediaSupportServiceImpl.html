65        name = "wcm.io Media Handler Dynamic Media Support",
66        description = "Configures dynamic media support in media handling.")
67    @interface Config {
68  
69      @AttributeDefinition(
70          name = "Enabled",
71          description = "Enable support for dynamic media. "
72              + "Only gets active when dynamic media is actually enabled for the instance.")
73      boolean enabled() default true;
74  
75      @AttributeDefinition(
76          name = "Dynamic Media Capability",
77          description = "Whether to detect automatically if Dynamic Media is actually for a given asset by looking for existing DM metadata. "
78              + "Setting to ON disables the auto-detection and forces it to enabled for all asssets, setting to OFF forced it to disabled.")
79      DynamicMediaCapabilityDetection dmCapabilityDetection() default DynamicMediaCapabilityDetection.AUTO;
80  
81      @AttributeDefinition(
82          name = "Author Preview Mode",
83          description = "Loads dynamic media images via author instance - to allow previewing unpublished images. "
84              + "Must not be enabled on publish instances.")
85      boolean authorPreviewMode() default false;
86  
87      @AttributeDefinition(
88          name = "Disable AEM Fallback",
89          description = "Disable the automatic fallback to AEM-based rendering of renditions (via Media Handler) "
90              + "if Dynamic Media is enabled, but the asset has not the appropriate Dynamic Media metadata.")
91      boolean disableAemFallback() default false;
92  
93      @AttributeDefinition(
94          name = "Validte Smart Crop Rendition Sizes",
95          description = "Validates that the renditions defined via smart cropping fulfill the requested image width/height to avoid upscaling or white borders.")
96      boolean validateSmartCropRenditionSizes() default true;
97  
98      @AttributeDefinition(
99          name = "Image width limit",
100         description = "The configured width value for 'Reply Image Size Limit'.")
101     long imageSizeLimitWidth() default 2000;
102 
103     @AttributeDefinition(
104         name = "Image height limit",
105         description = "The configured height value for 'Reply Image Size Limit'.")
106     long imageSizeLimitHeight() default 2000;

127   @Activate
128   private void activate(Config config) {
129     this.enabled = config.enabled();
130     this.dmCapabilityDetection = config.dmCapabilityDetection();
131     this.authorPreviewMode = config.authorPreviewMode();
132     this.disableAemFallback = config.disableAemFallback();
133     this.validateSmartCropRenditionSizes = config.validateSmartCropRenditionSizes();
134     this.imageSizeLimit = new Dimension(config.imageSizeLimitWidth(), config.imageSizeLimitHeight());
135 
136     if (this.enabled) {
137       log.info("DynamicMediaSupport: enabled={}, capabilityEnabled={}, capabilityDetection={}, "
138           + "authorPreviewMode={}, disableAemFallback={}, imageSizeLimit={}",
139           this.enabled, this.dmCapabilityDetection, this.dmCapabilityDetection,
140           this.authorPreviewMode, this.disableAemFallback, this.imageSizeLimit);
141     }
142   }

149   @Override
150   public boolean isDynamicMediaCapabilityEnabled(boolean isDynamicMediaAsset) {
151     switch (dmCapabilityDetection) {
152       case AUTO:
153         return isDynamicMediaAsset;
154       case ON:
155         return true;
156       case OFF:
157       default:
158         return false;
159     }
160   }

177   @Override
178   public @Nullable ImageProfile getImageProfile(@NotNull String profilePath) {
179     try (ResourceResolver resourceResolver = resourceResolverFactory
180         .getServiceResourceResolver(Map.of(ResourceResolverFactory.SUBSERVICE, SERVICEUSER_SUBSERVICE))) {
181       Resource profileResource = resourceResolver.getResource(profilePath);
182       if (profileResource != null) {
183         log.debug("Loaded image profile: {}", profilePath);
184         return new ImageProfileImpl(profileResource);
185       }
186     }
187     catch (LoginException ex) {
188       log.error("Missing service user mapping for 'io.wcm.handler.media:dynamic-media-support' - see https://wcm.io/handler/media/configuration.html", ex);
189     }
190     log.debug("Image profile not found: {}", profilePath);
191     return null;
192   }

194   @Override
195   public @Nullable ImageProfile getImageProfileForAsset(@NotNull Asset asset) {
196     Resource assetResource = AdaptTo.notNull(asset, Resource.class);
197     Resource folderResource = assetResource.getParent();
198     if (folderResource != null) {
199       return getImageProfileForAssetFolder(folderResource);
200     }
201     return null;
202   }
203 
204   private @Nullable ImageProfile getImageProfileForAssetFolder(@NotNull Resource folderResource) {
205     if (!DAM_PATH_PATTERN.matcher(folderResource.getPath()).matches()) {
206       return null;
207     }
208     Resource folderContentResource = folderResource.getChild(JCR_CONTENT);
209     if (folderContentResource != null) {
210       String imageProfilePath = folderContentResource.getValueMap().get(DamConstants.IMAGE_PROFILE, String.class);
211       if (imageProfilePath != null) {
212         return getImageProfile(imageProfilePath);
213       }
214     }
215     Resource parentFolderResource = folderResource.getParent();
216     if (parentFolderResource != null) {
217       return getImageProfileForAssetFolder(parentFolderResource);
218     }
219     else {
220       return null;
221     }
222   }

224   @Override
225   public @Nullable String getDynamicMediaServerUrl(@NotNull Asset asset, @Nullable UrlMode urlMode, @NotNull Adaptable adaptable) {
226     Resource assetResource = AdaptTo.notNull(asset, Resource.class);
227     if (authorPreviewMode && !forcePublishMode(urlMode)) {
228       // route dynamic media requests through author instance for preview
229       // return configured author URL, or empty string if none configured
230       SiteConfig siteConfig = AdaptTo.notNull(adaptable, SiteConfig.class);
231       String siteUrlAUthor = StringUtils.defaultString(siteConfig.siteUrlAuthor());
232       UrlHandler urlHandler = AdaptTo.notNull(adaptable, UrlHandler.class);
233       return urlHandler.applySiteUrlAutoDetection(siteUrlAUthor);
234     }
235     try {
236       String[] productionAssetUrls = dynamicMediaPublishUtils.externalizeImageDeliveryAsset(assetResource);
237       if (productionAssetUrls != null && productionAssetUrls.length > 0) {
238         return productionAssetUrls[0];
239       }
240     }
241     catch (RepositoryException ex) {
242       log.warn("Unable to get dynamic media production asset URLs for {}", assetResource.getPath(), ex);
243     }
244     log.warn("Unable to get dynamic media production asset URLs for {}", assetResource.getPath());
245     return null;
246   }
247 
248   /**
249    * If URL mode is target for publish instance, use dynamic media production URL.
250    * @param urlMode URL mode
251    * @return true if publish mode should be forced
252    */
253   private boolean forcePublishMode(@Nullable UrlMode urlMode) {
254     return urlMode != null && (urlMode.equals(UrlModes.FULL_URL_PUBLISH)
255         || urlMode.equals(UrlModes.FULL_URL_PUBLISH_FORCENONSECURE)
256         || urlMode.equals(UrlModes.FULL_URL_PUBLISH_FORCESECURE)
257         || urlMode.equals(UrlModes.FULL_URL_PUBLISH_PROTOCOLRELATIVE));
258   }
