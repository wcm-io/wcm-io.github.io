76    /**

80      this.resourceResolver = resourceResolver;
81      this.assetStore = assetStore;
82    }

84    /**
85     * Generate/validate rendition metadata of all renditions of this asset.
86     * @param asset Asset
87     */
88    public void processAllRenditions(Asset asset) {
89      Set<String> existingRenditionNames = new HashSet<>();
90      List<String> renditionPaths = new ArrayList<>();
91  
92      int addUpdateCount = 0;
93      int removeCount = 0;
94      int errorCount = 0;
95  
96      // get existing rendition names and paths
97      for (Rendition rendition : asset.getRenditions()) {
98        existingRenditionNames.add(rendition.getName());
99        renditionPaths.add(rendition.getPath());
100     }
101 
102     // get existing rendition names for which metadata exists (some may be obsolete)
103     Set<String> existingMetadataRenditionNames = new HashSet<>();
104     Resource metadataResource = resourceResolver.getResource(asset.getPath() + "/" + JCR_CONTENT + "/" + NN_RENDITIONS_METADATA);
105     if (metadataResource != null) {
106       for (Resource metadataItem : metadataResource.getChildren()) {
107         existingMetadataRenditionNames.add(metadataItem.getName());
108       }
109     }
110 
111     // generate metadata for all existing renditions
112     for (String renditionPath : renditionPaths) {
113       try {
114         if (renditionAddedOrUpdated(renditionPath)) {
115           addUpdateCount++;
116         }
117       }
118       catch (PersistenceException ex) {
119         log.error(ex.getMessage(), ex);
120         errorCount++;
121       }
122     }
123 
124     // remove obsolete metadata
125     existingMetadataRenditionNames.removeAll(existingRenditionNames);
126     for (String obsoleteRenditionName : existingMetadataRenditionNames) {
127       String nonexistingRenditionPath = asset.getPath() + "/" + JCR_CONTENT + "/" + RENDITIONS_FOLDER
128           + "/" + obsoleteRenditionName;
129       try {
130         if (renditionRemoved(nonexistingRenditionPath)) {
131           removeCount++;
132         }
133       }
134       catch (PersistenceException ex) {
135         log.error(ex.getMessage(), ex);
136         errorCount++;
137       }
138     }
139 
140     log.info("Processed rendition metadata for {}, {} added/updated, {} removed, {} error(s)",
141         asset.getPath(), addUpdateCount, removeCount, errorCount);
142   }

144   /**
145    * Create or update rendition metadata if rendition is created or updated.
146    * @param renditionPath Rendition path
147    * @throws PersistenceException Persistence exception
148    * @return true if rendition data was added or updated
149    */
150   public boolean renditionAddedOrUpdated(String renditionPath) throws PersistenceException {
151 
152     // check for resource existence and try to get layer from image
153     Resource renditionResource = resourceResolver.getResource(renditionPath);
154     if (renditionResource == null) {
155       log.debug("Skip generation of metadata for non-existing rendition {}", renditionPath);
156       return false;
157     }
158 
159     // check if rendition is original
160     boolean isOriginal = StringUtils.equals(ResourceUtil.getName(renditionPath), ORIGINAL_FILE);
161     if (isOriginal) {
162       // skip original unless it is an SVG file (for which AEM does not generated width/height metadata properties)
163       boolean isSVG = StringUtils.equals(getContentType(renditionResource), ContentType.SVG);
164       if (!isSVG) {
165         log.debug("Skip original rendition {}", renditionPath);
166         return false;
167       }
168     }
169     else {
170       // ensure rendition is an image rendition for which metadata can be generated
171       String fileExtension = FilenameUtils.getExtension(renditionPath);
172       if (!MediaFileType.isImage(fileExtension)) {
173         log.debug("Skip non-image rendition {}", renditionPath);
174         return false;
175       }
176     }
177 
178     // Compare timestamps of rendition and rendition metadata
179     Calendar renditionTimestamp = getLastModified(renditionResource);
180     String metdataResourcePath = getRenditionMetadataResourcePath(renditionPath);
181     Resource metadataResource = resourceResolver.getResource(metdataResourcePath);
182     Calendar renditionsMetadataTimestamp = getLastModified(metadataResource);
183     boolean metadataOutdated = (renditionTimestamp == null)
184         || (renditionsMetadataTimestamp == null)
185         || renditionsMetadataTimestamp.before(renditionTimestamp);
186     if (!metadataOutdated) {
187       log.debug("Skip re-generation of metadata for unchanged rendition {}", renditionPath);
188       return false;
189     }
190 
191     // calculate rendition dimension
192     // (record duration of converting resource to layer for debugging)
193     long startTime = System.currentTimeMillis();
194     Dimension dimension = getRenditionDimension(renditionResource);
195     long conversionDuration = System.currentTimeMillis() - startTime;
196     if (dimension == null) {
197       log.debug("Unable to calculate dimension of rendition {}", renditionPath);
198       return false;
199     }
200 
201     // write metadata
202     try {
203       log.debug("Update rendition metadata at {} (width={}, height={}); duration={}ms.",
204           metdataResourcePath, dimension.getWidth(), dimension.getHeight(), conversionDuration);
205 
206       if (metadataResource == null) {
207         metadataResource = ResourceUtil.getOrCreateResource(resourceResolver,
208             metdataResourcePath,
209             Map.<String, Object>of(JCR_PRIMARYTYPE, NT_UNSTRUCTURED),
210             null, false);
211       }
212 
213       ModifiableValueMap props = AdaptTo.notNull(metadataResource, ModifiableValueMap.class);
214       props.put(PN_IMAGE_WIDTH, dimension.getWidth());
215       props.put(PN_IMAGE_HEIGHT, dimension.getHeight());
216       props.put(JCR_LASTMODIFIED, Calendar.getInstance());
217       props.put(JCR_LAST_MODIFIED_BY, resourceResolver.getUserID());
218       resourceResolver.commit();
219       return true;
220     }
221     catch (PersistenceException ex) {
222       throw new PersistenceException("Unable to create or update rendition metadata node for " + renditionPath, ex);
223     }
224   }

226   private @Nullable String getContentType(@NotNull Resource renditionResource) {
227     return renditionResource.getValueMap().get(JCR_CONTENT + "/" + JCR_MIMETYPE, String.class);
228   }
229 
230   private Calendar getLastModified(@Nullable Resource resource) {
231     Calendar lastModified = null;
232     if (resource != null) {
233       // if a rendition is updated it's last modified date is stored in the jcr:content child node
234       Resource contentResource = resource.getChild(JCR_CONTENT);
235       if (contentResource != null) {
236         lastModified = contentResource.getValueMap().get(JCR_LASTMODIFIED, Calendar.class);
237       }
238       if (lastModified == null) {
239         lastModified = resource.getValueMap().get(JCR_LASTMODIFIED, Calendar.class);
240       }
241       if (lastModified == null) {
242         lastModified = resource.getValueMap().get(JCR_CREATED, Calendar.class);
243       }
244     }
245     return lastModified;
246   }

248   /**
249    * Remove rendition metadata node if rendition is removed.
250    * @param renditionPath Rendition path
251    * @throws PersistenceException Persistence exception
252    * @return true if rendition data was removed
253    */
254   public boolean renditionRemoved(String renditionPath) throws PersistenceException {
255 
256     // check if rendition still exist (or exists again) - in this case skip removing of renditions metadata
257     Resource renditionResource = resourceResolver.getResource(renditionPath);
258     if (renditionResource != null) {
259       log.debug("Skip removing of metadata for existing rendition {}", renditionPath);
260       return false;
261     }
262 
263     // remove rendition metadata for non-existing rendition
264     String metdataResourcePath = getRenditionMetadataResourcePath(renditionPath);
265     Resource metadataResource = resourceResolver.getResource(metdataResourcePath);
266     if (metadataResource == null) {
267       return false;
268     }
269     try {
270       log.debug("Remove rendition metadata at {}.", metadataResource.getPath());
271       resourceResolver.delete(metadataResource);
272       resourceResolver.commit();
273       return true;
274     }
275     catch (PersistenceException ex) {
276       throw new PersistenceException("Unable to delete rendition metadata node for " + renditionPath, ex);
277     }
278   }

280   /**
281    * Get dimension (width/height) of rendition.
282    * @param renditionResource Rendition
283    * @return Dimension or null if it could not be detected
284    */
285   private Dimension getRenditionDimension(Resource renditionResource) {
286     Layer layer = ResourceLayerUtil.toLayer(renditionResource, assetStore);
287     if (layer == null) {
288       return null;
289     }
290     return new Dimension(layer.getWidth(), layer.getHeight());
291   }
292 
293   /**
294    * Get resource path for metadata for given rendition.
295    * @param renditionPath Rendition path
296    * @return Metadata resource or null if none exist
297    */
298   private String getRenditionMetadataResourcePath(String renditionPath) {
299     String assetPath = StringUtils.substringBefore(renditionPath, "/" + JCR_CONTENT + "/" + RENDITIONS_FOLDER + "/");
300     String renditionName = Text.getName(renditionPath);
301     return assetPath + "/" + JCR_CONTENT + "/" + NN_RENDITIONS_METADATA + "/" + renditionName;
302   }
