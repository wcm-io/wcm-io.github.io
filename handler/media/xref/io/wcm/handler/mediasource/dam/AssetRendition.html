57  /**
58   * Helper methods for getting metadata for DAM renditions.
59   */
60  @ProviderType
61  public final class AssetRendition {

63    private static final Logger log = LoggerFactory.getLogger(AssetRendition.class);
64  
65    private static final Pattern ASSET_THUMBNAIL_RENDITION_NAME = Pattern.compile("^" + Pattern.quote(PREFIX_ASSET_THUMBNAIL) + "\\..*$");
66    private static final Pattern ASSET_WEB_RENDITION_NAME = Pattern.compile("^" + Pattern.quote(PREFIX_ASSET_WEB) + "\\..*$");

68    private AssetRendition() {
69      // static methods only
70    }

72    /**
73     * Get dimension (width, height) of given DAM rendition.
74     * <p>
75     * It reads the dimension information from the
76     * asset metadata for the original rendition, or from the rendition metadata generated by the
77     * "DamRenditionMetadataService". If both is not available it gets the dimension from the renditions
78     * binary file, but this is inefficient and should not happen under sound conditions.
79     * </p>
80     * @param rendition Rendition
81     * @return Dimension or null if dimension could not be detected, not even in fallback mode
82     */
83    public static @Nullable Dimension getDimension(@NotNull Rendition rendition) {
84      return getDimension(rendition, false);
85    }
86  
87    /**
88     * Get dimension (width, height) of given DAM rendition.
89     * <p>
90     * It reads the dimension information from the
91     * asset metadata for the original rendition, or from the rendition metadata generated by the
92     * "DamRenditionMetadataService". If both is not available it gets the dimension from the renditions
93     * binary file, but this is inefficient and should not happen under sound conditions.
94     * </p>
95     * @param rendition Rendition
96     * @param suppressLogWarningNoRenditionsMetadata If set to true, no log warnings is generated when
97     *          renditions metadata containing the width/height of the rendition does not exist (yet).
98     * @return Dimension or null if dimension could not be detected, not even in fallback mode
99     */
100   public static @Nullable Dimension getDimension(@NotNull Rendition rendition,
101       boolean suppressLogWarningNoRenditionsMetadata) {
102 
103     boolean isOriginal = isOriginal(rendition);
104     String fileExtension = FilenameUtils.getExtension(getFilename(rendition));
105 
106     // get image width/height
107     Dimension dimension = null;
108     if (isOriginal) {
109       // get width/height from metadata for original renditions
110       dimension = getDimensionFromOriginal(rendition);
111     }
112 
113     // dimensions for non-original renditions only supported for image binaries
114     if (MediaFileType.isImage(fileExtension)) {
115       if (dimension == null) {
116         // otherwise get from rendition metadata written by {@link DamRenditionMetadataService}
117         dimension = getDimensionFromRenditionMetadata(rendition);
118       }
119 
120       // fallback: if width/height could not be read from either asset or rendition metadata load the image
121       // into memory and get width/height from there - but log an warning because this is inefficient
122       if (dimension == null) {
123         dimension = getDimensionFromImageBinary(rendition, suppressLogWarningNoRenditionsMetadata);
124       }
125     }
126 
127     return dimension;
128   }
129 
130   /**
131    * Read dimension for original rendition from asset metadata.
132    * @param rendition Rendition
133    * @return Dimension or null
134    */
135   private static @Nullable Dimension getDimensionFromOriginal(@NotNull Rendition rendition) {
136     Asset asset = rendition.getAsset();
137     // asset may have stored dimension in different property names
138     long width = getAssetMetadataValueAsLong(asset, TIFF_IMAGEWIDTH, EXIF_PIXELXDIMENSION);
139     long height = getAssetMetadataValueAsLong(asset, TIFF_IMAGELENGTH, EXIF_PIXELYDIMENSION);
140     return toDimension(width, height);
141   }
142 
143   private static long getAssetMetadataValueAsLong(Asset asset, String... propertyNames) {
144     for (String propertyName : propertyNames) {
145       long value = NumberUtils.toLong(StringUtils.defaultString(asset.getMetadataValueFromJcr(propertyName), "0"));
146       if (value > 0L) {
147         return value;
148       }
149     }
150     return 0L;
151   }
152 
153   /**
154    * Read dimension for non-original rendition from renditions metadata generated by "DamRenditionMetadataService".
155    * @param rendition Rendition
156    * @return Dimension or null
157    */
158   private static @Nullable Dimension getDimensionFromRenditionMetadata(@NotNull Rendition rendition) {
159     Asset asset = rendition.getAsset();
160     String metadataPath = JCR_CONTENT + "/" + NN_RENDITIONS_METADATA + "/" + rendition.getName();
161     Resource metadataResource = AdaptTo.notNull(asset, Resource.class).getChild(metadataPath);
162     if (metadataResource != null) {
163       ValueMap props = metadataResource.getValueMap();
164       long width = props.get(PN_IMAGE_WIDTH, 0L);
165       long height = props.get(PN_IMAGE_HEIGHT, 0L);
166       return toDimension(width, height);
167     }
168     return null;
169   }
170 
171   /**
172    * Fallback: Read dimension by loading image binary into memory.
173    * @param rendition Rendition
174    * @param suppressLogWarningNoRenditionsMetadata If set to true, no log warnings is generated when
175    *          renditions metadata containing the width/height of the rendition does not exist (yet).
176    * @return Dimension or null
177    */
178   @SuppressWarnings("PMD.GuardLogStatement")
179   private static @Nullable Dimension getDimensionFromImageBinary(@NotNull Rendition rendition,
180       boolean suppressLogWarningNoRenditionsMetadata) {
181     try (InputStream is = rendition.getStream()) {
182       if (is != null) {
183         if (!suppressLogWarningNoRenditionsMetadata) {
184           log.warn("Unable to detect rendition metadata for {}, "
185               + "fallback to inefficient detection by loading image into in memory. "
186               + "Please check if the service user for the bundle 'io.wcm.handler.media' is configured properly.",
187               rendition.getPath());
188         }
189         Layer layer = new Layer(is);
190         long width = layer.getWidth();
191         long height = layer.getHeight();
192         return toDimension(width, height);
193       }
194       else {
195         log.warn("Unable to get binary stream for rendition {}", rendition.getPath());
196       }
197     }
198     catch (IOException ex) {
199       log.warn("Unable to read binary stream to layer for rendition {}", rendition.getPath(), ex);
200     }
201     return null;
202   }
203 
204   /**
205    * Convert width/height to dimension
206    * @param width Width
207    * @param height Height
208    * @return Dimension or null if width or height are not valid
209    */
210   private static @Nullable Dimension toDimension(long width, long height) {
211     if (width > 0L && height > 0L) {
212       return new Dimension(width, height);
213     }
214     return null;
215   }

217   /**
218    * Checks if the given rendition is the original file of the asset
219    * @param rendition DAM rendition
220    * @return true if rendition is the original
221    */
222   public static boolean isOriginal(@NotNull Rendition rendition) {
223     return StringUtils.equals(rendition.getName(), ORIGINAL_FILE);
224   }
225 
226   /**
227    * Checks if the given rendition is a thumbnail rendition generated automatically by the AEM asset workflows
228    * (with <code>cq5dam.thumbnail</code> prefix).
229    * @param rendition DAM rendition
230    * @return true if rendition is a thumbnail rendition
231    */
232   public static boolean isThumbnailRendition(@NotNull Rendition rendition) {
233     return ASSET_THUMBNAIL_RENDITION_NAME.matcher(rendition.getName()).matches();
234   }
235 
236   /**
237    * Checks if the given rendition is a web rendition generated automatically by the AEM asset workflows
238    * (with <code>cq5dam.web</code> prefix).
239    * @param rendition DAM rendition
240    * @return true if rendition is a web rendition
241    */
242   public static boolean isWebRendition(@NotNull Rendition rendition) {
243     return ASSET_WEB_RENDITION_NAME.matcher(rendition.getName()).matches();
244   }
245 
246   /**
247    * Get file name of given rendition. If it is the original rendition get asset name as file name.
248    * @param rendition Rendition
249    * @return File extension or null if it could not be detected
250    */
251   public static String getFilename(@NotNull Rendition rendition) {
252     boolean isOriginal = isOriginal(rendition);
253     if (isOriginal) {
254       return rendition.getAsset().getName();
255     }
256     else {
257       return rendition.getName();
258     }
259   }
