81    /**
82     * @return All renditions that are available for this asset
83     */
84    Set<RenditionMetadata> getAvailableRenditions(MediaArgs mediaArgs) {
85      if (this.renditions == null) {
86        // gather rendition infos of all renditions and sort them by size (smallest or virtual crop rendition first)
87        Set<RenditionMetadata> candidates = new TreeSet<>();
88        for (Rendition rendition : damContext.getAsset().getRenditions()) {
89          addRendition(candidates, rendition, mediaArgs);
90        }
91  
92        // special handling for dynamic media
93        if (damContext.isDynamicMediaEnabled() && damContext.isDynamicMediaAsset()) {
94          // check if there are matching smart crop renditions for the requested media format(s)
95          // and return those instead of the original rendition for further processing
96          String fileExtension = FilenameUtils.getExtension(damContext.getAsset().getName());
97          if (damContext.isDynamicMediaValidateSmartCropRenditionSizes()
98              && MediaFileType.isImage(fileExtension) && !MediaFileType.isVectorImage(fileExtension)) {
99            List<CropDimension> cropDimensions = getDynamicMediaCropDimensions(mediaArgs);
100           if (!cropDimensions.isEmpty()) {
101             candidates.addAll(cropDimensions.stream()
102                 .map(cropDimension -> new VirtualTransformedRenditionMetadata(originalRendition.getRendition(),
103                     cropDimension.getWidth(), cropDimension.getHeight(), mediaArgs.getEnforceOutputFileExtension(), cropDimension,
104                     null, mediaArgs.getImageQualityPercentage()))
105                 .collect(Collectors.toList()));
106           }
107         }
108       }
109 
110       candidates = postProcessCandidates(candidates, mediaArgs);
111       this.renditions = Collections.unmodifiableSet(candidates);
112     }
113     return this.renditions;
114   }

126   /**
127    * adds rendition to the list of candidates, if it should be available for resolving
128    * @param candidates Candidates
129    * @param rendition Rendition
130    */
131   private void addRendition(Set<RenditionMetadata> candidates, Rendition rendition, MediaArgs mediaArgs) {
132     AemRenditionType aemRenditionType = AemRenditionType.forRendition(rendition);
133     if (aemRenditionType != null && !getIncludeAssetAemRenditions(mediaArgs).contains(aemRenditionType)) {
134       // ignore all other AEM-generated renditions unless allowed via mediaargs
135       return;
136     }
137 
138     if (!AssetRendition.isOriginal(rendition)
139         && ((damContext.isDynamicMediaEnabled() && damContext.isDynamicMediaAsset()) || damContext.isWebOptimizedImageDeliveryEnabled())) {
140       // skip all non-original renditions for dynamic media and web-optimized delivery - they are not supported
141       return;
142     }
143 
144     RenditionMetadata renditionMetadata = createRenditionMetadata(rendition);
145     candidates.add(renditionMetadata);
146   }

148   /**
149    * Get combined set of allowed AEM-generated rendition types.
150    * @param mediaArgs Media args
151    * @return All allowed AEM-generated rendition types
152    */
153   @SuppressWarnings("deprecation")
154   private @NotNull Set<AemRenditionType> getIncludeAssetAemRenditions(MediaArgs mediaArgs) {
155     Set<AemRenditionType> fromMediaArgs = mediaArgs.getIncludeAssetAemRenditions();
156     if (fromMediaArgs == null) {
157       fromMediaArgs = Set.of();
158     }
159     Set<AemRenditionType> result = new HashSet<>(fromMediaArgs);
160     // check deprecated flags for web renditions and asset thumbnails
161     Boolean includeAssetWebRenditions = mediaArgs.isIncludeAssetWebRenditions();
162     if (includeAssetWebRenditions != null) {
163       if (includeAssetWebRenditions) {
164         result.add(AemRenditionType.WEB_RENDITION);
165       }
166       else {
167         result.remove(AemRenditionType.WEB_RENDITION);
168       }
169     }
170     Boolean includeAssetThumbnails = mediaArgs.isIncludeAssetThumbnails();
171     if (includeAssetThumbnails != null) {
172       if (includeAssetThumbnails) {
173         result.add(AemRenditionType.THUMBNAIL_RENDITION);
174       }
175       else {
176         result.remove(AemRenditionType.THUMBNAIL_RENDITION);
177       }
178     }
179     return result;
180   }

182   /**
183    * Try to get actual smart crop dimensions for the requested ratio(s) for the current asset.
184    * @param mediaArgs Media Args with requested media formats
185    * @return Cropping dimensions or empty list if not found
186    */
187   private @NotNull List<CropDimension> getDynamicMediaCropDimensions(MediaArgs mediaArgs) {
188     MediaFormatOption[] mediaFormatOptions = mediaArgs.getMediaFormatOptions();
189     if (mediaFormatOptions == null) {
190       return Collections.emptyList();
191     }
192     List<CropDimension> result = new ArrayList<>();
193     for (MediaFormatOption mediaFormatOption : mediaFormatOptions) {
194       MediaFormat mediaFormat = mediaFormatOption.getMediaFormat();
195       if (mediaFormat != null && mediaFormat.hasRatio()) {
196         NamedDimension smartCropDef = SmartCrop.getDimensionForRatio(damContext.getImageProfile(), mediaFormat.getRatio());
197         if (smartCropDef != null) {
198           CropDimension cropDimension = SmartCrop.getCropDimensionForAsset(damContext.getAsset(), damContext.getResourceResolver(), smartCropDef);
199           if (cropDimension != null) {
200             result.add(cropDimension);
201           }
202         }
203       }
204     }
205     return result;
206   }

217   /**
218    * Get all renditions that match the requested list of file extension.
219    * @param fileExtensions List of file extensions
220    * @return Matching renditions
221    */
222   private Set<RenditionMetadata> getRendtionsMatchingFileExtensions(String[] fileExtensions, MediaArgs mediaArgs) {
223 
224     // if no file extension restriction get all renditions
225     Set<RenditionMetadata> allRenditions = getAvailableRenditions(mediaArgs);
226     if (fileExtensions == null || fileExtensions.length == 0) {
227       return allRenditions;
228     }
229 
230     // otherwise return those with matching extensions
231     Set<RenditionMetadata> matchingRenditions = new TreeSet<>();
232     for (RenditionMetadata rendition : allRenditions) {
233       for (String fileExtension : fileExtensions) {
234         if (StringUtils.equalsIgnoreCase(fileExtension, rendition.getFileExtension())) {
235           matchingRenditions.add(rendition);
236           break;
237         }
238       }
239     }
240     return matchingRenditions;
241   }

243   /**
244    * Get rendition (probably virtual) for given media arguments.
245    * @param mediaArgs Media arguments
246    * @return Rendition or null if none is matching
247    */
248   @Override
249   public RenditionMetadata getRendition(MediaArgs mediaArgs) {
250 
251     // get list of file extensions requested
252     String[] requestedFileExtensions = getRequestedFileExtensions(mediaArgs);
253 
254     // if the array is null file extensions constraints are applied, but do not match to each other
255     // - no rendition can fulfill these constraints
256     if (requestedFileExtensions == null) {
257       return null;
258     }
259 
260     // check if a specific media size is requested
261     boolean isSizeMatchingRequest = isSizeMatchingRequest(mediaArgs, requestedFileExtensions);
262 
263     // get rendition candidates matching for file extensions
264     Set<RenditionMetadata> candidates = getRendtionsMatchingFileExtensions(requestedFileExtensions, mediaArgs);
265 
266     if (log.isTraceEnabled()) {
267       log.trace("GetRendition: requestedFileExtensions={}, isSizeMatchingRequest={}, mediaArgs={}, candidates={}",
268           StringUtils.join(requestedFileExtensions, ","), isSizeMatchingRequest, mediaArgs, candidates);
269     }
270 
271     // if request does not contain any size restrictions return original image or first by filename matching rendition
272     if (!isSizeMatchingRequest) {
273       return getOriginalOrFirstRendition(candidates);
274     }
275 
276     // original rendition is a image - check for matching rendition or build virtual one
277     RenditionMetadata exactMatchRendition = getExactMatchRendition(candidates, mediaArgs);
278     if (exactMatchRendition != null && !enforceVirtualRendition(exactMatchRendition, mediaArgs)) {
279       return exactMatchRendition;
280     }
281 
282     // get rendition virtual rendition downscaled from existing one
283     RenditionMetadata virtualRendition = getVirtualRendition(candidates, mediaArgs);
284     if (virtualRendition != null) {
285       return virtualRendition;
286     }
287 
288     // no match found
289     return null;
290   }

292   protected boolean enforceVirtualRendition(RenditionMetadata rendition, MediaArgs mediaArgs) {
293     if (rendition.isImage() && !rendition.isVectorImage()) {
294       if (damContext.getMediaHandlerConfig().enforceVirtualRenditions()) {
295         return true;
296       }
297       if (mediaArgs.getEnforceOutputFileExtension() != null) {
298         return !StringUtils.equalsIgnoreCase(rendition.getFileExtension(), mediaArgs.getEnforceOutputFileExtension());
299       }
300     }
301     return false;
302   }

304   /**
305    * Checks if the media args contain any with/height restriction, that means a rendition matching
306    * the given size constraints is requested. Additionally it is checked that at least one image file
307    * extension is requested.
308    * @param mediaArgs Media arguments
309    * @return true if any size restriction was defined.
310    */
311   private boolean isSizeMatchingRequest(MediaArgs mediaArgs, String[] requestedFileExtensions) {
312 
313     // check that at least one image file extension is in the list of requested extensions
314     boolean anyImageFileExtension = false;
315     for (String fileExtension : requestedFileExtensions) {
316       if (MediaFileType.isImage(fileExtension)) {
317         anyImageFileExtension = true;
318         break;
319       }
320     }
321     if (!anyImageFileExtension && mediaArgs.getFixedWidth() == 0 && mediaArgs.getFixedHeight() == 0) {
322       return false;
323     }
324 
325     // check for size restriction
326     if (mediaArgs.getFixedWidth() > 0 || mediaArgs.getFixedHeight() > 0) {
327       return true;
328     }
329     Boolean isSizeMatchingMediaFormat = visitMediaFormats(mediaArgs, new MediaFormatVisitor<Boolean>() {
330       @Override
331       public @Nullable Boolean visit(@NotNull MediaFormat mediaFormat) {
332         // check if any width or ratio restrictions are defined for the media format
333         if (mediaFormat.getEffectiveMinWidth() > 0
334             || mediaFormat.getEffectiveMaxWidth() > 0
335             || mediaFormat.getEffectiveMinHeight() > 0
336             || mediaFormat.getEffectiveMaxHeight() > 0
337             || mediaFormat.getMinWidthHeight() > 0
338             || mediaFormat.getRatio() > 0) {
339           return true;
340         }
341         // alternatively check if responsive image is requested via image sizes or picture sources
342         if (mediaArgs.getImageSizes() != null || mediaArgs.getPictureSources() != null) {
343           return true;
344         }
345         return null;
346       }
347     });
348     return isSizeMatchingMediaFormat != null && isSizeMatchingMediaFormat.booleanValue();
349   }

351   /**
352    * Get rendition that matches exactly with the given media args requirements.
353    * @param candidates Rendition candidates
354    * @param mediaArgs Media args
355    * @return Rendition or null if none found
356    */
357   @SuppressWarnings("java:S3776") // ignore complexity
358   private RenditionMetadata getExactMatchRendition(final Set<RenditionMetadata> candidates, MediaArgs mediaArgs) {
359     MediaFormat[] mediaFormats = mediaArgs.getMediaFormats();
360     // check for fixed width and/or height request
361     if (mediaArgs.getFixedWidth() > 0 || mediaArgs.getFixedHeight() > 0) {
362       for (RenditionMetadata candidate : candidates) {
363         if (candidate.matches(mediaArgs.getFixedWidth(), mediaArgs.getFixedHeight())) {
364           return candidate;
365         }
366       }
367     }
368 
369     // otherwise check for media format restriction
370     else if (mediaFormats != null && mediaFormats.length > 0) {
371       return visitMediaFormats(mediaArgs, new MediaFormatVisitor<RenditionMetadata>() {
372         @Override
373         public @Nullable RenditionMetadata visit(@NotNull MediaFormat mediaFormat) {
374           for (RenditionMetadata candidate : candidates) {
375             if (candidate.matches(mediaFormat.getEffectiveMinWidth(),
376                 mediaFormat.getEffectiveMinHeight(),
377                 mediaFormat.getEffectiveMaxWidth(),
378                 mediaFormat.getEffectiveMaxHeight(),
379                 mediaFormat.getMinWidthHeight(),
380                 mediaFormat.getRatio())) {
381               candidate.setMediaFormat(mediaFormat);
382               return candidate;
383             }
384           }
385           return null;
386         }
387       });
388     }
389 
390     // no restriction - return original or first rendition
391     else {
392       return getOriginalOrFirstRendition(candidates);
393     }
394 
395     // none found
396     return null;
397   }

399   /**
400    * Returns original rendition - if it is contained in the candidate set. Otherwise first candidate is returned.
401    * If a VirtualCropRenditionMetadata is present always the first one is returned.
402    * @param candidates Candidates
403    * @return Original or first rendition of candidates or null
404    */
405   private RenditionMetadata getOriginalOrFirstRendition(Set<RenditionMetadata> candidates) {
406     if (this.originalRendition != null && candidates.contains(this.originalRendition)) {
407       return this.originalRendition;
408     }
409     else if (!candidates.isEmpty()) {
410       return candidates.iterator().next();
411     }
412     else {
413       return null;
414     }
415   }

417   /**
418    * Check if a rendition is available from which the required format can be downscaled from and returns
419    * a virtual rendition in this case.
420    * @param candidates Candidates
421    * @param mediaArgs Media args
422    * @return Rendition or null
423    */
424   private RenditionMetadata getVirtualRendition(final Set<RenditionMetadata> candidates, MediaArgs mediaArgs) {
425 
426     // get from fixed width/height
427     if (mediaArgs.getFixedWidth() > 0 || mediaArgs.getFixedHeight() > 0) {
428       long destWidth = mediaArgs.getFixedWidth();
429       long destHeight = mediaArgs.getFixedHeight();
430       double destRatio = 0;
431       if (destWidth > 0 && destHeight > 0) {
432         destRatio = Ratio.get(destWidth, destHeight);
433       }
434       return getVirtualRendition(candidates, destWidth, destHeight, 0, destRatio,
435           mediaArgs.getEnforceOutputFileExtension(), mediaArgs.getImageQualityPercentage());
436     }
437 
438     // or from any media format
439     return visitMediaFormats(mediaArgs, new MediaFormatVisitor<RenditionMetadata>() {
440       @Override
441       public @Nullable RenditionMetadata visit(@NotNull MediaFormat mediaFormat) {
442         long destWidth = mediaFormat.getEffectiveMinWidth();
443         long destHeight = mediaFormat.getEffectiveMinHeight();
444         long minWidthHeight = mediaFormat.getMinWidthHeight();
445         double destRatio = mediaFormat.getRatio();
446         // try to find matching rendition, otherwise check for next media format
447         RenditionMetadata rendition = getVirtualRendition(candidates, destWidth, destHeight, minWidthHeight, destRatio,
448             mediaArgs.getEnforceOutputFileExtension(), mediaArgs.getImageQualityPercentage());
449         if (rendition != null) {
450           rendition.setMediaFormat(mediaFormat);
451         }
452         return rendition;
453       }
454     });
455   }

457   /**
458    * Check if a rendition is available from which the required format can be downscaled from and returns
459    * a virtual rendition in this case.
460    * @param candidates Candidates
461    * @param destWidth Destination width
462    * @param destHeight Destination height
463    * @param minWidthHeight Min. width/height (longest edge)
464    * @param destRatio Destination ratio
465    * @param enforceOutputFileExtension Enforce output file extension
466    * @return Rendition or null
467    */
468   private RenditionMetadata getVirtualRendition(@NotNull Set<RenditionMetadata> candidates,
469       long destWidth, long destHeight, long minWidthHeight, double destRatio,
470       @Nullable String enforceOutputFileExtension, @Nullable Double imageQualityPercentage) {
471 
472     // if ratio is defined get first rendition with matching ratio and same or bigger size
473     if (destRatio > 0) {
474       for (RenditionMetadata candidate : candidates) {
475         if (candidate.matches(destWidth, destHeight, 0, 0, minWidthHeight, destRatio)) {
476           return getVirtualRendition(candidate, destWidth, destHeight, destRatio, enforceOutputFileExtension, imageQualityPercentage);
477         }
478       }
479     }
480     // otherwise get first rendition which is same or bigger in width and height
481     else {
482       for (RenditionMetadata candidate : candidates) {
483         if (candidate.matches(destWidth, destHeight, 0, 0, minWidthHeight, 0d)) {
484           return getVirtualRendition(candidate, destWidth, destHeight, 0d, enforceOutputFileExtension, imageQualityPercentage);
485         }
486       }
487     }
488 
489     // none found
490     return null;
491   }

493   /**
494    * Get virtual rendition for given width/height/ratio.
495    * @param rendition Rendition
496    * @param widthValue Width
497    * @param heightValue Height
498    * @param ratioValue Ratio
499    * @param enforceOutputFileExtension Enforce output file extension
500    * @param imageQualityPercentage Image quality
501    * @return Rendition or null
502    */
503   private RenditionMetadata getVirtualRendition(@NotNull RenditionMetadata rendition, long widthValue, long heightValue,
504       double ratioValue, @Nullable String enforceOutputFileExtension, @Nullable Double imageQualityPercentage) {
505 
506     long width = widthValue;
507     long height = heightValue;
508     double ratio = ratioValue;
509 
510     // if ratio is missing: calculate from given rendition
511     if (ratio < MediaFormatHandler.RATIO_TOLERANCE) {
512       ratio = Ratio.get(rendition.getWidth(), rendition.getHeight());
513     }
514 
515     // if height is missing - calculate from width
516     if (height == 0 && width > 0) {
517       height = Math.round(width / ratio);
518     }
519 
520     // if width is missing - calculate from height
521     if (width == 0 && height > 0) {
522       width = Math.round(height * ratio);
523     }
524 
525     // return virtual rendition
526     if (width > 0 && height > 0) {
527       if (rendition instanceof VirtualTransformedRenditionMetadata) {
528         VirtualTransformedRenditionMetadata cropRendition = (VirtualTransformedRenditionMetadata)rendition;
529         return new VirtualTransformedRenditionMetadata(cropRendition.getRendition(), width, height, enforceOutputFileExtension,
530             cropRendition.getCropDimension(), cropRendition.getRotation(), imageQualityPercentage);
531       }
532       else {
533         return new VirtualRenditionMetadata(rendition.getRendition(), width, height, enforceOutputFileExtension, imageQualityPercentage);
534       }
535     }
536     else {
537       return null;
538     }
539   }
