66    /**
67     * @param damContext DAM context objects
68     */
69    DefaultRenditionHandler(DamContext damContext) {
70      this.damContext = damContext;
71  
72      Rendition damOriginalRendition = damContext.getAsset().getOriginal();
73      originalRendition = damOriginalRendition != null ? new RenditionMetadata(damOriginalRendition) : null;
74    }

76    protected RenditionMetadata getOriginalRendition() {
77      return this.originalRendition;
78    }

80    /**
81     * @return All renditions that are available for this asset
82     */
83    Set<RenditionMetadata> getAvailableRenditions(MediaArgs mediaArgs) {
84      if (this.renditions == null) {
85        // gather rendition infos of all renditions and sort them by size (smallest or virtual crop rendition first)
86        Set<RenditionMetadata> candidates = new TreeSet<>();
87        for (Rendition rendition : damContext.getAsset().getRenditions()) {
88          addRendition(candidates, rendition, mediaArgs);
89        }
90        candidates = postProcessCandidates(candidates, mediaArgs);
91        this.renditions = ImmutableSet.<RenditionMetadata>copyOf(candidates);
92      }
93      return this.renditions;
94    }

106   /**
107    * adds rendition to the list of candidates, if it should be available for resolving
108    * @param candidates Candidates
109    * @param rendition Rendition
110    */
111   private void addRendition(Set<RenditionMetadata> candidates, Rendition rendition, MediaArgs mediaArgs) {
112     // ignore AEM-generated thumbnail renditions unless allowed via mediaargs
113     if (!mediaArgs.isIncludeAssetThumbnails() && AssetRendition.isThumbnailRendition(rendition)) {
114       return;
115     }
116     // ignore AEM-generated web renditions unless allowed via mediaargs
117     boolean isIncludeAssetWebRenditions = mediaArgs.isIncludeAssetWebRenditions() == null || mediaArgs.isIncludeAssetWebRenditions();
118     if (!isIncludeAssetWebRenditions && AssetRendition.isWebRendition(rendition)) {
119       return;
120     }
121 
122     // special handling for dynamic media
123     if (damContext.isDynamicMediaEnabled() && damContext.isDynamicMediaAsset()) {
124       // skip all non-original renditions for dynamic media assets. dynamic media does not support them.
125       if (!AssetRendition.isOriginal(rendition)) {
126         return;
127       }
128 
129       // check if there are matching smart crop renditions for the requested media format(s)
130       // and return those instead of the original rendition for further processing
131       String fileExtension = FilenameUtils.getExtension(damContext.getAsset().getName());
132       if (damContext.isDynamicMediaValidateSmartCropRenditionSizes()
133           && MediaFileType.isImage(fileExtension) && !MediaFileType.isVectorImage(fileExtension)) {
134         List<CropDimension> cropDimensions = getDynamicMediaCropDimensions(mediaArgs);
135         if (!cropDimensions.isEmpty()) {
136           candidates.addAll(cropDimensions.stream()
137               .map(cropDimension -> new VirtualTransformedRenditionMetadata(originalRendition.getRendition(),
138                   cropDimension.getWidth(), cropDimension.getHeight(), mediaArgs.getEnforceOutputFileExtension(), cropDimension, null))
139               .collect(Collectors.toList()));
140           return;
141         }
142       }
143     }
144 
145     RenditionMetadata renditionMetadata = createRenditionMetadata(rendition);
146     candidates.add(renditionMetadata);
147   }

149   /**
150    * Try to get actual smart crop dimensions for the requested ratio(s) for the current asset.
151    * @param mediaArgs Media Args with requested media formats
152    * @return Cropping dimensions or empty list if not found
153    */
154   private @NotNull List<CropDimension> getDynamicMediaCropDimensions(MediaArgs mediaArgs) {
155     if (mediaArgs.getMediaFormatOptions() == null) {
156       return Collections.emptyList();
157     }
158     List<CropDimension> result = new ArrayList<>();
159     for (MediaFormatOption mediaFormatOption : mediaArgs.getMediaFormatOptions()) {
160       MediaFormat mediaFormat = mediaFormatOption.getMediaFormat();
161       if (mediaFormat != null && mediaFormat.hasRatio()) {
162         NamedDimension smartCropDef = SmartCrop.getDimensionForRatio(damContext.getImageProfile(), mediaFormat.getRatio());
163         if (smartCropDef != null) {
164           CropDimension cropDimension =  SmartCrop.getCropDimensionForAsset(damContext.getAsset(), damContext.getResourceResolver(), smartCropDef);
165           if (cropDimension != null) {
166             result.add(cropDimension);
167           }
168         }
169       }
170     }
171     return result;
172   }

183   /**
184    * Get all renditions that match the requested list of file extension.
185    * @param fileExtensions List of file extensions
186    * @return Matching renditions
187    */
188   private Set<RenditionMetadata> getRendtionsMatchingFileExtensions(String[] fileExtensions, MediaArgs mediaArgs) {
189 
190     // if no file extension restriction get all renditions
191     Set<RenditionMetadata> allRenditions = getAvailableRenditions(mediaArgs);
192     if (fileExtensions == null || fileExtensions.length == 0) {
193       return allRenditions;
194     }
195 
196     // otherwise return those with matching extensions
197     Set<RenditionMetadata> matchingRenditions = new TreeSet<>();
198     for (RenditionMetadata rendition : allRenditions) {
199       for (String fileExtension : fileExtensions) {
200         if (StringUtils.equalsIgnoreCase(fileExtension, rendition.getFileExtension())) {
201           matchingRenditions.add(rendition);
202           break;
203         }
204       }
205     }
206     return matchingRenditions;
207   }

209   /**
210    * Get rendition (probably virtual) for given media arguments.
211    * @param mediaArgs Media arguments
212    * @return Rendition or null if none is matching
213    */
214   @Override
215   public RenditionMetadata getRendition(MediaArgs mediaArgs) {
216 
217     // get list of file extensions requested
218     String[] requestedFileExtensions = getRequestedFileExtensions(mediaArgs);
219 
220     // if the array is null file extensions constraints are applied, but do not match to each other
221     // - no rendition can fulfill these constraints
222     if (requestedFileExtensions == null) {
223       return null;
224     }
225 
226     // check if a specific media size is requested
227     boolean isSizeMatchingRequest = isSizeMatchingRequest(mediaArgs, requestedFileExtensions);
228 
229     // get rendition candidates matching for file extensions
230     Set<RenditionMetadata> candidates = getRendtionsMatchingFileExtensions(requestedFileExtensions, mediaArgs);
231 
232     if (log.isTraceEnabled()) {
233       log.trace("GetRendition: requestedFileExtensions={}, isSizeMatchingRequest={}, mediaArgs={}, candidates={}",
234           StringUtils.join(requestedFileExtensions, ","), isSizeMatchingRequest, mediaArgs, candidates);
235     }
236 
237     // if request does not contain any size restrictions return original image or first by filename matching rendition
238     if (!isSizeMatchingRequest) {
239       return getOriginalOrFirstRendition(candidates);
240     }
241 
242     // original rendition is a image - check for matching rendition or build virtual one
243     RenditionMetadata exactMatchRendition = getExactMatchRendition(candidates, mediaArgs);
244     if (exactMatchRendition != null && !enforceVirtualRendition(exactMatchRendition, mediaArgs)) {
245       return exactMatchRendition;
246     }
247 
248     // get rendition virtual rendition downscaled from existing one
249     RenditionMetadata virtualRendition = getVirtualRendition(candidates, mediaArgs);
250     if (virtualRendition != null) {
251       return virtualRendition;
252     }
253 
254     // no match found
255     return null;
256   }

258   private boolean enforceVirtualRendition(RenditionMetadata rendition, MediaArgs mediaArgs) {
259     if (rendition.isImage() && !rendition.isVectorImage()) {
260       if (damContext.getMediaHandlerConfig().enforceVirtualRenditions()) {
261         return true;
262       }
263       if (mediaArgs.getEnforceOutputFileExtension() != null) {
264         return !StringUtils.equalsIgnoreCase(rendition.getFileExtension(), mediaArgs.getEnforceOutputFileExtension());
265       }
266     }
267     return false;
268   }

270   /**
271    * Checks if the media args contain any with/height restriction, that means a rendition matching
272    * the given size constraints is requested. Additionally it is checked that at least one image file
273    * extension is requested.
274    * @param mediaArgs Media arguments
275    * @return true if any size restriction was defined.
276    */
277   private boolean isSizeMatchingRequest(MediaArgs mediaArgs, String[] requestedFileExtensions) {
278 
279     // check that at least one image file extension is in the list of requested extensions
280     boolean anyImageFileExtension = false;
281     for (String fileExtension : requestedFileExtensions) {
282       if (MediaFileType.isImage(fileExtension)) {
283         anyImageFileExtension = true;
284         break;
285       }
286     }
287     if (!anyImageFileExtension && mediaArgs.getFixedWidth() == 0 && mediaArgs.getFixedHeight() == 0) {
288       return false;
289     }
290 
291     // check for size restriction
292     if (mediaArgs.getFixedWidth() > 0 || mediaArgs.getFixedHeight() > 0) {
293       return true;
294     }
295     Boolean isSizeMatchingMediaFormat = visitMediaFormats(mediaArgs, new MediaFormatVisitor<Boolean>() {
296       @Override
297       public @Nullable Boolean visit(@NotNull MediaFormat mediaFormat) {
298         // check if any width or ratio restrictions are defined for the media format
299         if (mediaFormat.getEffectiveMinWidth() > 0
300             || mediaFormat.getEffectiveMaxWidth() > 0
301             || mediaFormat.getEffectiveMinHeight() > 0
302             || mediaFormat.getEffectiveMaxHeight() > 0
303             || mediaFormat.getMinWidthHeight() > 0
304             || mediaFormat.getRatio() > 0) {
305           return true;
306         }
307         // alternatively check if responsive image is requested via image sizes or picture sources
308         if (mediaArgs.getImageSizes() != null || mediaArgs.getPictureSources() != null) {
309           return true;
310         }
311         return null;
312       }
313     });
314     return isSizeMatchingMediaFormat != null && isSizeMatchingMediaFormat.booleanValue();
315   }

317   /**
318    * Get rendition that matches exactly with the given media args requirements.
319    * @param candidates Rendition candidates
320    * @param mediaArgs Media args
321    * @return Rendition or null if none found
322    */
323   private RenditionMetadata getExactMatchRendition(final Set<RenditionMetadata> candidates, MediaArgs mediaArgs) {
324     // check for fixed width and/or height request
325     if (mediaArgs.getFixedWidth() > 0 || mediaArgs.getFixedHeight() > 0) {
326       for (RenditionMetadata candidate : candidates) {
327         if (candidate.matches(mediaArgs.getFixedWidth(), mediaArgs.getFixedHeight())) {
328           return candidate;
329         }
330       }
331     }
332 
333     // otherwise check for media format restriction
334     else if (mediaArgs.getMediaFormats() != null && mediaArgs.getMediaFormats().length > 0) {
335       return visitMediaFormats(mediaArgs, new MediaFormatVisitor<RenditionMetadata>() {
336         @Override
337         public @Nullable RenditionMetadata visit(@NotNull MediaFormat mediaFormat) {
338           for (RenditionMetadata candidate : candidates) {
339             if (candidate.matches(mediaFormat.getEffectiveMinWidth(),
340                 mediaFormat.getEffectiveMinHeight(),
341                 mediaFormat.getEffectiveMaxWidth(),
342                 mediaFormat.getEffectiveMaxHeight(),
343                 mediaFormat.getMinWidthHeight(),
344                 mediaFormat.getRatio())) {
345               candidate.setMediaFormat(mediaFormat);
346               return candidate;
347             }
348           }
349           return null;
350         }
351       });
352     }
353 
354     // no restriction - return original or first rendition
355     else {
356       return getOriginalOrFirstRendition(candidates);
357     }
358 
359     // none found
360     return null;
361   }

363   /**
364    * Returns original rendition - if it is contained in the candidate set. Otherwise first candidate is returned.
365    * If a VirtualCropRenditionMetadata is present always the first one is returned.
366    * @param candidates Candidates
367    * @return Original or first rendition of candidates or null
368    */
369   private RenditionMetadata getOriginalOrFirstRendition(Set<RenditionMetadata> candidates) {
370     if (this.originalRendition != null && candidates.contains(this.originalRendition)) {
371       return this.originalRendition;
372     }
373     else if (!candidates.isEmpty()) {
374       return candidates.iterator().next();
375     }
376     else {
377       return null;
378     }
379   }

381   /**
382    * Check if a rendition is available from which the required format can be downscaled from and returns
383    * a virtual rendition in this case.
384    * @param candidates Candidates
385    * @param mediaArgs Media args
386    * @return Rendition or null
387    */
388   private RenditionMetadata getVirtualRendition(final Set<RenditionMetadata> candidates, MediaArgs mediaArgs) {
389 
390     // get from fixed width/height
391     if (mediaArgs.getFixedWidth() > 0 || mediaArgs.getFixedHeight() > 0) {
392       long destWidth = mediaArgs.getFixedWidth();
393       long destHeight = mediaArgs.getFixedHeight();
394       double destRatio = 0;
395       if (destWidth > 0 && destHeight > 0) {
396         destRatio = Ratio.get(destWidth, destHeight);
397       }
398       return getVirtualRendition(candidates, destWidth, destHeight, 0, destRatio,
399           mediaArgs.getEnforceOutputFileExtension());
400     }
401 
402     // or from any media format
403     return visitMediaFormats(mediaArgs, new MediaFormatVisitor<RenditionMetadata>() {
404       @Override
405       public @Nullable RenditionMetadata visit(@NotNull MediaFormat mediaFormat) {
406         long destWidth = mediaFormat.getEffectiveMinWidth();
407         long destHeight = mediaFormat.getEffectiveMinHeight();
408         long minWidthHeight = mediaFormat.getMinWidthHeight();
409         double destRatio = mediaFormat.getRatio();
410         // try to find matching rendition, otherwise check for next media format
411         RenditionMetadata rendition = getVirtualRendition(candidates, destWidth, destHeight, minWidthHeight, destRatio,
412             mediaArgs.getEnforceOutputFileExtension());
413         if (rendition != null) {
414           rendition.setMediaFormat(mediaFormat);
415         }
416         return rendition;
417       }
418     });
419   }

421   /**
422    * Check if a rendition is available from which the required format can be downscaled from and returns
423    * a virtual rendition in this case.
424    * @param candidates Candidates
425    * @param destWidth Destination width
426    * @param destHeight Destination height
427    * @param minWidthHeight Min. width/height (longest edge)
428    * @param destRatio Destination ratio
429    * @param enforceOutputFileExtension Enforce output file extension
430    * @return Rendition or null
431    */
432   private RenditionMetadata getVirtualRendition(Set<RenditionMetadata> candidates,
433       long destWidth, long destHeight, long minWidthHeight, double destRatio,
434       String enforceOutputFileExtension) {
435 
436     // if ratio is defined get first rendition with matching ratio and same or bigger size
437     if (destRatio > 0) {
438       for (RenditionMetadata candidate : candidates) {
439         if (candidate.matches(destWidth, destHeight, 0, 0, minWidthHeight, destRatio)) {
440           return getVirtualRendition(candidate, destWidth, destHeight, destRatio, enforceOutputFileExtension);
441         }
442       }
443     }
444     // otherwise get first rendition which is same or bigger in width and height
445     else {
446       for (RenditionMetadata candidate : candidates) {
447         if (candidate.matches(destWidth, destHeight, 0, 0, minWidthHeight, 0d)) {
448           return getVirtualRendition(candidate, destWidth, destHeight, 0d, enforceOutputFileExtension);
449         }
450       }
451     }
452 
453     // none found
454     return null;
455   }

457   /**
458    * Get virtual rendition for given width/height/ratio.
459    * @param rendition Rendition
460    * @param widthValue Width
461    * @param heightValue Height
462    * @param ratioValue Ratio
463    * @param enforceOutputFileExtension Enforce output file extension
464    * @return Rendition or null
465    */
466   private RenditionMetadata getVirtualRendition(RenditionMetadata rendition, long widthValue, long heightValue,
467       double ratioValue, String enforceOutputFileExtension) {
468 
469     long width = widthValue;
470     long height = heightValue;
471     double ratio = ratioValue;
472 
473     // if ratio is missing: calculate from given rendition
474     if (ratio < MediaFormatHandler.RATIO_TOLERANCE) {
475       ratio = Ratio.get(rendition.getWidth(), rendition.getHeight());
476     }
477 
478     // if height is missing - calculate from width
479     if (height == 0 && width > 0) {
480       height = Math.round(width / ratio);
481     }
482 
483     // if width is missing - calculate from height
484     if (width == 0 && height > 0) {
485       width = Math.round(height * ratio);
486     }
487 
488     // return virtual rendition
489     if (width > 0 && height > 0) {
490       if (rendition instanceof VirtualTransformedRenditionMetadata) {
491         VirtualTransformedRenditionMetadata cropRendition = (VirtualTransformedRenditionMetadata)rendition;
492         return new VirtualTransformedRenditionMetadata(cropRendition.getRendition(), width, height, enforceOutputFileExtension,
493             cropRendition.getCropDimension(), cropRendition.getRotation());
494       }
495       else {
496         return new VirtualRenditionMetadata(rendition.getRendition(), width, height, enforceOutputFileExtension);
497       }
498     }
499     else {
500       return null;
501     }
502   }
