44  /**
45   * Builds URL to render image rendition via NextGen Dynamic Media.
46   * <p>
47   * Example URL that might be build:
48   * https://host/adobe/assets/urn:aaid:aem:12345678-abcd-abcd-abcd-abcd12345678/as/my-image.jpg?preferwebp=true&quality=85&width=300&crop=16:9,smart
49   * </p>
50   */
51  public final class NextGenDynamicMediaImageUrlBuilder {
52  
53    static final String PARAM_PREFER_WEBP = "preferwebp";
54    static final String PARAM_WIDTH = "width";
55    static final String PARAM_HEIGHT = "height";
56    static final String PARAM_CROP = "crop";
57    static final String PARAM_SMARTCROP = "smartcrop";
58    static final String PARAM_ROTATE = "rotate";
59    static final String PARAM_QUALITY = "quality";
60  
61    private static final Set<String> SUPPORTED_FORMATS = Set.of(
62        FileExtension.JPEG,
63        FileExtension.PNG,
64        FileExtension.GIF,
65        FileExtension.WEBP);
66  
67    private final NextGenDynamicMediaContext context;
68  
69    /**
70     * @param context Context
71     */
72    public NextGenDynamicMediaImageUrlBuilder(@NotNull NextGenDynamicMediaContext context) {
73      this.context = context;
74    }
75  
76    /**
77     * Builds the URL for a rendition.
78     * @param params Parameters
79     * @return URL or null if invalid/not possible
80     */
81    public @Nullable String build(@NotNull NextGenDynamicMediaImageDeliveryParams params) {
82  
83      // get parameters from nextgen dynamic media config for URL parameters
84      String repositoryId;
85      if (context.getReference().getAsset() != null) {
86        repositoryId = context.getNextGenDynamicMediaConfig().getLocalAssetsRepositoryId();
87      }
88      else {
89        repositoryId = context.getNextGenDynamicMediaConfig().getRemoteAssetsRepositoryId();
90      }
91      String imageDeliveryPath = context.getNextGenDynamicMediaConfig().getImageDeliveryBasePath();
92      if (repositoryId == null || imageDeliveryPath == null || StringUtils.isAnyBlank(repositoryId, imageDeliveryPath)) {
93        return null;
94      }
95  
96      // replace placeholders in delivery path
97      String seoName = FilenameUtils.getBaseName(context.getReference().getFileName());
98      String format = getFileExtension();
99      imageDeliveryPath = StringUtils.replace(imageDeliveryPath, PLACEHOLDER_ASSET_ID, context.getReference().getAssetId());
100     imageDeliveryPath = StringUtils.replace(imageDeliveryPath, PLACEHOLDER_SEO_NAME, seoName);
101     imageDeliveryPath = StringUtils.replace(imageDeliveryPath, PLACEHOLDER_FORMAT, format);
102 
103     // prepare URL params
104     Long width = params.getWidth();
105     String widthPlaceholder = params.getWidthPlaceholder();
106     Dimension cropSmartRatio = params.getCropSmartRatio();
107     Integer rotation = params.getRotation();
108     Integer quality = params.getQuality();
109 
110     SortedMap<String, String> urlParamMap = new TreeMap<>();
111     urlParamMap.put(PARAM_PREFER_WEBP, "true");
112     if (widthPlaceholder != null) {
113       urlParamMap.put(PARAM_WIDTH, widthPlaceholder);
114     }
115     else if (width != null) {
116       urlParamMap.put(PARAM_WIDTH, width.toString());
117     }
118     if (cropSmartRatio != null) {
119       boolean hasWidthDefined = width != null || widthPlaceholder != null;
120       applyCroppingParams(urlParamMap, cropSmartRatio, hasWidthDefined);
121     }
122     if (rotation != null && rotation != 0) {
123       urlParamMap.put(PARAM_ROTATE, rotation.toString());
124     }
125     if (quality != null) {
126       urlParamMap.put(PARAM_QUALITY, quality.toString());
127     }
128 
129     // build URL
130     return buildImageUrl(repositoryId, imageDeliveryPath, urlParamMap);
131   }
132 
133   /**
134    * Builds image URL based on URL parameter map.
135    */
136   private static @NotNull String buildImageUrl(@NotNull String repositoryId, @NotNull String imageDeliveryPath,
137       @NotNull SortedMap<String, String> urlParamMap) {
138     StringBuilder url = new StringBuilder();
139     url.append("https://")
140         .append(repositoryId)
141         .append(imageDeliveryPath);
142     String urlParams = urlParamMap.entrySet().stream()
143         .map(entry -> toUrlParam(entry.getKey(), entry.getValue()))
144         .collect(Collectors.joining("&"));
145     if (StringUtils.isNotEmpty(urlParams)) {
146       if (url.indexOf("?") < 0) {
147         url.append("?");
148       }
149       else {
150         url.append("&");
151       }
152       url.append(urlParams);
153     }
154     return url.toString();
155   }
156 
157   /**
158    * Generates URL parameter key/value pair with escaping where appropriate.
159    */
160   private static @NotNull String toUrlParam(@NotNull String key, @NotNull String value) {
161     StringBuilder sb = new StringBuilder();
162     sb.append(key).append("=");
163     // we only need to encode crop, all other parameters are numbers only
164     if (StringUtils.equalsAny(key, PARAM_CROP, PARAM_SMARTCROP)) {
165       sb.append(URLEncoder.encode(value, StandardCharsets.UTF_8));
166     }
167     else {
168       sb.append(value);
169     }
170     return sb.toString();
171   }
172 
173   /**
174    * Generates cropping/smart cropping URL parameters with or without named smart crop.
175    */
176   private void applyCroppingParams(@NotNull SortedMap<String, String> urlParamMap,
177       @NotNull Dimension cropSmartRatio, boolean hasWidthDefined) {
178     SmartCrop namedSmartCrop = getMatchingNamedSmartCrop(cropSmartRatio);
179     if (namedSmartCrop != null) {
180       urlParamMap.put(PARAM_SMARTCROP, namedSmartCrop.getName());
181       if (!hasWidthDefined) {
182         // if no width given apply default width/height to not rely on dimensions defined in AEM image profile
183         String imageWidthHeightDefault = Long.toString(context.getNextGenDynamicMediaConfig().getImageWidthHeightDefault());
184         if (namedSmartCrop.getCropDimension().getWidth() >= namedSmartCrop.getCropDimension().getHeight()) {
185           urlParamMap.put(PARAM_WIDTH, imageWidthHeightDefault);
186         }
187         else {
188           urlParamMap.put(PARAM_HEIGHT, imageWidthHeightDefault);
189         }
190       }
191     }
192     else {
193       urlParamMap.put(PARAM_CROP, cropSmartRatio.getWidth() + ":" + cropSmartRatio.getHeight() + ",smart");
194     }
195   }
196 
197   /**
198    * Looks up named smart crop definition matching the requested ratio.
199    * @param cropSmartRatio Requested ratio
200    * @return Matching named smart crop or null if none found
201    */
202   private @Nullable SmartCrop getMatchingNamedSmartCrop(@NotNull Dimension cropSmartRatio) {
203     NextGenDynamicMediaMetadata metadata = context.getMetadata();
204     if (metadata == null) {
205       return null;
206     }
207     double requestedRatio = Ratio.get(cropSmartRatio);
208     return metadata.getSmartCrops().stream()
209         .filter(smartCrop -> Ratio.matches(smartCrop.getRatio(), requestedRatio))
210         .findFirst()
211         .orElse(null);
212   }
213 
214   /**
215    * @return Get file extension used for rendering via DM API.
216    */
217   public @NotNull String getFileExtension() {
218     String format = context.getDefaultMediaArgs().getEnforceOutputFileExtension();
219     if (StringUtils.isEmpty(format)) {
220       format = StringUtils.toRootLowerCase(FilenameUtils.getExtension(context.getReference().getFileName()));
221     }
222     if (format == null || !SUPPORTED_FORMATS.contains(format)) {
223       format = FileExtension.JPEG;
224     }
225     return format;
226   }
