20  package io.wcm.handler.mediasource.ngdm.markup;
21  
22  import org.apache.sling.api.SlingHttpServletRequest;
23  import org.apache.sling.api.resource.Resource;
24  import org.apache.sling.models.annotations.Model;
25  import org.jetbrains.annotations.NotNull;
26  import org.osgi.annotation.versioning.ConsumerType;
27  
28  import io.wcm.handler.commons.dom.HtmlElement;
29  import io.wcm.handler.media.Dimension;
30  import io.wcm.handler.media.Media;
31  import io.wcm.handler.media.Rendition;
32  import io.wcm.handler.media.markup.MediaMarkupBuilderUtil;
33  import io.wcm.handler.media.spi.MediaMarkupBuilder;
34  
35  /**
36   * Generates an iframe for Next Gen. Dynamic Media hosted player.
37   */
38  @Model(adaptables = {
39      SlingHttpServletRequest.class, Resource.class
40  })
41  @ConsumerType
42  public class NextGenDynamicMediaHostedPlayerMarkupBuilder implements MediaMarkupBuilder {
43  
44    @Override
45    public boolean accepts(@NotNull Media media) {
46      Rendition rendition = media.getRendition();
47      if (!media.isValid() || rendition == null || !media.getMediaRequest().getMediaArgs().isHostedVideoPlayer()) {
48        return false;
49      }
50      return rendition.isVideo();
51    }
52  
53    @Override
54    public HtmlElement build(@NotNull Media media) {
55      Rendition rendition = media.getRendition();
56      if (rendition == null) {
57        return null;
58      }
59      Dimension dimension = MediaMarkupBuilderUtil.getMediaformatDimension(media);
60  
61      HtmlElement iframe = new HtmlElement("iframe");
62      iframe.setAttribute("src", rendition.getUrl());
63      iframe.setAttribute("width", String.valueOf(dimension.getWidth()));
64      iframe.setAttribute("height", String.valueOf(dimension.getHeight()));
65      iframe.setAttribute("allowfullscreen", "true");
66      iframe.setAttribute("frameborder", "0");
67  
68      return iframe;
69    }
70  
71    @Override
72    public boolean isValidMedia(@NotNull HtmlElement element) {
73      return "iframe".equalsIgnoreCase(element.getName());
74    }
75  
76  }
