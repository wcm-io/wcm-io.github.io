20  package io.wcm.handler.mediasource.ngdm.markup;
21  
22  import org.apache.sling.api.SlingHttpServletRequest;
23  import org.apache.sling.api.resource.Resource;
24  import org.apache.sling.api.resource.ValueMap;
25  import org.apache.sling.models.annotations.Model;
26  import org.jetbrains.annotations.NotNull;
27  import org.osgi.annotation.versioning.ConsumerType;
28  
29  import io.wcm.handler.commons.dom.HtmlElement;
30  import io.wcm.handler.commons.dom.Video;
31  import io.wcm.handler.media.Dimension;
32  import io.wcm.handler.media.Media;
33  import io.wcm.handler.media.Rendition;
34  import io.wcm.handler.media.markup.MediaMarkupBuilderUtil;
35  import io.wcm.handler.media.spi.MediaMarkupBuilder;
36  
37  /**
38   * Generates an HTML5 video element for Next Gen. Dynamic Media video renditions.
39   */
40  @Model(adaptables = {
41      SlingHttpServletRequest.class, Resource.class
42  })
43  @ConsumerType
44  public class NextGenDynamicMediaVideoMarkupBuilder implements MediaMarkupBuilder {
45  
46    @Override
47    public boolean accepts(@NotNull Media media) {
48      Rendition rendition = media.getRendition();
49      if (!media.isValid() || rendition == null || media.getMediaRequest().getMediaArgs().isHostedVideoPlayer()) {
50        return false;
51      }
52      return rendition.isVideo();
53    }
54  
55    @Override
56    public HtmlElement build(@NotNull Media media) {
57      Rendition rendition = media.getRendition();
58      if (rendition == null) {
59        return null;
60      }
61      Dimension dimension = MediaMarkupBuilderUtil.getMediaformatDimension(media);
62  
63      Video video = new Video();
64      video.setWidth(dimension.getWidth());
65      video.setHeight(dimension.getHeight());
66      video.setControls(true);
67  
68      ValueMap properties = rendition.getProperties();
69      String posterUrl = properties.get("posterUrl", String.class);
70      if (posterUrl != null) {
71        video.setPoster(posterUrl);
72      }
73  
74      // add source
75      video.createSource()
76          .setType(rendition.getMimeType())
77          .setSrc(rendition.getUrl());
78  
79      return video;
80    }
81  
82    @Override
83    public boolean isValidMedia(@NotNull HtmlElement element) {
84      return element instanceof Video;
85    }
86  
87  }
