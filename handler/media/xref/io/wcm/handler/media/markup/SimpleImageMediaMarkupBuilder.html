58  /**
59   * Basic implementation of {@link io.wcm.handler.media.spi.MediaMarkupBuilder} for images.
60   * <p>
61   * If image sizes or picture sources are set on the media handler this markup builder also
62   * generates markup for responsive images using <code>img</code> with <code>sizes</code> and <code>srcset</code>
63   * attributes or <code>picture</code> with <code>source</code> elements.
64   * </p>
65   */
66  @Model(adaptables = {
67      SlingHttpServletRequest.class, Resource.class
68  })
69  @ConsumerType
70  public class SimpleImageMediaMarkupBuilder extends AbstractImageMediaMarkupBuilder {
71  
72    @Override
73    public final boolean accepts(@NotNull Media media) {
74      // accept if rendition is an image rendition, and resolving was successful
75      return media.isValid()
76          && media.getRendition() != null
77          && media.getRendition().isBrowserImage();
78    }
79  
80    @Override
81    public final HtmlElement<?> build(@NotNull Media media) {
82  
83      // render media element for rendition
84      HtmlElement<?> mediaElement = getMediaElement(media);
85  
86      // further processing in edit or preview mode
87      applyWcmMarkup(mediaElement, media);
88  
89      return mediaElement;
90    }

92    /**
93     * Create <code>img</code> or <code>picture</code> media element.
94     * @param media Media metadata
95     * @return Media element with properties or null if media metadata is invalid
96     */
97    protected @Nullable HtmlElement<?> getMediaElement(@NotNull Media media) {
98      PictureSource[] pictureSources = media.getMediaRequest().getMediaArgs().getPictureSources();
99      if (pictureSources != null && pictureSources.length > 0) {
100       return getPictureElement(media);
101     }
102     else {
103       return getImageElement(media);
104     }
105   }
106 
107   /**
108    * Create an <code>img</code> element that displays the given rendition image.
109    * @param media Media metadata
110    * @return <code>img</code> element with properties or null if media metadata is invalid
111    */
112   protected @Nullable HtmlElement<?> getPictureElement(@NotNull Media media) {
113     PictureSource[] pictureSources = media.getMediaRequest().getMediaArgs().getPictureSources();
114 
115     Picture picture = new Picture();
116 
117     // add source elements (only if matching renditions found)
118     boolean foundAnySource = false;
119     for (PictureSource pictureSource : pictureSources) {
120       Source source = new Source();
121       if (pictureSource.getMedia() != null) {
122         source.setMedia(pictureSource.getMedia());
123       }
124       if (pictureSource.getSizes() != null) {
125         source.setSizes(pictureSource.getSizes());
126       }
127       MediaFormat mediaFormat = pictureSource.getMediaFormat();
128       if (mediaFormat != null) {
129         String srcSet = getSrcSetRenditions(media, mediaFormat, pictureSource.getWidthOptions());
130         if (srcSet != null) {
131           source.setSrcSet(srcSet);
132           picture.add(source);
133           foundAnySource = true;
134         }
135       }
136     }
137 
138     // add image element
139     HtmlElement<?> image = getImageElement(media);
140     if (image == null) {
141       return null;
142     }
143 
144     if (foundAnySource) {
145       if (image instanceof Span) {
146         // if image was wrapped in span, add content of span element, not the span itself
147         for (Element element : ImmutableList.copyOf(image.getChildren())) {
148           element.detach();
149           picture.addContent(element);
150         }
151       }
152       else {
153         picture.addContent(image);
154       }
155       return picture;
156     }
157     else {
158       return image;
159     }
160   }

162   /**
163    * Create an <code>img</code> element that displays the given rendition image.
164    * @param media Media metadata
165    * @return <code>img</code> element with properties or null if media metadata is invalid
166    */
167   protected @Nullable HtmlElement<?> getImageElement(@NotNull Media media) {
168     Image img = null;
169 
170     MediaArgs mediaArgs = media.getMediaRequest().getMediaArgs();
171     Asset asset = media.getAsset();
172     Rendition rendition = media.getRendition();
173 
174     String url = null;
175     if (rendition != null) {
176       url = rendition.getUrl();
177     }
178 
179     if (url != null) {
180       img = new Image(url);
181 
182       // Alternative text
183       String altText = null;
184       if (asset != null) {
185         altText = asset.getAltText();
186       }
187       if (altText != null) {
188         img.setAlt(altText);
189       }
190 
191       // set width/height
192       if (rendition != null
193           && !rendition.isVectorImage()
194           && mediaArgs.getImageSizes() == null
195           && mediaArgs.getPictureSources() == null) {
196         long height = rendition.getHeight();
197         if (height > 0) {
198           img.setHeight(height);
199         }
200         long width = rendition.getWidth();
201         if (width > 0) {
202           img.setWidth(width);
203         }
204       }
205 
206       // set image sizes/srcset
207       ImageSizes imageSizes = mediaArgs.getImageSizes();
208       if (imageSizes != null) {
209         MediaFormat primaryMediaFormat = getFirstMediaFormat(media);
210         if (primaryMediaFormat != null) {
211           String srcSet = getSrcSetRenditions(media, primaryMediaFormat, imageSizes.getWidthOptions());
212           if (srcSet != null) {
213             img.setSrcSet(srcSet);
214             img.setSizes(imageSizes.getSizes());
215           }
216         }
217       }
218 
219     }
220 
221     // set additional attributes
222     setAdditionalAttributes(img, media);
223 
224     // apply image map markup
225     return applyImageMap(img, media);
226   }

228   /**
229    * Generate srcset list from the resolved renditions for the ratio of the given media formats and the given widths.
230    * Widths that have no match are ignored.
231    * @param media Media
232    * @param mediaFormat Media format
233    * @param widths widths
234    * @return srcset String or null if no matching renditions found
235    */
236   protected @Nullable String getSrcSetRenditions(@NotNull Media media, @NotNull MediaFormat mediaFormat,
237       @NotNull WidthOption @Nullable... widths) {
238     if (widths == null) {
239       return null;
240     }
241     return getSrcSetRenditions(media, mediaFormat, Arrays.stream(widths)
242         .mapToLong(WidthOption::getWidth)
243         .toArray());
244   }
245 
246   /**
247    * Generate srcset list from the resolved renditions for the ratio of the given media formats and the given widths.
248    * Widths that have no match are ignored.
249    * @param media Media
250    * @param mediaFormat Media format
251    * @param widths widths
252    * @return srcset String or null if no matching renditions found
253    */
254   protected @Nullable String getSrcSetRenditions(@NotNull Media media, @NotNull MediaFormat mediaFormat,
255       long @NotNull... widths) {
256     StringBuilder srcset = new StringBuilder();
257 
258     for (long width : widths) {
259       Optional<String> url = media.getRenditions().stream()
260           .filter(rendition -> (Ratio.matches(rendition.getRatio(), mediaFormat.getRatio())
261               || Ratio.matches(mediaFormat.getRatio(), 0d))
262               && rendition.getWidth() == width)
263           .map(Rendition::getUrl)
264           .findFirst();
265       if (url.isPresent()) {
266         if (srcset.length() > 0) {
267           srcset.append(", ");
268         }
269         srcset.append(url.get()).append(" ").append(Long.toString(width)).append("w");
270       }
271     }
272 
273     if (srcset.length() > 0) {
274       return srcset.toString();
275     }
276     else {
277       return null;
278     }
279   }

281   /**
282    * Get first media format from the media formats of the media args that has a ratio set.
283    * @param media Media
284    * @return Media format or null if none found
285    */
286   protected final @Nullable MediaFormat getFirstMediaFormatWithRatio(@NotNull Media media) {
287     MediaFormat[] mediaFormats = media.getMediaRequest().getMediaArgs().getMediaFormats();
288     if (mediaFormats != null) {
289       for (MediaFormat mediaFormat : mediaFormats) {
290         if (mediaFormat.hasRatio()) {
291           return mediaFormat;
292         }
293       }
294     }
295     return null;
296   }

298   /**
299    * Get first media format from the resolved media renditions.
300    * @param media Media
301    * @return Media format or null if none found
302    */
303   protected final @Nullable MediaFormat getFirstMediaFormat(@NotNull Media media) {
304     return media.getRenditions().stream()
305         .map(Rendition::getMediaFormat)
306         .filter(Objects::nonNull)
307         .findFirst()
308         .orElse(null);
309   }

311   /**
312    * If a image map was resolved apply map markup to given image element. As a result both image
313    * and map markup are wrapped in a span element.
314    * @param element Image Element
315    * @param media Media
316    * @return Unchanged element or wrapped element with map
317    */
318   protected final @Nullable HtmlElement<?> applyImageMap(@Nullable HtmlElement<?> element, @NotNull Media media) {
319     List<ImageMapArea> mapData = media.getMap();
320     if (!(element instanceof Image) || mapData == null) {
321       return element;
322     }
323 
324     // build unique name for map
325     String mapName = buildImageMapName(mapData, media);
326 
327     // build wrapper element that will contain both image and map element
328     Span span = new Span();
329     ((Image)element).setUseMap("#" + mapName);
330     span.addContent(element);
331 
332     // build image map markup
333     Map map = new Map();
334     map.setMapName(mapName);
335     for (ImageMapArea areaData : mapData) {
336       Area area = new Area();
337       area.setShape(areaData.getShape());
338       area.setCoords(areaData.getCoordinates());
339       area.setHRef(areaData.getLinkUrl());
340       if (areaData.getLinkWindowTarget() != null) {
341         area.setTarget(areaData.getLinkWindowTarget());
342       }
343       if (areaData.getAltText() != null) {
344         area.setAlt(areaData.getAltText());
345       }
346       map.addContent(area);
347     }
348     span.addContent(map);
349 
350     return span;
351   }

353   /**
354    * Builds an ID for the image map that is unique within the page.
355    * @param map Map data
356    * @param media Media
357    * @return Unique ID
358    */
359   protected final @NotNull String buildImageMapName(@NotNull List<ImageMapArea> map, @NotNull Media media) {
360     HashCodeBuilder builder = new HashCodeBuilder();
361     for (ImageMapArea area : map) {
362       builder.append(area);
363     }
364     return "map-" + builder.hashCode();
365   }

367   @Override
368   public final boolean isValidMedia(@NotNull HtmlElement<?> element) {
369     if (element instanceof Image) {
370       Image img = (Image)element;
371       return StringUtils.isNotEmpty(img.getSrc())
372           && !StringUtils.contains(img.getCssClass(), MediaNameConstants.CSS_DUMMYIMAGE);
373     }
374     if (element instanceof Picture) {
375       Element imgChild = element.getChild("img");
376       if (imgChild instanceof Image) {
377         Image img = (Image)imgChild;
378         return StringUtils.isNotEmpty(img.getSrc())
379             && !StringUtils.contains(element.getCssClass(), MediaNameConstants.CSS_DUMMYIMAGE);
380       }
381     }
382     if (element instanceof Span) {
383       Optional<Element> firstChild = element.getChildren().stream().findFirst();
384       if (firstChild.isPresent()) {
385         return isValidMedia((HtmlElement)firstChild.get());
386       }
387     }
388     return false;
389   }
