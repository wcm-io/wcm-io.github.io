20  package io.wcm.handler.media.impl;
21  
22  import java.text.DecimalFormat;
23  import java.text.DecimalFormatSymbols;
24  import java.text.NumberFormat;
25  import java.util.Locale;
26  
27  import org.jetbrains.annotations.NotNull;
28  
29  import io.wcm.handler.media.CropDimension;
30  import io.wcm.handler.media.Dimension;
31  
32  /**
33   * Creates relative crop string with percentage values as required by the Web-Optimized Image Delivery API.
34   * It uses one fractional digit for the percentage values.
35   */
36  public final class RelativeCroppingString {
37  
38    private static final NumberFormat DECIMAL_FORMAT = new DecimalFormat("0.0", new DecimalFormatSymbols(Locale.US));
39  
40    private RelativeCroppingString() {
41      // static methods only
42    }
43  
44    /**
45     * Create relative cropping string from crop dimension and image dimension.
46     * @param cropDimension Crop dimension
47     * @param imageDimension Image dimension
48     * @return Cropping string
49     */
50    public static @NotNull String createFromCropDimension(
51        @NotNull CropDimension cropDimension, @NotNull Dimension imageDimension) {
52      double x1 = cropDimension.getLeft();
53      double y1 = cropDimension.getTop();
54      double left = x1 / imageDimension.getWidth();
55      double top = y1 / imageDimension.getHeight();
56      double width = (double)cropDimension.getWidth() / imageDimension.getWidth();
57      double height = (double)cropDimension.getHeight() / imageDimension.getHeight();
58      return create(left, top, width, height);
59    }
60  
61    /**
62     * Create relative cropping string from percentage values.
63     * @param left Left
64     * @param top Top
65     * @param width Width
66     * @param height Height
67     * @return Cropping string
68     */
69    public static @NotNull String create(double left, double top, double width, double height) {
70      return String.format("%sp,%sp,%sp,%sp",
71          toPercentage(left), toPercentage(top),
72          toPercentage(width), toPercentage(height));
73    }
74  
75    private static String toPercentage(double fraction) {
76      double percentage = Math.round(fraction * 1000d) / 10d;
77      percentage = Math.max(0.0, percentage);
78      percentage = Math.min(100.0, percentage);
79      return DECIMAL_FORMAT.format(percentage);
80    }
81  
82  }
