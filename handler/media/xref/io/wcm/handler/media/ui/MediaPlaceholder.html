44  /**
45   * Model for media replacement placeholder.
46   * <p>
47   * Mandatory use parameters:
48   * </p>
49   * <ul>
50   * <li><code>media</code> (io.wcm.handler.media.Media):
51   * The result object of the media handler (usually in invalid state)</li>
52   * <li><code>classAppend</code> (String):
53   * Additional CSS classes for placeholder element</li>
54   * </ul>
55   */
56  @Model(adaptables = SlingHttpServletRequest.class)
57  public class MediaPlaceholder {
58  
59    @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
60    private Object media;
61    @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
62    private String classAppend;
63  
64    private String classAppendCombined;
65    private String mediaInvalidReason;
66  
67    private static final Logger log = LoggerFactory.getLogger(MediaPlaceholder.class);
68  
69    @PostConstruct
70    private void activate() {
71      Media mediaMetadata = getMediaMetadata();
72      if (mediaMetadata != null) {
73        this.classAppendCombined = mergeClassAppend(getMediaDropCssClass(mediaMetadata), classAppend);
74        this.mediaInvalidReason = getMediaInvalidReasonText(mediaMetadata);
75      }
76    }
77  
78    @SuppressWarnings("PMD.GuardLogStatement")
79    private Media getMediaMetadata() {
80      if (media == null) {
81        log.warn("No 'media' parameter passed to MediaPlaceholder model.");
82        return null;
83      }
84      if (media instanceof Media) {
85        return (Media)media;
86      }
87      else if (media instanceof ResourceMedia) {
88        return ((ResourceMedia)media).getMetadata();
89      }
90      log.warn("Invalid 'media' parameter passed to MediaPlaceholder model. "
91          + "Expected: " + Media.class.getName() + ", actual: " + media.getClass().getName());
92      return null;
93    }
94  
95    private String getMediaDropCssClass(Media mediaMetadata) {
96      Image dummyImage = new Image();
97      mediaMetadata.getMediaSource().enableMediaDrop(dummyImage, mediaMetadata.getMediaRequest());
98      return dummyImage.getCssClass();
99    }
100 
101   private String getMediaInvalidReasonText(Media mediaMetadata) {
102     if (mediaMetadata.getMediaInvalidReason() != null
103         && mediaMetadata.getMediaInvalidReason() != MediaInvalidReason.MEDIA_REFERENCE_MISSING) {
104       // build i18n key
105       return MEDIA_INVALID_REASON_I18N_PREFIX + mediaMetadata.getMediaInvalidReason().name();
106     }
107     else {
108       return null;
109     }
110   }
111 
112   /**
113    * Merges multiple list of CSS Class appends and removes duplicate css classes.
114    * @param classAppends List of class appends (may be null or empty)
115    * @return Class append
116    */
117   private @Nullable String mergeClassAppend(@Nullable String @NotNull... classAppends) {
118     Set<String> result = new LinkedHashSet<>();
119     for (String classAppendItem : classAppends) {
120       if (StringUtils.isNotBlank(classAppendItem)) {
121         result.addAll(Arrays.asList(StringUtils.split(classAppendItem, " ")));
122       }
123     }
124     if (result.isEmpty()) {
125       return null;
126     }
127     else {
128       return StringUtils.join(result, " ");
129     }
130   }
131 
132   /**
133    * Gets additional CSS classes for the replacement placeholder to
134    * allow drag&amp;drop of assets into an empty component.
135    * @return CSS class or null
136    */
137   public @Nullable String getClassAppend() {
138     return this.classAppendCombined;
139   }
140 
141   /**
142    * Additional text to append to empty placeholder message.
143    * @return Empty text
144    */
145   public @Nullable String getMediaInvalidReason() {
146     return this.mediaInvalidReason;
147   }
148 
149 }
