53  /**
54   * Generic resource-based media model.
55   * <p>
56   * Optional use parameters when referencing model from Sightly template:
57   * </p>
58   * <ul>
59   * <li><code>mediaFormat</code>: Media format name to restrict the allowed media items</li>
60   * <li><code>refProperty</code>: Name of the property from which the media reference path is read, or node name for
61   * inline media.</li>
62   * <li><code>cropProperty</code>: Name of the property which contains the cropping parameters</li>
63   * <li><code>rotationProperty</code>: Name of the property which contains the rotation parameter</li>
64   * <li><code>cssClass</code>: CSS classes to be applied on the generated media element (most cases img element)</li>
65   * <li><code>autoCrop</code>: Sets the auto cropping behavior of the media handler. This will override the component
66   * property
67   * {@value io.wcm.handler.media.MediaNameConstants#PN_COMPONENT_MEDIA_AUTOCROP}.</li>
68   * <li><code>imageWidths</code>: Responsive rendition widths for image. Example:
69   * "{@literal 2560?,1920,?1280,640,320}".<br>
70   * Appending the suffix "{@literal ?}" makes the width optional, eg. "1440?"<br>
71   * Use always 'imageWidths' together with 'imageSizes' property.<br>
72   * Cannot be used together with the picture source parameters.</li>
73   * <li><code>imageSizes</code>: "Sizes" string for img element. Example:
74   * "{@literal (min-width: 400px) 400px, 100vw}".<br>
75   * Cannot be used together with the picture source parameters.</li>
76   * <li><code>pictureSourceMediaFormat</code>: List of media formats for the picture source elements. Example:
77   * "{@literal ['mf_16_9']}"<br>
78   * You have to define the same number of array items in all pictureSource* properties.<br>
79   * Cannot be used together with image sizes.</li>
80   * <li><code>pictureSourceMedia</code>: List of media expressions for the picture source elements.
81   * Example: "{@literal ['(max-width: 799px)', '(min-width: 800px)']}"<br>
82   * You have to define the same number of array items in all pictureSource* properties.<br>
83   * Cannot be used together with image sizes.</li>
84   * <li><code>pictureSourceWidths</code>: List of widths for the picture source elements.
85   * Example: "{@literal 479,719,959,1279,1439?,1440?}".<br>
86   * Appending the suffix "{@literal ?}" makes the width optional, eg. "1440?"<br>
87   * You have to define the same number of array items in all pictureSource* properties.<br>
88   * Cannot be used together with image sizes.</li>
89   * <li><code>property:&lt;propertyname&gt;</code>: Custom properties for MediaArgs can be added with the name prefix
90   * {@value #RA_PROPERTY_PREFIX},
91   * e.g. {@literal "property:myprop1"="value1"} which adds property {@literal "myprop1"} to the the MediaArgs. Custom
92   * properties with null value will be ignored.</li>
93   * </ul>
94   */
95  @Model(adaptables = SlingHttpServletRequest.class)
96  public class ResourceMedia {
97  
98    /**
99     * Name prefix for request attributes that will be put into the media builder properties
100    */
101   private static final String RA_PROPERTY_PREFIX = "property:";
102 
103   /**
104    * Regex pattern that matches request attribute names with the prefix {@value #RA_PROPERTY_PREFIX}
105    */
106   private static final Pattern PROPERTY_NAME_PATTERN = Pattern.compile("^" + RA_PROPERTY_PREFIX + ".+$");
107 
108   /**
109    * Optional: Media format to be used.
110    * By default the media formats are read from the component properties of the component and this
111    * parameter should not be set. But for components that allow to choose one from the allowed media
112    * formats via their edit dialog the format can be set here.
113    * To be used together with 'imageSizes' and 'widths'.<br>
114    * Cannot be used together with the picture source parameters.
115    */
116   @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
117   private String mediaFormat;
118 
119   @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
120   private String refProperty;
121 
122   @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
123   private String cropProperty;
124 
125   @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
126   private String rotationProperty;
127 
128   @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
129   private String cssClass;
130 
131   @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
132   private Boolean autoCrop;
133 
134   /**
135    * Defines responsive rendition widths for image.
136    * To be used together with 'imageSizes' property.
137    * Example: "{@literal 2560?,1920,?1280,640,320}" <br>
138    * Widths are by default required. To declare an optional width append the "{@literal ?}" suffix, eg. "1440?"<br>
139    * Cannot be used together with the picture source parameters.
140    */
141   @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
142   private String imageWidths;
143 
144   /**
145    * "Sizes" string for img element.
146    * Example: "{@literal (min-width: 400px) 400px, 100vw}"<br>
147    * Cannot be used together with the picture source parameters.
148    */
149   @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
150   private String imageSizes;
151 
152   /**
153    * List of media formats for the picture source elements.
154    * Example: "{@literal ['mf_16_9']}"<br>
155    * You have to define the same number of array items in all pictureSource* properties.<br>
156    * Cannot be used together with image sizes.
157    */
158   @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
159   private Object[] pictureSourceMediaFormat;
160 
161   /**
162    * List of media expressions for the picture source elements.
163    * Example: "{@literal ['(max-width: 799px)', '(min-width: 800px)']}"<br>
164    * You have to define the same number of array items in all pictureSource* properties.<br>
165    * Cannot be used together with image sizes.
166    */
167   @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
168   private Object[] pictureSourceMedia;
169 
170   /**
171    * List of widths for the picture source elements.
172    * Example: "{@literal 479,719,959,1279,1439?,1440?}"<br>
173    * You have to define the same number of array items in all pictureSource* properties.
174    * Widths are by default required. To declare an optional width append the "{@literal ?}" suffix, eg. "1440?"<br>
175    * Cannot be used together with image sizes.
176    */
177   @RequestAttribute(injectionStrategy = InjectionStrategy.OPTIONAL)
178   private Object[] pictureSourceWidths;
179 
180   @Self
181   private MediaHandler mediaHandler;
182   @Self
183   private MediaFormatHandler mediaFormatHandler;
184   @SlingObject
185   private Resource resource;
186   @Self
187   private SlingHttpServletRequest request;
188 
189   private Media media;
190 
191   @PostConstruct
192   @SuppressWarnings("null")
193   private void activate() {
194     MediaBuilder builder = mediaHandler.get(resource);
195 
196     if (StringUtils.isNotEmpty(mediaFormat)) {
197       builder.mediaFormatName(mediaFormat);
198     }
199     if (StringUtils.isNotEmpty(refProperty)) {
200       builder.refProperty(refProperty);
201     }
202     if (StringUtils.isNotEmpty(cropProperty)) {
203       builder.cropProperty(cropProperty);
204     }
205     if (StringUtils.isNotEmpty(rotationProperty)) {
206       builder.rotationProperty(rotationProperty);
207     }
208     if (autoCrop != null) {
209       builder.autoCrop(autoCrop);
210     }
211     if (StringUtils.isNotEmpty(cssClass)) {
212       builder.property(PROP_CSS_CLASS, cssClass);
213     }
214 
215     // apply responsive image handling - either via image sizes or picture sources
216     if (StringUtils.isNotEmpty(imageSizes)) {
217       WidthOption[] widthOptionsArray = parseWidths(imageWidths);
218       if (widthOptionsArray != null) {
219         builder.imageSizes(imageSizes, widthOptionsArray);
220       }
221     }
222     else if (pictureSourceMediaFormat != null && pictureSourceMedia != null && pictureSourceWidths != null) {
223       ImageUtils.applyPictureSources(mediaFormatHandler, builder,
224           toStringArray(pictureSourceMediaFormat),
225           toStringArray(pictureSourceMedia),
226           toStringArray(pictureSourceWidths));
227     }
228 
229     setCustomProperties(builder);
230 
231     media = builder.build();
232   }
233 
234   /**
235    * Puts all request attributes that their name starts with the prefix {@value #RA_PROPERTY_PREFIX} into the properties of the media builder
236    * @param builder Media builder
237    */
238   private void setCustomProperties(MediaBuilder builder) {
239     getCustomPropertiesFromRequestAttributes()
240         .forEach((name, value) -> builder.property(name, value));
241   }
242 
243   /**
244    * Gathers all request attributes whose name begins with the prefix {@value #RA_PROPERTY_PREFIX}, strips the prefix to get the property name
245    * and returns a map of property name to request attribute value
246    * @return map of custom properties
247    */
248   @NotNull
249   private Map<String, Object> getCustomPropertiesFromRequestAttributes() {
250     return enumToList(request.getAttributeNames()).stream()
251         .filter(this::isMediaPropAttribute)
252         .filter(this::attributeValueIsNotNull)
253         .collect(Collectors.toMap(this::toPropertyName, request::getAttribute));
254   }
255 
256   @NotNull
257   private List<String> enumToList(@Nullable Enumeration<?> enumeration) {
258     List<String> list = new ArrayList<>();
259 
260     if (enumeration != null) {
261       while (enumeration.hasMoreElements()) {
262         list.add(String.valueOf(enumeration.nextElement()));
263       }
264     }
265 
266     return list;
267   }
268 
269   private boolean isMediaPropAttribute(@NotNull String requestAttributeName) {
270     return PROPERTY_NAME_PATTERN.matcher(requestAttributeName).matches();
271   }
272 
273   private boolean attributeValueIsNotNull(String attributeName) {
274     return Objects.nonNull(request.getAttribute(attributeName));
275   }
276 
277   @NotNull
278   private String toPropertyName(@NotNull String requestAttributeName) {
279     return StringUtils.substringAfter(requestAttributeName, RA_PROPERTY_PREFIX);
280   }
281 
282   /**
283    * For some reason passing in arrays from HTL works only with Object[], not with String[].
284    * Thus, convert it here to String[].
285    *
286    * @param objectArray Array of objects
287    * @return Array with objects converted to strings
288    */
289   private static String[] toStringArray(Object... objectArray) {
290     return Arrays.stream(objectArray)
291         .map(obj -> obj == null ? "" : obj.toString())
292         .toArray(String[]::new);
293   }

295   /**
296    * Returns a {@link Media} object with the metadata of the resolved media.
297    * Result is never null, check for validness with the {@link Media#isValid()} method.
298    * @return Media
299    */
300   public @NotNull Media getMetadata() {
301     return media;
302   }
303 
304   /**
305    * Returns true if the media was resolved successful.
306    * @return Media is valid
307    */
308   public boolean isValid() {
309     return media.isValid();
310   }
311 
312   /**
313    * Returns the XHTML markup for the resolved media object (if valid).
314    * This is in most cases an img element, but may also contain other arbitrary markup.
315    * @return Media markup
316    */
317   public @Nullable String getMarkup() {
318     return media.getMarkup();
319   }
