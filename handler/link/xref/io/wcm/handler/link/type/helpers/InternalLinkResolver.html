88    /**
89     * Check if a given page is valid and acceptable to link upon.
90     * @param page Page
91     * @param options Options
92     * @return true if link is acceptable
93     */
94    @SuppressWarnings("deprecation")
95    public boolean acceptPage(@Nullable Page page, @NotNull InternalLinkResolverOptions options) {
96      if (page == null) {
97        return false;
98      }
99  
100     // check for jcr:content node
101     if (!page.hasContent()) {
102       return false;
103     }
104 
105     // check if page is valid concerning on/off-time (only in publish environment)
106     if (RunMode.isPublish(slingSettings.getRunModes()) && !page.isValid()) {
107       return false;
108     }
109 
110     // check if page is acceptable based on link handler config
111     if (!linkHandlerConfig.isValidLinkTarget(page)) {
112       return false;
113     }
114 
115     return true;
116   }

119    * Resolves a link and stores the result in the link object.
120    * Primary it sets the property "linkReferenceInvalid" and the URL is link resolving was successful.
121    * @param link Link
122    * @param options Options to influence the link resolution process
123    * @return Resolved link object

127     ValueMap props = linkRequest.getResourceProperties();
128 
129     // flag to indicate whether any link reference parameter set
130     boolean referenceSet = false;
131 
132     // first try to get direct link target page
133     Page targetPage = link.getLinkRequest().getPage();
134     if (targetPage != null) {
135       referenceSet = true;
136     }
137 
138     // if no target page is set get internal path that points to target page
139     if (targetPage == null) {
140       String targetPath = props.get(options.getPrimaryLinkRefProperty(), String.class);
141       if (StringUtils.isEmpty(targetPath)) {
142         targetPath = link.getLinkRequest().getReference();
143       }
144       if (StringUtils.isNotEmpty(targetPath)) {
145         referenceSet = true;
146       }
147       targetPage = getTargetPage(targetPath, options);
148     }
149 
150     UrlHandlerConfig resolvingUrlHandlerConfig = urlHandlerConfig;
151     UrlHandler resolvingUrlHandler = urlHandler;
152 
153     // use URL handler from target context for link URL building
154     if (targetPage != null && useTargetContext(options)) {
155       Resource resource = targetPage.getContentResource();
156       resolvingUrlHandlerConfig = AdaptTo.notNull(resource, UrlHandlerConfig.class);
157       resolvingUrlHandler = AdaptTo.notNull(resource, UrlHandler.class);
158     }
159 
160     // if target page is a redirect or integrator page recursively resolve link to which the redirect points to
161     // (skip this redirection if edit mode is active)
162     if (targetPage != null
163         && (linkHandlerConfig.isRedirect(targetPage) || resolvingUrlHandlerConfig.isIntegrator(targetPage))
164         && wcmMode != WCMMode.EDIT) {
165       return recursiveResolveLink(targetPage, link);
166     }
167 
168     // build link url
169     String linkUrl = null;
170     if (targetPage != null) {
171       link.setTargetPage(targetPage);
172 
173       LinkArgs linkArgs = linkRequest.getLinkArgs();
174       String selectors = linkArgs.getSelectors();
175       String fileExtension = StringUtils.defaultString(linkArgs.getExtension(), FileExtension.HTML);
176       String suffix = linkArgs.getSuffix();
177       String queryString = linkArgs.getQueryString();
178       String fragment = linkArgs.getFragment();
179 
180       // optionally override query parameters and fragment from link resource
181       if (queryString == null) {
182         queryString = props.get(LinkNameConstants.PN_LINK_QUERY_PARAM, String.class);
183       }
184       else {
185         queryString = props.get(LinkNameConstants.PN_LINK_QUERY_PARAM, queryString);
186       }
187       if (fragment == null) {
188         fragment = props.get(LinkNameConstants.PN_LINK_FRAGMENT, String.class);
189       }
190       else {
191         fragment = props.get(LinkNameConstants.PN_LINK_FRAGMENT, fragment);
192       }
193 
194       // build link url
195       linkUrl = resolvingUrlHandler.get(targetPage)
196           .selectors(selectors)
197           .extension(fileExtension)
198           .suffix(suffix)
199           .queryString(queryString)
200           .fragment(fragment)
201           .urlMode(linkArgs.getUrlMode())
202           .buildExternalLinkUrl(targetPage);
203     }
204 
205     // mark link as invalid if a reference was set that could not be resolved
206     if (linkUrl == null && referenceSet) {
207       link.setLinkReferenceInvalid(true);
208     }
209 
210     // set link url
211     link.setUrl(linkUrl);
212 
213     return link;
214   }

216   /**
217    * Resolves link of redirect or integrator page. Those pages contain the link reference information in their
218    * content resource (jcr:content node). This information is used to resolve the link.
219    * @param redirectPage Redirect or integrator page
220    * @param link Link metadata
221    * @return Link metadata
222    */

224 
225     // set link reference to content resource of redirect page, keep other parameters
226     LinkRequest linkRequest = link.getLinkRequest();
227     LinkRequestkRequest">LinkRequest redirectLinkRequest = new LinkRequest(
228         redirectPage.getContentResource(),
229         null,
230         linkRequest.getLinkArgs());
231 
232     // check of maximum recursive calls via threadlocal to avoid endless loops, return invalid link if one is detected
233     LinkResolveCounter linkResolveCounter = LinkResolveCounter.get();
234     try {
235       linkResolveCounter.increaseCount();
236 
237       if (linkResolveCounter.isMaximumReached()) {
238         // endless loop detected - set link to invalid link
239         link.setUrl(null);
240         return link;
241       }
242 
243       // resolve link by recursive call to link handler, track recursion count
244       Link resolvedLink = linkHandler.get(redirectLinkRequest).build();
245       resolvedLink.addRedirectPage(redirectPage);
246       return resolvedLink;
247     }
248     finally {
249       linkResolveCounter.decreaseCount();
250     }
251   }

253   /**
254    * Returns the target page for the given internal content link reference.
255    * Checks validity of page.
256    * @param targetPath Repository path
257    * @return Target page or null if target reference is invalid.
258    */
259   private Page getTargetPage(String targetPath, InternalLinkResolverOptions options) {
260 
261     if (StringUtils.isEmpty(targetPath)) {
262       return null;
263     }
264 
265     // Rewrite target to current site context
266     String rewrittenPath;
267     if (options.isRewritePathToContext() && !useTargetContext(options)) {
268       rewrittenPath = urlHandler.rewritePathToContext(SyntheticNavigatableResource.get(targetPath, resourceResolver));
269     }
270     else {
271       rewrittenPath = targetPath;
272     }
273     if (StringUtils.isEmpty(rewrittenPath)) {
274       return null;
275     }
276 
277     // Get target page referenced by target path and check for acceptance
278     Page targetPage = pageManager.getPage(rewrittenPath);
279     if (acceptPage(targetPage, options)) {
280       return targetPage;
281     }
282     else {
283       return null;
284     }
285   }

287   /**
288    * Checks if target context should be used.
289    * @param options Link resolver options
290    * @return true if target context should be used
291    */
292   private boolean useTargetContext(InternalLinkResolverOptions options) {
293     if (options.isUseTargetContext() && !options.isRewritePathToContext()) {
294       return true;
295     }
296     // even is use target context is not activated use it if current page is an experience fragment
297     // otherwise it will be always impossible to resolve internal links
298     else if (currentPage != null && Path.isExperienceFragmentPath(currentPage.getPath())) {
299       return true;
300     }
301     return false;
302   }
