45  /**
46   * Default implementation of {@link io.wcm.handler.link.spi.LinkType} for media links.
47   * Media links are links to media items from media sources
48   * that implement the {@link io.wcm.handler.media.spi.MediaSource} interface.
49   */
50  @Model(adaptables = {
51      SlingHttpServletRequest.class, Resource.class
52  })
53  @ProviderType
54  public final class MediaLinkType extends LinkType {
55  
56    /**
57     * Default root folder f√ºr DAM
58     */
59    private static final String DEFAULT_DAM_ROOT = "/content/dam/";
60  
61    /**
62     * Link type ID
63     */
64    public static final @NotNull String ID = "media";
65  
66    @Self
67    private MediaHandler mediaHandler;
68  
69    /**
70     * @return Link type ID (is stored as identifier in repository)
71     */
72    @Override
73    public @NotNull String getId() {
74      return ID;
75    }
76  
77    @Override
78    public @NotNull String getLabel() {
79      return "Asset";
80    }
81  
82    @Override
83    public String getPrimaryLinkRefProperty() {
84      return LinkNameConstants.PN_LINK_MEDIA_REF;
85    }
86  
87    @Override
88    public @Nullable String getEditComponentResourceType() {
89      return "wcm-io/handler/link/components/granite/form/linktype/media";
90    }
91  
92    @Override
93    public boolean hasRichTextPlugin() {
94      return true;
95    }
96  
97    @Override
98    public boolean accepts(@NotNull String linkRef) {
99      // accept as media link if the ref is inside default media subtrees
100     return MediaLinkType.isDefaultMediaContentPath(linkRef);
101   }
102 
103   @SuppressWarnings("null")
104   @Override
105   public @NotNull Link resolveLink(@NotNull Link link) {
106     LinkRequest linkRequest = link.getLinkRequest();
107     ValueMap props = linkRequest.getResourceProperties();
108 
109     // get properties
110     String mediaRef = props.get(LinkNameConstants.PN_LINK_MEDIA_REF, link.getLinkRequest().getReference());
111     boolean isDownload = props.get(LinkNameConstants.PN_LINK_MEDIA_DOWNLOAD, false);
112 
113     MediaArgs mediaArgs = new MediaArgs()
114         // only allow linking to "download" media formats
115         .download(true)
116         .contentDispositionAttachment(isDownload)
117         .urlMode(linkRequest.getLinkArgs().getUrlMode());
118 
119     // resolve media library reference
120     Media media = mediaHandler.get(mediaRef, mediaArgs).build();
121 
122     // set resovled media references information in link metadata
123     link.setUrl(media.getUrl());
124     link.setTargetAsset(media.getAsset());
125     link.setTargetRendition(media.getRendition());
126 
127     // mark link as invalid if a reference was set that could not be resolved
128     if (link.getUrl() == null && StringUtils.isNotEmpty(mediaRef)) {
129       link.setLinkReferenceInvalid(true);
130     }
131 
132     return link;
133   }
134 
135   /**
136    * @param path Content path
137    * @return true if Path is located below DAM default root folders.
138    */
139   public static boolean isDefaultMediaContentPath(String path) {
140     return StringUtils.startsWith(path, DEFAULT_DAM_ROOT);
141   }
142 
143   /**
144    * Get synthetic link resource for this link type.
145    * @param resourceResolver Resource resolver
146    * @param path Resource path. Can be a non-existing path, but the path should be located somewhere within the
147    *          applications content paths to make sure the handler configuration looked up via context-aware services
148    *          is the expected one.
149    * @param mediaRef Media asset reference
150    * @return Synthetic link resource
151    */
152   public static @NotNull Resource getSyntheticLinkResource(@NotNull ResourceResolver resourceResolver,
153       @NotNull String path, @NotNull String mediaRef) {
154     Map<String, Object> map = new HashMap<>();
155     map.put(LinkNameConstants.PN_LINK_TYPE, ID);
156     map.put(LinkNameConstants.PN_LINK_MEDIA_REF, mediaRef);
157     return new SyntheticLinkResource(resourceResolver, path, map);
158   }
159 
160   @Override
161   public String toString() {
162     return ID;
163   }
164 
165 }
