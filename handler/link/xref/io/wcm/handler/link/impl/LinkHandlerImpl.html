51  /**
52   * Default implementation of a {@link LinkHandler}
53   */
54  @Model(adaptables = {
55      SlingHttpServletRequest.class, Resource.class
56  }, adapters = LinkHandler.class)
57  public final class LinkHandlerImpl implements LinkHandler {
58  
59    @Self
60    private Adaptable adaptable;
61    @Self
62    private LinkHandlerConfig linkHandlerConfig;
63    @AemObject(injectionStrategy = InjectionStrategy.OPTIONAL)
64    private Page currentPage;
65    @OSGiService(injectionStrategy = InjectionStrategy.OPTIONAL)
66    private ComponentPropertyResolverFactory componentPropertyResolverFactory;

88    /**
89     * Resolves the link
90     * @param linkRequest Link request
91     * @return Link metadata (never null)
92     */
93    @NotNull
94    @SuppressWarnings({ "null", "unused" })
95    @SuppressFBWarnings({ "CORRECTNESS", "STYLE" })
96    Link processRequest(@NotNull LinkRequest linkRequest) {
97  
98      // detect link type - first accepting wins
99      LinkType linkType = null;
100     List<Class<? extends LinkType>> linkTypes = linkHandlerConfig.getLinkTypes();
101     if (linkTypes == null || linkTypes.isEmpty()) {
102       throw new RuntimeException("No link types defined.");
103     }
104     for (Class<? extends LinkType> candidateLinkTypeClass : linkTypes) {
105       LinkType candidateLinkType = AdaptTo.notNull(adaptable, candidateLinkTypeClass);
106       if (candidateLinkType.accepts(linkRequest)) {
107         linkType = candidateLinkType;
108         break;
109       }
110     }
111     Link link = new Link(linkType, linkRequest);
112 
113     // preprocess link before resolving
114     List<Class<? extends LinkProcessor>> linkPreProcessors = linkHandlerConfig.getPreProcessors();
115     if (linkPreProcessors != null) {
116       for (Class<? extends LinkProcessor> processorClass : linkPreProcessors) {
117         LinkProcessor processor = AdaptTo.notNull(adaptable, processorClass);
118         link = processor.process(link);
119         if (link == null) {
120           throw new RuntimeException("LinkPreProcessor '" + processor + "' returned null, page '" + (currentPage != null ? currentPage.getPath() : "-") + "'.");
121         }
122       }
123     }
124 
125     // resolve link
126     if (linkType != null) {
127       link = linkType.resolveLink(link);
128       if (link == null) {
129         throw new RuntimeException("LinkType '" + linkType + "' returned null, page '" + (currentPage != null ? currentPage.getPath() : "-") + "'.");
130       }
131     }
132 
133     // if link is invalid - check if a fallback link property is set and try resolution with it
134     if (!link.isValid()) {
135       LinkRequest fallbackLinkRequest = getFallbackLinkRequest(linkRequest);
136       if (fallbackLinkRequest != null) {
137         Link fallbackLink = processRequest(fallbackLinkRequest);
138         if (fallbackLink.isValid()) {
139           return fallbackLink;
140         }
141       }
142     }
143 
144     // generate markup (if markup builder is available) - first accepting wins
145     List<Class<? extends LinkMarkupBuilder>> linkMarkupBuilders = linkHandlerConfig.getMarkupBuilders();
146     if (linkMarkupBuilders != null) {
147       for (Class<? extends LinkMarkupBuilder> linkMarkupBuilderClass : linkMarkupBuilders) {
148         LinkMarkupBuilder linkMarkupBuilder = AdaptTo.notNull(adaptable, linkMarkupBuilderClass);
149         if (linkMarkupBuilder.accepts(link)) {
150           link.setAnchor(linkMarkupBuilder.build(link));
151           break;
152         }
153       }
154     }
155 
156     // postprocess link after resolving
157     List<Class<? extends LinkProcessor>> linkPostProcessors = linkHandlerConfig.getPostProcessors();
158     if (linkPostProcessors != null) {
159       for (Class<? extends LinkProcessor> processorClass : linkPostProcessors) {
160         LinkProcessor processor = AdaptTo.notNull(adaptable, processorClass);
161         link = processor.process(link);
162         if (link == null) {
163           throw new RuntimeException("LinkPostProcessor '" + processor + "' returned null, page '" + (currentPage != null ? currentPage.getPath() : "-") + "'.");
164         }
165       }
166     }
167 
168     return link;
169   }
170 
171   @Override
172   public Link invalid() {
173     // build invalid link with first link type
174     Class<? extends LinkType> linkTypeClass = linkHandlerConfig.getLinkTypes().stream().findFirst().orElse(null);
175     if (linkTypeClass == null) {
176       throw new RuntimeException("No link types defined.");
177     }
178     LinkType linkType = AdaptTo.notNull(adaptable, linkTypeClass);
179     return new Link(linkType, new LinkRequest(null, null, null));
180   }
181 
182   /**
183    * Checks if a link target URL is defined in a fallback property and prepare a link request
184    * to try to resolve this as link instead.
185    * @param linkRequest Original link request
186    * @return Fallback link request or null
187    */
188   private @Nullable LinkRequest getFallbackLinkRequest(@NotNull LinkRequest linkRequest) {
189     Resource resource = linkRequest.getResource();
190 
191     // works only when resolution based on a resource
192     if (resource == null) {
193       return null;
194     }
195 
196     // check if a fallback property name was given
197     String[] linkTargetUrlFallbackProperty = linkRequest.getLinkArgs().getLinkTargetUrlFallbackProperty();
198     if (linkTargetUrlFallbackProperty == null || linkTargetUrlFallbackProperty.length == 0) {
199       return null;
200     }
201 
202     // check if a link target URL is set in the fallback property
203     String linkTargetUrl = null;
204     for (String propertyName : linkTargetUrlFallbackProperty) {
205       linkTargetUrl = resource.getValueMap().get(propertyName, String.class);
206       if (StringUtils.isNotBlank(linkTargetUrl)) {
207         break;
208       }
209     }
210     if (StringUtils.isBlank(linkTargetUrl)) {
211       return null;
212     }
213 
214     LinkArgs fallbackLinkArgs = linkRequest.getLinkArgs().clone();
215     @NotNull
216     String @Nullable [] nullArray = null;
217     fallbackLinkArgs.linkTargetUrlFallbackProperty(nullArray);
218     return new LinkRequest(null, null, linkTargetUrl, fallbackLinkArgs);
219   }
