20  package io.wcm.handler.richtext.util;
21  
22  import org.jetbrains.annotations.NotNull;
23  
24  /**
25   * Definition of XHTML resources.
26   */
27  enum XHtmlResource {
28  
29    /**
30     * DTD XHTML 1.0 Strict
31     */
32    XHTML1_STRICT("-//W3C//DTD XHTML 1.0 Strict//EN", "http://www.w3.org/TR/xhtml1/DTD/xhtml1-strict.dtd", "xhtml1-strict.dtd"),
33  
34    /**
35     * DTD XHTML 1.0 Transitional
36     */
37    XHTML1_TRANSITIONAL("-//W3C//DTD XHTML 1.0 Transitional//EN", "http://www.w3.org/TR/xhtml1/DTD/xhtml1-transitional.dtd", "xhtml1-transitional.dtd"),
38  
39    /**
40     * DTD XHTML 1.0 Frameset
41     */
42    XHTML1_FRAMESET("-//W3C//DTD XHTML 1.0 Frameset//EN", "http://www.w3.org/TR/xhtml1/DTD/xhtml1-frameset.dtd", "xhtml1-frameset.dtd"),
43  
44    /**
45     * ENTITIES Latin 1 for XHTML
46     */
47    ENTITIES_LAT1("-//W3C//ENTITIES Latin 1 for XHTML//EN", "xhtml-lat1.ent", "xhtml-lat1.ent"),
48  
49    /**
50     * ENTITIES Symbols for XHTML
51     */
52    ENTITIES_SYMBOL("-//W3C//ENTITIES Symbols for XHTML//EN", "xhtml-symbol.ent", "xhtml-symbol.ent"),
53  
54    /**
55     * ENTITIES Special for XHTML
56     */
57    ENTITIES_SPECIAL("-//W3C//ENTITIES Special for XHTML//EN", "xhtml-special.ent", "xhtml-special.ent");
58  
59    private final String publicId;
60    private final String systemId;
61    private final String filename;
62  
63    XHtmlResource(String publicId, String systemId, String filename) {
64      this.publicId = publicId;
65      this.systemId = systemId;
66      this.filename = filename;
67    }
68  
69    /**
70     * Public Id
71     * @return Public Id
72     */
73    public @NotNull String getPublicId() {
74      return this.publicId;
75    }
76  
77    /**
78     * System Id
79     * @return System Id
80     */
81    public @NotNull String getSystemId() {
82      return this.systemId;
83    }
84  
85    /**
86     * Local filename
87     * @return Local filename
88     */
89    public @NotNull String getFilename() {
90      return this.filename;
91    }
92  
93  }
