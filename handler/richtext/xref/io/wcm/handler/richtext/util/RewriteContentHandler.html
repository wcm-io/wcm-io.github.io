20  package io.wcm.handler.richtext.util;
21  
22  import java.util.List;
23  
24  import org.jdom2.Content;
25  import org.jdom2.Element;
26  import org.jdom2.Text;
27  import org.jetbrains.annotations.NotNull;
28  import org.jetbrains.annotations.Nullable;
29  import org.osgi.annotation.versioning.ConsumerType;
30  
31  /**
32   * Allows to rewrite DOM elements and text elements.
33   * <p>
34   * If used for {@link io.wcm.handler.richtext.spi.RichTextHandlerConfig} this interface has to be
35   * implemented by a Sling Model class. The adaptables should be
36   * {@link org.apache.sling.api.SlingHttpServletRequest} and {@link org.apache.sling.api.resource.Resource}.
37   * </p>
38   */
39  @ConsumerType
40  public interface RewriteContentHandler {
41  
42    /**
43     * Checks if the given element has to be rewritten.
44     * Is called for every child single element of the parent given to rewriteContent method.
45     * @param element Element to check
46     * @return null if nothing is to do with this element.
47     *         Return empty list to remove this element.
48     *         Return list with other content to replace element with new content.
49     */
50    @Nullable
51    List<Content> rewriteElement(@NotNull Element element);
52  
53    /**
54     * Checks if the given text node has to be rewritten.
55     * Is called for every text node found as parent of given to rewriteContent method.
56     * @param text Text node
57     * @return null if nothing is to do with this element.
58     *         Return empty list to remove the element.
59     *         Return list with other content to replace element with new content.
60     */
61    @Nullable
62    List<Content> rewriteText(@NotNull Text text);
63  
64  }
