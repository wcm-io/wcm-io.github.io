20  package io.wcm.sling.commons.caservice;
21  
22  import org.jetbrains.annotations.NotNull;
23  import org.osgi.annotation.versioning.ConsumerType;
24  
25  /**
26   * Marker interface for Context-Aware services.
27   * <p>
28   * If multiple implementations of an OSGi service interface or class exists the {@link ContextAwareServiceResolver} can
29   * be used to picks the implementation with the best match for the current resource context/path in the resource
30   * hierarchy.
31   * </p>
32   * <p>
33   * This interface has no methods, the metadata for detecting the matching paths is published via the properties
34   * declared by the constants of this interface. The properties can be set as service properties or as bundle header.
35   * Service property is checked first, if it is not set there the bundle header of the bundle which registered the
36   * service is checked as fallback.
37   * </p>
38   * <p>
39   * If multiple service implementations exist which match for a given resource path the one with the highest service
40   * ranking is picked.
41   * </p>
42   */
43  //CHECKSTYLE:OFF
44  @ConsumerType
45  @SuppressWarnings({ "PMD.ConstantsInInterface", "java:S1214" })
46  public interface ContextAwareService {
47  //CHECKSTYLE:ON
48  
49    /**
50     * Context path whitelist expression.
51     * Service property or bundle header defining a regular expression the context resource path is matched against.
52     * Only if the path is matching the service is considered as candidate.
53     * If the property is not set all resource paths are accepted.
54     */
55    @NotNull
56    String PROPERTY_CONTEXT_PATH_PATTERN = "Wcmio-CAService-ContextPathRegex";
57  
58    /**
59     * Context path blacklist expression.
60     * Service property or bundle header defining a regular expression the context resource path is matched against.
61     * If the path is matching the service is not considered as candidate.
62     * If the property is not set no resource paths are blacklisted.
63     */
64    @NotNull
65    String PROPERTY_CONTEXT_PATH_BLACKLIST_PATTERN = "Wcmio-CAService-ContextPathBlacklistRegex";
66  
67    /**
68     * Accepts empty context paths.
69     * Service property or bundle header that can be set to "true" to signal that this service implementation also should
70     * be considered as candidate if no context resource exists and thus the context path is empty (null).
71     * If the property is not set the implementation is not considered as candidate for empty resource paths.
72     */
73    @NotNull
74    String PROPERTY_ACCEPTS_CONTEXT_PATH_EMPTY = "Wcmio-CASService-AcceptsContextPathEmpty";
75  
76  }
