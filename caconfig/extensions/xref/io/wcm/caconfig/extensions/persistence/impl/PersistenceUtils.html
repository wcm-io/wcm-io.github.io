68    public static boolean containsJcrContent(String path) {
69      if (path == null) {
70        return false;
71      }
72      return JCR_CONTENT_PATTERN.matcher(path).matches();

76     * Ensure that a containing page exists for the given path inside a content page.
77     * If no containing page exists a page is created with the path before /jcr:content/*.
78     * If the path does not contain /jcr:content nothing is done.
79     * @param resolver Resource resolver
80     * @param configResourcePath Configuration resource path
81     * @param resourceType Resource type for page (if not template is set)
82     * @param configurationManagementSettings Configuration management settings
83     */
84    public static void ensureContainingPage(ResourceResolver resolver, String configResourcePath,
85        String resourceType, ConfigurationManagementSettings configurationManagementSettings) {
86      ensureContainingPage(resolver, configResourcePath, null, resourceType, null, configurationManagementSettings);
87    }

89    /**
90     * Ensure that a containing page exists for the given path inside a content page.
91     * If no containing page exists a page is created with the path before /jcr:content/*.
92     * If the path does not contain /jcr:content nothing is done.
93     * @param resolver Resource resolver
94     * @param configResourcePath Configuration resource path
95     * @param template Template for page
96     * @param resourceType Resource type for page (if not template is set)
97     * @param parentTemplate Template for parent/intermediate pages
98     * @param configurationManagementSettings Configuration management settings
99     */
100   @SuppressWarnings("PMD.UseObjectForClearerAPI")
101   public static void ensureContainingPage(ResourceResolver resolver, String configResourcePath,
102       String template, String resourceType, String parentTemplate,
103       ConfigurationManagementSettings configurationManagementSettings) {
104     Matcher matcher = PAGE_PATH_PATTERN.matcher(configResourcePath);
105     if (!matcher.matches()) {
106       return;
107     }
108     String pagePath = matcher.group(1);
109     ensurePage(resolver, pagePath, template, resourceType, parentTemplate, configurationManagementSettings);
110   }

112   /**
113    * Ensure that a page at the given path exists, if the path is not already contained in a page.
114    * @param resolver Resource resolver
115    * @param pagePath Page path
116    * @param resourceType Resource type for page (if not template is set)
117    * @param configurationManagementSettings Configuration management settings
118    * @return Resource for AEM page or resource inside a page.
119    */
120   public static Resource ensurePageIfNotContainingPage(ResourceResolver resolver, String pagePath,
121       String resourceType, ConfigurationManagementSettings configurationManagementSettings) {
122     Matcher matcher = PAGE_PATH_PATTERN.matcher(pagePath);
123     if (matcher.matches()) {
124       // ensure that shorted path part that ends with /jcr:content is created as AEM page (if not existent already)
125       String detectedPagePath = matcher.group(1);
126       ensurePage(resolver, detectedPagePath, null, resourceType, null, configurationManagementSettings);
127       return getOrCreateResource(resolver, pagePath, DEFAULT_FOLDER_NODE_TYPE_IN_PAGE, null, configurationManagementSettings);
128     }
129     return ensurePage(resolver, pagePath, null, resourceType, null, configurationManagementSettings);
130   }

132   private static Resource ensurePage(ResourceResolver resolver, String pagePath,
133       String template, String resourceType, String parentTemplate,
134       ConfigurationManagementSettings configurationManagementSettings) {
135     // check if page or resource already exists
136     Resource resource = resolver.getResource(pagePath);
137     if (resource != null) {
138       return resource;
139     }
140 
141     // ensure parent page or resource exists
142     String parentPath = ResourceUtil.getParent(pagePath);
143     String pageName = ResourceUtil.getName(pagePath);
144     Resource parentResource;
145     if (StringUtils.isNotEmpty(parentTemplate)) {
146       parentResource = ensurePage(resolver, parentPath, parentTemplate, null, parentTemplate, configurationManagementSettings);
147     }
148     else {
149       parentResource = getOrCreateResource(resolver, parentPath, DEFAULT_FOLDER_NODE_TYPE, null, configurationManagementSettings);
150     }
151 
152     // create page
153     return createPage(resolver, parentResource, pageName, template, resourceType);
154   }

156   private static Resource createPage(ResourceResolver resolver, Resource parentResource, String pageName,
157       String template, String resourceType) {
158     String pagePath = parentResource.getPath() + "/" + pageName;
159     log.trace("! Create cq:Page node at {}", pagePath);
160     try {
161       // create page directly via Sling API instead of PageManager because page name may contain dots (.)
162       Map<String, Object> props = new HashMap<>();
163       props.put(JcrConstants.JCR_PRIMARYTYPE, NameConstants.NT_PAGE);
164       Resource pageResource = resolver.create(parentResource, pageName, props);
165 
166       // create jcr:content node
167       props = new HashMap<>();
168       props.put(JcrConstants.JCR_PRIMARYTYPE, "cq:PageContent");
169       if (StringUtils.isNotEmpty(template)) {
170         applyPageTemplate(resolver, props, pageName, template);
171       }
172       if (StringUtils.isNotEmpty(resourceType) && props.get(PROPERTY_RESOURCE_TYPE) == null) {
173         props.put(PROPERTY_RESOURCE_TYPE, resourceType);
174       }
175       resolver.create(pageResource, JCR_CONTENT, props);
176 
177       return pageResource;
178     }
179     catch (PersistenceException ex) {
180       throw convertPersistenceException("Unable to create config page at " + pagePath, ex);
181     }
182   }
183 
184   private static void applyPageTemplate(ResourceResolver resolver, Map<String, Object> props, String pageName, String template) {
185     // set template
186     props.put(NameConstants.PN_TEMPLATE, template);
187 
188     // also set title for author when template is set
189     props.put(JcrConstants.JCR_TITLE, pageName);
190 
191     // get sling:resourceType from template definition
192     Resource templateContentResource = resolver.getResource(template + "/" + JCR_CONTENT);
193     if (templateContentResource != null) {
194       props.put(PROPERTY_RESOURCE_TYPE, templateContentResource.getValueMap().get(PROPERTY_RESOURCE_TYPE, String.class));
195     }
196   }

198   public static Resource getOrCreateResource(ResourceResolver resolver, String path, String defaultNodeType, Map<String, Object> properties,
199       ConfigurationManagementSettings configurationManagementSettings) {
200     try {
201       Resource resource = ResourceUtil.getOrCreateResource(resolver, path, defaultNodeType, defaultNodeType, false);
202       if (properties != null) {
203         replaceProperties(resource, properties, configurationManagementSettings);
204       }
205       return resource;
206     }
207     catch (PersistenceException ex) {
208       throw convertPersistenceException("Unable to create resource at " + path, ex);
209     }
210   }

212   /**
213    * Delete children that are no longer contained in list of collection items.
214    * @param resource Parent resource
215    * @param data List of collection items
216    */
217   public static void deleteChildrenNotInCollection(Resource resource, ConfigurationCollectionPersistData data) {
218     Set<String> collectionItemNames = data.getItems().stream()
219         .map(ConfigurationPersistData::getCollectionItemName)
220         .collect(Collectors.toSet());
221 
222     for (Resource child : resource.getChildren()) {
223       if (!collectionItemNames.contains(child.getName()) && !StringUtils.equals(JCR_CONTENT, child.getName())) {
224         deletePageOrResource(child);
225       }
226     }
227   }

229   public static void replaceProperties(Resource resource, Map<String, Object> properties,
230       ConfigurationManagementSettings configurationManagementSettings) {
231     if (log.isTraceEnabled()) {
232       log.trace("! Store properties for resource {}: {}", resource.getPath(), properties);
233     }
234     ModifiableValueMap modValueMap = resource.adaptTo(ModifiableValueMap.class);
235     if (modValueMap == null) {
236       throw new ConfigurationPersistenceAccessDeniedException("No write access: Unable to store configuration data to " + resource.getPath() + ".");
237     }
238 
239     // remove all existing properties that are not filtered
240     Set<String> propertyNamesToRemove = new HashSet<>(modValueMap.keySet());
241     PropertiesFilterUtil.removeIgnoredProperties(propertyNamesToRemove, configurationManagementSettings);
242     for (String propertyName : propertyNamesToRemove) {
243       modValueMap.remove(propertyName);
244     }
245 
246     modValueMap.putAll(properties);
247   }

249   public static void updatePageLastMod(ResourceResolver resolver, PageManager pageManager, String configResourcePath) {
250     Page page = pageManager.getContainingPage(configResourcePath);
251     if (page == null) {
252       return;
253     }
254     Resource contentResource = page.getContentResource();
255     if (contentResource != null) {
256       ModifiableValueMap contentProps = contentResource.adaptTo(ModifiableValueMap.class);
257       if (contentProps == null) {
258         throw new ConfigurationPersistenceAccessDeniedException("No write access: Unable to update page " + configResourcePath + ".");
259       }
260 
261       Object user = resolver.getAttribute(ResourceResolverFactory.USER);
262       Calendar now = Calendar.getInstance();
263 
264       contentProps.put(NameConstants.PN_LAST_MOD, now);
265       contentProps.put(NameConstants.PN_LAST_MOD_BY, user);
266 
267       // check if resource has cq:lastModified because it is created in site admin
268       if (contentProps.containsKey(NameConstants.PN_PAGE_LAST_MOD)) {
269         contentProps.put(NameConstants.PN_PAGE_LAST_MOD, now);
270         contentProps.put(NameConstants.PN_PAGE_LAST_MOD_BY, user);
271       }
272     }
273   }

276     try {
277       resourceResolver.commit();

284   /**
285    * Checks if the given item is modified or newly added by comparing its properties with the current state of the resource.
286    *
287    * @param resolver     The ResourceResolver to access the resource.
288    * @param resourcePath The path of the resource to compare against.
289    * @param item         The ConfigurationPersistData item containing the properties to compare.
290    * @param settings     The ConfigurationManagementSettings to determine which properties to ignore.
291    * @return true if the resource does not exist or if any property value differs, false otherwise.
292    */
293   public static boolean isItemModifiedOrNewlyAdded(ResourceResolver resolver, String resourcePath, ConfigurationPersistData item, ConfigurationManagementSettings settings) {
294     Resource resource = resolver.getResource(resourcePath);
295     if (resource == null) {
296       return true; // Resource does not exist, so it is considered modified
297     }
298 
299     Map<String, Object> currentProperties = new HashMap<>(resource.getValueMap());
300     Map<String, Object> newProperties = new HashMap<>(item.getProperties());
301 
302     // Filter out ignored properties
303     PropertiesFilterUtil.removeIgnoredProperties(currentProperties, settings);
304     PropertiesFilterUtil.removeIgnoredProperties(newProperties, settings);
305 
306     return !currentProperties.equals(newProperties);
307   }

309   /**
310    * If the given resource points to an AEM page, delete the page using PageManager.
311    * Otherwise delete the resource using ResourceResolver.
312    * @param resource Resource to delete
313    */
314   public static void deletePageOrResource(Resource resource) {
315     Page configPage = resource.adaptTo(Page.class);
316     if (configPage != null) {
317       try {
318         log.trace("! Delete page {}", configPage.getPath());
319         PageManager pageManager = configPage.getPageManager();
320         pageManager.delete(configPage, false);
321       }
322       catch (WCMException ex) {
323         throw convertWCMException("Unable to delete configuration page at " + resource.getPath(), ex);
324       }
325     }
326     else {
327       try {
328         log.trace("! Delete resource {}", resource.getPath());
329         resource.getResourceResolver().delete(resource);
330       }
331       catch (PersistenceException ex) {
332         throw convertPersistenceException("Unable to delete configuration resource at " + resource.getPath(), ex);
333       }
334     }
335   }

337   private static ConfigurationPersistenceException convertWCMException(String message, WCMException ex) {
338     String causeClsName = ex.getCause().getClass().getName();
339     if (StringUtils.equals(causeClsName, "com.day.cq.replication.AccessDeniedException")
340         || StringUtils.equals(causeClsName, "javax.jcr.AccessDeniedException")) {
341       return new ConfigurationPersistenceAccessDeniedException("No write access: " + message, ex);
342     }
343     return new ConfigurationPersistenceException(message, ex);
344   }
345 
346   private static ConfigurationPersistenceException convertPersistenceException(String message, PersistenceException ex) {
347     if (StringUtils.equals(ex.getCause().getClass().getName(), "javax.jcr.AccessDeniedException")) {
348       // detect if commit failed due to read-only access to repository
349       return new ConfigurationPersistenceAccessDeniedException("No write access: " + message, ex);
350     }
351     return new ConfigurationPersistenceException(message, ex);
352   }
