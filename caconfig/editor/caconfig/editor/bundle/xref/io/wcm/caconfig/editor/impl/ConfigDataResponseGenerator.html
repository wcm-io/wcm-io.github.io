88    ConfigDataResponseGenerator(@NotNull SlingHttpServletRequest request,
89        @NotNull ConfigurationManager configManager,
90        @NotNull ConfigurationPersistenceStrategyMultiplexer configurationPersistenceStrategy,
91        @NotNull DropdownOptionProviderService dropdownOptionProviderService,
92        @NotNull PathBrowserRootPathProviderService pathBrowserRootPathProviderService,
93        @NotNull TagBrowserRootPathProviderService tagBrowserRootPathProviderService) {
94      this.configManager = configManager;
95      this.configurationPersistenceStrategy = configurationPersistenceStrategy;
96      this.dropdownOptionProviderService = dropdownOptionProviderService;
97      this.pathBrowserRootPathProviderService = pathBrowserRootPathProviderService;
98      this.tagBrowserRootPathProviderService = tagBrowserRootPathProviderService;
99  
100     Session session = request.getResourceResolver().adaptTo(Session.class);
101     if (session != null) {
102       try {
103         this.accessControlManager = session.getAccessControlManager();
104         this.jcrWritePrivilege = accessControlManager.privilegeFromName(Privilege.JCR_WRITE);
105       }
106       catch (RepositoryException ex) {
107         log.warn("Unable to prepare JCR AccessControlManager.", ex);
108       }
109     }
110   }

112   Object getConfiguration(@NotNull Resource contextResource, String configName, boolean collection) {
113     Object result;
114     if (collection) {
115       ConfigurationData newItem = configManager.newCollectionItem(contextResource, configName);
116       if (newItem == null) {
117         throw new ConfigurationPersistenceException("Invalid configuration name: " + configName);
118       }
119       result = fromConfigCollection(contextResource,
120           configManager.getConfigurationCollection(contextResource, configName), newItem, configName);
121     }
122     else {
123       ConfigurationData configData = configManager.getConfiguration(contextResource, configName);
124       if (configData != null) {
125         result = fromConfig(contextResource, configData, configData.isInherited(), configName);
126       }
127       else {
128         result = null;
129       }
130     }
131     return result;
132   }

134   private ConfigCollectionItem fromConfigCollection(@NotNull Resource contextResource,
135       ConfigurationCollectionData configCollection, ConfigurationData newItem, String fullConfigName) {
136     ConfigCollectionItem result = new ConfigCollectionItem();
137     result.setConfigName(configCollection.getConfigName());
138     result.setConfigSourcePath(configCollection.getResourcePath());
139     result.setReadOnly(isReadOnly(configCollection.getResourcePath(), false));
140 
141     if (!configCollection.getProperties().isEmpty()) {
142       Map<String, Object> properties = new TreeMap<>();
143       for (Map.Entry<String, Object> entry : configCollection.getProperties().entrySet()) {
144         properties.put(entry.getKey(), entry.getValue());
145       }
146       result.setProperties(properties);
147     }
148 
149     List<ConfigItem> items = new ArrayList<>();
150     for (ConfigurationData configData : configCollection.getItems()) {
151       items.add(fromConfig(contextResource, configData, configData.isInherited(), fullConfigName));
152     }
153     result.setItems(items);
154 
155     result.setNewItem(fromConfig(contextResource, newItem, null, fullConfigName));
156 
157     return result;
158   }

160   @SuppressWarnings("java:S3776")
161   private ConfigItem fromConfig(@NotNull Resource contextResource, ConfigurationData config, Boolean inherited, String fullConfigName) {
162     ConfigItem result = new ConfigItem();
163 
164     result.setConfigName(config.getConfigName());
165     result.setCollectionItemName(config.getCollectionItemName());
166     result.setOverridden(config.isOverridden());
167     result.setInherited(inherited);
168     result.setConfigSourcePath(config.getResourcePath());
169     result.setReadOnly(isReadOnly(config.getResourcePath(), config.isOverridden()));
170 
171     List<PropertyItem> props = new ArrayList<>();
172     for (String propertyName : config.getPropertyNames()) {
173       ValueInfo<?> item = config.getValueInfo(propertyName);
174       if (item == null) {
175         continue;
176       }
177       PropertyMetadata<?> itemMetadata = item.getPropertyMetadata();
178 
179       PropertyItem prop = new PropertyItem();
180       prop.setName(item.getName());
181 
182       // special handling for nested configs and nested config collections
183       if (itemMetadata != null && itemMetadata.isNestedConfiguration()) {
184         PropertyItemMetadata metadata = new PropertyItemMetadata();
185         metadata.setLabel(itemMetadata.getLabel());
186         metadata.setDescription(itemMetadata.getDescription());
187         metadata.setProperties(toJsonWithValueConversion(itemMetadata.getProperties(), contextResource));
188         prop.setMetadata(metadata);
189 
190         if (itemMetadata.getType().isArray()) {
191           ConfigurationData[] configDatas = (ConfigurationData[])item.getValue();
192           if (configDatas != null) {
193             ConfigCollectionItem nestedConfigCollection = new ConfigCollectionItem();
194             StringBuilder collectionConfigName = new StringBuilder();
195             if (config.getCollectionItemName() != null) {
196               collectionConfigName.append(configurationPersistenceStrategy.getCollectionItemConfigName(fullConfigName
197                   + "/" + config.getCollectionItemName(), config.getResourcePath()));
198             }
199             else {
200               collectionConfigName.append(configurationPersistenceStrategy.getConfigName(fullConfigName, config.getResourcePath()));
201             }
202             collectionConfigName.append("/").append(itemMetadata.getConfigurationMetadata().getName());
203             nestedConfigCollection.setConfigName(collectionConfigName.toString());
204             List<ConfigItem> items = new ArrayList<>();
205             for (ConfigurationData configData : configDatas) {
206               items.add(fromConfig(contextResource, configData, false, collectionConfigName.toString()));
207             }
208             nestedConfigCollection.setItems(items);
209             prop.setNestedConfigCollection(nestedConfigCollection);
210           }
211         }
212         else {
213           ConfigurationData configData = (ConfigurationData)item.getValue();
214           if (configData != null) {
215             prop.setNestedConfig(fromConfig(contextResource, configData, null, fullConfigName
216                 + "/" + itemMetadata.getConfigurationMetadata().getName()));
217           }
218         }
219       }
220 
221       // property data and metadata
222       else {
223         prop.setValue(item.getValue());
224         prop.setEffectiveValue(item.getEffectiveValue());
225         prop.setConfigSourcePath(item.getConfigSourcePath());
226         prop.setIsDefault(item.isDefault());
227         prop.setInherited(item.isInherited());
228         prop.setOverridden(item.isOverridden());
229         prop.setReadOnly(result.getReadOnly());
230 
231         if (itemMetadata != null) {
232           PropertyItemMetadata metadata = new PropertyItemMetadata();
233           if (itemMetadata.getType().isArray()) {
234             metadata.setType(ClassUtils.primitiveToWrapper(itemMetadata.getType().getComponentType()).getSimpleName());
235             metadata.setMultivalue(true);
236           }
237           else {
238             metadata.setType(ClassUtils.primitiveToWrapper(itemMetadata.getType()).getSimpleName());
239           }
240           metadata.setDefaultValue(itemMetadata.getDefaultValue());
241           metadata.setLabel(itemMetadata.getLabel());
242           metadata.setDescription(itemMetadata.getDescription());
243           metadata.setProperties(toJsonWithValueConversion(itemMetadata.getProperties(), contextResource));
244           prop.setMetadata(metadata);
245         }
246       }
247       props.add(prop);
248     }
249     result.setProperties(props);
250 
251     return result;
252   }

254   /**
255    * Converts the given map to JSON. Each map value is checked for a valid JSON string - if this is the case it's
256    * inserted as JSON objects and not as string.
257    * @param properties Map
258    * @param contextResource Context resource
259    * @return JSON object or null
260    */
261   @SuppressWarnings({ "PMD.ReturnEmptyCollectionRatherThanNull", "java:S3776" })
262   private @Nullable Map<String, Object> toJsonWithValueConversion(@Nullable Map<String, String> properties,
263       @NotNull Resource contextResource) {
264     if (properties == null || properties.isEmpty()) {
265       return null;
266     }
267 
268     Map<String, Object> metadataProps = new TreeMap<>();
269     for (Map.Entry<String, String> entry : properties.entrySet()) {
270       metadataProps.put(entry.getKey(), tryConvertJsonString(entry.getValue()));
271     }
272 
273     // check for dynamic dropdown option injection
274     boolean isDropdown = WIDGET_TYPE_DROPDOWN.equals(metadataProps.get(PROPERTY_WIDGET_TYPE));
275     if (isDropdown) {
276       Optional<String> dynamicProvider = Optional.ofNullable(metadataProps.get(PROPERTY_DROPDOWN_OPTIONS_PROVIDER))
277           .filter(Objects::nonNull)
278           .map(String::valueOf)
279           .filter(StringUtils::isNotBlank);
280       if (dynamicProvider.isPresent()) {
281         List<Map<String, Object>> items = dropdownOptionProviderService.getDropdownOptions(dynamicProvider.get(), contextResource);
282         if (!items.isEmpty()) {
283           metadataProps.put(PROPERTY_DROPDOWN_OPTIONS, items);
284         }
285         metadataProps.remove(PROPERTY_DROPDOWN_OPTIONS_PROVIDER);
286       }
287     }
288 
289     // check for dynamic root path injection
290     boolean isPathBrowser = WIDGET_TYPE_PATHBROWSER.equals(metadataProps.get(PROPERTY_WIDGET_TYPE));
291     if (isPathBrowser) {
292       Optional<String> dynamicProvider = Optional.ofNullable(metadataProps.get(PROPERTY_PATHBROWSER_ROOT_PATH_PROVIDER))
293           .filter(Objects::nonNull)
294           .map(String::valueOf)
295           .filter(StringUtils::isNotBlank);
296       if (dynamicProvider.isPresent()) {
297         String rootPath = pathBrowserRootPathProviderService.getRootPath(dynamicProvider.get(), contextResource);
298         if (rootPath != null) {
299           metadataProps.put(PROPERTY_PATHBROWSER_ROOT_PATH, rootPath);
300         }
301         metadataProps.remove(PROPERTY_PATHBROWSER_ROOT_PATH_PROVIDER);
302       }
303     }
304 
305     boolean isTagBrowser = WIDGET_TYPE_TAGBROWSER.equals(metadataProps.get(PROPERTY_WIDGET_TYPE));
306     if (isTagBrowser) {
307       Optional<String> dynamicProvider = Optional.ofNullable(metadataProps.get(PROPERTY_TAGBROWSER_ROOT_PATH_PROVIDER))
308           .filter(Objects::nonNull)
309           .map(String::valueOf)
310           .filter(StringUtils::isNotBlank);
311       if (dynamicProvider.isPresent()) {
312         String rootPath = tagBrowserRootPathProviderService.getRootPath(dynamicProvider.get(), contextResource);
313         if (rootPath != null) {
314           metadataProps.put(PROPERTY_TAGBROWSER_ROOT_PATH, rootPath);
315         }
316         metadataProps.remove(PROPERTY_TAGBROWSER_ROOT_PATH_PROVIDER);
317       }
318     }
319 
320     return metadataProps;
321   }

323   private @Nullable Object tryConvertJsonString(@Nullable String value) {
324     if (value == null) {
325       return null;
326     }
327     if (JSON_STRING_ARRAY_PATTERN.matcher(value).matches()) {
328       try {
329         return OBJECT_MAPPER.readValue(value, List.class);
330       }
331       catch (IOException ex) {
332         // no valid json - ignore
333         log.trace("Conversion to JSON array value failed for: {}", value, ex);
334       }
335     }
336     if (JSON_STRING_OBJECT_PATTERN.matcher(value).matches()) {
337       try {
338         return OBJECT_MAPPER.readValue(value, Map.class);
339       }
340       catch (IOException ex) {
341         // no valid json - ignore
342         log.trace("Conversion to JSON object value failed for: {}", value, ex);
343       }
344     }
345     return value;
346   }

348   private @Nullable Boolean isReadOnly(@Nullable String resourcePath, boolean isOverridden) {
349     if (accessControlManager != null && jcrWritePrivilege != null) {
350       try {
351         if (isOverridden) {
352           return true;
353         }
354         else if (resourcePath != null) {
355           return !accessControlManager.hasPrivileges(resourcePath, new Privilege[] { jcrWritePrivilege });
356         }
357       }
358       catch (RepositoryException ex) {
359         log.warn("Unable to check JCR write privilege for resource: {}", resourcePath, ex);
360       }
361     }
362     return null;
363   }
