50  /**
51   * Manages getting and assigning configuration categories.
52   * This service is only available if the bundle io.wcm.sling.commons is present in the system.
53   */
54  @Component(service = ConfigurationCategoryProviderService.class)
55  public class ConfigurationCategoryProviderService {
56  
57    @Reference(cardinality = ReferenceCardinality.MULTIPLE, fieldOption = FieldOption.UPDATE,
58        policy = ReferencePolicy.DYNAMIC, policyOption = ReferencePolicyOption.GREEDY)
59    private SortedSet<ServiceReference<ConfigurationCategoryProvider>> filters = new ConcurrentSkipListSet<>(
60        Collections.reverseOrder());
61  
62    @Reference
63    private ContextAwareServiceResolver serviceResolver;
64    private ContextAwareServiceCollectionResolver<ConfigurationCategoryProvider, Void> serviceCollectionResolver;
65  
66    @Activate
67    private void activate() {
68      // store a map with all categories per name for each provider
69      this.serviceCollectionResolver = serviceResolver.getCollectionResolver(this.filters);
70    }
71  
72    @Deactivate
73    private void deactivate() {
74      this.serviceCollectionResolver.close();
75    }
76  
77    /**
78     * Gets the category (including metadata) that is assigned to a configuration definition.
79     * @param contextResource Content resource
80     * @param configurationMetadata Configuration definition
81     * @return Assigned/detected category, or null if none is detected.
82     */
83    public @Nullable ConfigurationCategory getCategory(@NotNull Resource contextResource, @NotNull ConfigurationMetadata configurationMetadata) {
84  
85      // check for assigned or detected category name
86      String category = getCategoryName(contextResource, configurationMetadata);
87      if (category == null) {
88        return null;
89      }
90  
91      // get category metadata
92      return getCategoryMetadata(contextResource, category);
93    }
94  
95    /**
96     * Get category name that is in the metadata of the configuration definition.
97     * If none is set, check if a {@link ConfigurationCategoryProvider} provides a configuration name.
98     * Returns null if no category found.
99     */
100   private @Nullable String getCategoryName(@NotNull Resource contextResource, @NotNull ConfigurationMetadata configurationMetadata) {
101     String category = getPropertiesString(configurationMetadata.getProperties(), PROPERTY_CATEGORY);
102 
103     if (StringUtils.isEmpty(category)) {
104       category = serviceCollectionResolver.resolveAll(contextResource)
105           .map(provider -> provider.getCategory(configurationMetadata))
106           .filter(Objects::nonNull)
107           .findFirst().orElse(null);
108     }
109 
110     return category;
111   }
112 
113   /**
114    * Try to get configuration metadata from {@link ConfigurationCategoryProvider}.
115    * If not present, return metadata only based on the internal category name as fallback.
116    */
117   @SuppressWarnings("null")
118   private @NotNull ConfigurationCategory getCategoryMetadata(@NotNull Resource contextResource, @NotNull String category) {
119     return serviceCollectionResolver.resolveAll(contextResource)
120         .map(provider -> provider.getCategoryMetadata(category))
121         .filter(Objects::nonNull)
122         .findFirst()
123         .orElse(new ConfigurationCategory(category));
124   }
125 
126   /**
127    * Get directly assigned category bypassing any check for {@link ConfigurationCategoryProvider}.
128    * This is used when io.wcm.sling.commons is not present.
129    * @param configurationMetadata Configuration metadata
130    * @return Category metadata or null if no category assigned
131    */
132   static final @Nullable ConfigurationCategory getAssignedCategory(@NotNull ConfigurationMetadata configurationMetadata) {
133     String category = getPropertiesString(configurationMetadata.getProperties(), PROPERTY_CATEGORY);
134     if (StringUtils.isNotEmpty(category)) {
135       return new ConfigurationCategory(category);
136     }
137     return null;
138   }
139 
140   private static String getPropertiesString(Map<String, String> properties, String key) {
141     if (properties == null) {
142       return null;
143     }
144     return properties.get(key);
145   }
