88    @Override
89    @SuppressWarnings({ "null", "PMD.GuardLogStatement" })
90    protected void doPost(SlingHttpServletRequest request, SlingHttpServletResponse response) throws ServletException, IOException {
91      if (!editorConfig.isEnabled()) {
92        sendForbiddenWithMessage(response, "Configuration editor is disabled.");
93        return;
94      }
95  
96      // get parameters
97      String configName = request.getParameter(RP_CONFIGNAME);
98      if (StringUtils.isBlank(configName)) {
99        response.sendError(HttpServletResponse.SC_NOT_FOUND);
100       return;
101     }
102     boolean collection = BooleanUtils.toBoolean(request.getParameter(RP_COLLECTION));
103 
104     ConfigurationMetadata configMetadata = configManager.getConfigurationMetadata(configName);
105     if (configMetadata != null && configMetadata.isCollection() != collection) {
106       response.sendError(HttpServletResponse.SC_BAD_REQUEST, "Collection parameter mismatch.");
107       return;
108     }
109 
110     // parse JSON parameter data
111     ConfigurationPersistData persistData = null;
112     ConfigurationCollectionPersistData collectionPersistData = null;
113     try {
114       String jsonDataString = IOUtils.toString(request.getInputStream(), StandardCharsets.UTF_8);
115       JsonNode jsonData = OBJECT_MAPPER.readTree(jsonDataString);
116       if (collection) {
117         collectionPersistData = parseCollectionConfigData(jsonData, configMetadata);
118       }
119       else {
120         persistData = parseConfigData(jsonData, configMetadata);
121       }
122     }
123     catch (IOException ex) {
124       response.sendError(HttpServletResponse.SC_BAD_REQUEST, "Invalid JSON data: " + ex.getMessage());
125       return;
126     }
127 
128     // persist data
129     try {
130       if (collection) {
131         configManager.persistConfigurationCollection(request.getResource(), configName, collectionPersistData);
132       }
133       else {
134         configManager.persistConfiguration(request.getResource(), configName, persistData);
135       }
136     }
137     catch (ConfigurationPersistenceAccessDeniedException ex) {
138       sendForbiddenWithMessage(response, ex.getMessage());
139     }
140     catch (ConfigurationPersistenceException ex) {
141       log.warn("Unable to persist data for " + configName + (collection ? "[col]" : ""), ex);
142       response.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, "Unable to persist data: " + ex.getMessage());
143     }
144     /*CHECKSTYLE:OFF*/ catch (Exception ex) { /*CHECKSTYLE:ON*/
145       log.error("Error getting configuration for " + configName + (collection ? "[col]" : ""), ex);
146       response.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, ex.getMessage());
147     }
148   }

150   private ConfigurationCollectionPersistData parseCollectionConfigData(JsonNode jsonData, ConfigurationMetadata configMetadata) {
151     List<ConfigurationPersistData> items = new ArrayList<>();
152     ArrayNode itemsObject = (ArrayNode)jsonData.get("items");
153     for (int i = 0; i < itemsObject.size(); i++) {
154       JsonNode item = itemsObject.get(i);
155       items.add(parseConfigData(item, configMetadata));
156     }
157 
158     Map<String, Object> properties = null;
159     JsonNode propertiesObject = jsonData.get("properties");
160     if (propertiesObject != null) {
161       properties = new HashMap<>();
162       Iterator<String> propertyNames = propertiesObject.fieldNames();
163       while (propertyNames.hasNext()) {
164         String propertyName = propertyNames.next();
165         properties.put(propertyName, toSingle(propertiesObject.get(propertyName)));
166       }
167     }
168 
169     return new ConfigurationCollectionPersistData(items)
170         .properties(properties);
171   }

173   private ConfigurationPersistData parseConfigData(JsonNode item, ConfigurationMetadata configMetadata) {
174     Map<String, Object> props = new HashMap<>();
175     JsonNode properties = item.get("properties");
176     Iterator<String> propertyNames = properties.fieldNames();
177     while (propertyNames.hasNext()) {
178       String propertyName = propertyNames.next();
179       Class propertyType = null;
180       boolean isArray = false;
181       if (configMetadata != null) {
182         PropertyMetadata<?> propertyMetadata = configMetadata.getPropertyMetadata().get(propertyName);
183         if (propertyMetadata != null) {
184           isArray = propertyMetadata.getType().isArray();
185           if (isArray) {
186             propertyType = propertyMetadata.getType().getComponentType();
187           }
188           else {
189             propertyType = propertyMetadata.getType();
190           }
191         }
192       }
193       if (propertyType == ConfigurationMetadata.class || !properties.hasNonNull(propertyName)) {
194         // skip nested configuration and empty properties
195         continue;
196       }
197       else if (propertyType == null) {
198         JsonNode value = properties.get(propertyName);
199         if (value.isArray()) {
200           ArrayNode arrayValue = (ArrayNode)value;
201           if (arrayValue.size() == 0) {
202             props.put(propertyName, ArrayUtils.EMPTY_STRING_ARRAY);
203           }
204           JsonNode firstValue = arrayValue.get(0);
205           if (firstValue.canConvertToInt()) {
206             props.put(propertyName, toArray(arrayValue, int.class));
207           }
208           else if (firstValue.canConvertToLong()) {
209             props.put(propertyName, toArray(arrayValue, long.class));
210           }
211           else if (firstValue.isDouble()) {
212             props.put(propertyName, toArray(arrayValue, double.class));
213           }
214           else if (firstValue.isBoolean()) {
215             props.put(propertyName, toArray(arrayValue, boolean.class));
216           }
217           else if (firstValue.isTextual()) {
218             props.put(propertyName, toArray(arrayValue, String.class));
219           }
220         }
221         else {
222           props.put(propertyName, toSingle(value));
223         }
224       }
225       else if (isArray) {
226         JsonNode value = properties.get(propertyName);
227         if (value.isArray()) {
228           props.put(propertyName, toArray((ArrayNode)value, propertyType));
229         }
230       }
231       else {
232         JsonNode value = properties.get(propertyName);
233         props.put(propertyName, toSingle(value, propertyType));
234       }
235     }
236 
237     String collectionItemName = null;
238     if (configMetadata != null && configMetadata.isCollection()) {
239       collectionItemName = item.get("collectionItemName").textValue();
240     }
241 
242     return new ConfigurationPersistData(props)
243         .collectionItemName(collectionItemName);
244   }

246   private Object toSingle(@NotNull JsonNode value, @NotNull Class propertyType) {
247     if (propertyType.equals(String.class)) {
248       return toString(value);
249     }
250     else if (propertyType.equals(int.class)) {
251       return toInt(value);
252     }
253     else if (propertyType.equals(long.class)) {
254       return toLong(value);
255     }
256     else if (propertyType.equals(double.class)) {
257       return toDouble(value);
258     }
259     else if (propertyType.equals(boolean.class)) {
260       return toBoolean(value);
261     }
262     else {
263       throw new IllegalArgumentException("Unexpected type: " + propertyType.getName());
264     }
265   }

267   private @Nullable Object toSingle(@NotNull JsonNode value) {
268     if (value.isTextual()) {
269       return value.asText();
270     }
271     if (value.canConvertToInt()) {
272       return value.asInt();
273     }
274     if (value.canConvertToLong()) {
275       return value.asLong();
276     }
277     if (value.isDouble()) {
278       return value.doubleValue();
279     }
280     if (value.isBoolean()) {
281       return value.booleanValue();
282     }
283     if (log.isTraceEnabled()) {
284       log.trace("Value '{}' has unexpected type: {}", value, value.getClass().getName());
285     }
286     return null;
287   }

289   private @NotNull Object toArray(@NotNull ArrayNode array, @NotNull Class propertyType) {
290     if (propertyType.equals(String.class)) {
291       String[] values = new String[array.size()];
292       for (int i = 0; i < values.length; i++) {
293         values[i] = toString(array.get(i));
294       }
295       return values;
296     }
297     else if (propertyType.equals(int.class)) {
298       int[] values = new int[array.size()];
299       for (int i = 0; i < values.length; i++) {
300         values[i] = toInt(array.get(i));
301       }
302       return values;
303     }
304     else if (propertyType.equals(long.class)) {
305       long[] values = new long[array.size()];
306       for (int i = 0; i < values.length; i++) {
307         values[i] = toLong(array.get(i));
308       }
309       return values;
310     }
311     else if (propertyType.equals(double.class)) {
312       double[] values = new double[array.size()];
313       for (int i = 0; i < values.length; i++) {
314         values[i] = toDouble(array.get(i));
315       }
316       return values;
317     }
318     else if (propertyType.equals(boolean.class)) {
319       boolean[] values = new boolean[array.size()];
320       for (int i = 0; i < values.length; i++) {
321         values[i] = toBoolean(array.get(i));
322       }
323       return values;
324     }
325     else {
326       throw new IllegalArgumentException("Unexpected type: " + propertyType.getName());
327     }
328   }

330   private @Nullable String toString(@NotNull JsonNode value) {
331     return value.textValue();
332   }
333 
334   private int toInt(@NotNull JsonNode value) {
335     if (value.isTextual()) {
336       return parseNumericTextValue(value, Integer::parseInt, 0);
337     }
338     return value.intValue();
339   }
340 
341   private long toLong(@NotNull JsonNode value) {
342     if (value.isTextual()) {
343       return parseNumericTextValue(value, Long::parseLong, 0L);
344     }
345     return value.longValue();
346   }
347 
348   private double toDouble(@NotNull JsonNode value) {
349     if (value.isTextual()) {
350       return parseNumericTextValue(value, Double::parseDouble, 0d);
351     }
352     return value.doubleValue();
353   }
354 
355   private boolean toBoolean(@NotNull JsonNode value) {
356     if (value.isTextual()) {
357       return Boolean.parseBoolean(value.textValue());
358     }
359     return value.booleanValue();
360   }

362   @SuppressWarnings("null")
363   private <T> @NotNull T parseNumericTextValue(@NotNull JsonNode value, @NotNull Function<String, T> converter, @NotNull T defaultValue) {
364     String textValue = StringUtils.trimToNull(value.textValue());
365     if (textValue != null) {
366       try {
367         return converter.apply(textValue);
368       }
369       catch (NumberFormatException ex) {
370         log.trace("Unable to parse numeric value: {}", textValue);
371       }
372     }
373     return defaultValue;
374   }

376   @Override
377   @SuppressWarnings("PMD.GuardLogStatement")
378   protected void doDelete(@NotNull SlingHttpServletRequest request, @NotNull SlingHttpServletResponse response) throws ServletException, IOException {
379 
380     // get parameters
381     String configName = request.getParameter(RP_CONFIGNAME);
382     if (StringUtils.isBlank(configName)) {
383       response.sendError(HttpServletResponse.SC_NOT_FOUND);
384       return;
385     }
386 
387     // delete data
388     try {
389       configManager.deleteConfiguration(request.getResource(), configName);
390     }
391     catch (ConfigurationPersistenceAccessDeniedException ex) {
392       sendForbiddenWithMessage(response, ex.getMessage());
393     }
394     catch (ConfigurationPersistenceException ex) {
395       log.warn("Unable to delete data for " + configName, ex);
396       response.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, "Unable to delete data: " + ex.getMessage());
397     }
398     /*CHECKSTYLE:OFF*/ catch (Exception ex) { /*CHECKSTYLE:ON*/
399       log.error("Error deleting configuration for " + configName, ex);
400       response.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, ex.getMessage());
401     }
402   }

404   private void sendForbiddenWithMessage(SlingHttpServletResponse response, String message) throws IOException {
405     response.setContentType("text/plain;charset=" + StandardCharsets.UTF_8.name());
406     response.getWriter().write(message);
407     response.setStatus(HttpServletResponse.SC_FORBIDDEN);
408   }
