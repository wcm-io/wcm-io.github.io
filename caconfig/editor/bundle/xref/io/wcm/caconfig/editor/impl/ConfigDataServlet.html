94    @Override
95    @SuppressWarnings("PMD.GuardLogStatement")
96    protected void doGet(@NotNull SlingHttpServletRequest request, @NotNull SlingHttpServletResponse response) throws ServletException, IOException {
97      if (!editorConfig.isEnabled()) {
98        response.sendError(HttpServletResponse.SC_FORBIDDEN);
99        return;
100     }
101 
102     // get parameters
103     String configName = request.getParameter(RP_CONFIGNAME);
104     if (StringUtils.isBlank(configName)) {
105       response.sendError(HttpServletResponse.SC_NOT_FOUND);
106       return;
107     }
108     boolean collection = BooleanUtils.toBoolean(request.getParameter(RP_COLLECTION));
109 
110     // output configuration
111     try {
112       Object result = getConfiguration(request.getResource(), configName, collection);
113       if (result == null) {
114         response.sendError(HttpServletResponse.SC_NOT_FOUND);
115       }
116       else {
117         response.setContentType("application/json;charset=" + StandardCharsets.UTF_8.name());
118         response.getWriter().write(OBJECT_MAPPER.writeValueAsString(result));
119       }
120     }
121     /*CHECKSTYLE:OFF*/ catch (Exception ex) { /*CHECKSTYLE:ON*/
122       log.error("Error getting configuration for " + configName + (collection ? "[col]" : ""), ex);
123       response.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, ex.getMessage());
124     }
125   }

127   private Object getConfiguration(Resource contextResource, String configName, boolean collection) {
128     Object result;
129     if (collection) {
130       ConfigurationData newItem = configManager.newCollectionItem(contextResource, configName);
131       if (newItem == null) {
132         throw new ConfigurationPersistenceException("Invalid configuration name: " + configName);
133       }
134       result = fromConfigCollection(configManager.getConfigurationCollection(contextResource, configName), newItem, configName);
135     }
136     else {
137       ConfigurationData configData = configManager.getConfiguration(contextResource, configName);
138       if (configData != null) {
139         result = fromConfig(configData, configData.isInherited(), configName);
140       }
141       else {
142         result = null;
143       }
144     }
145     return result;
146   }

148   private ConfigCollectionItem fromConfigCollection(ConfigurationCollectionData configCollection, ConfigurationData newItem, String fullConfigName) {
149     ConfigCollectionItem result = new ConfigCollectionItem();
150     result.setConfigName(configCollection.getConfigName());
151 
152     if (!configCollection.getProperties().isEmpty()) {
153       Map<String, Object> properties = new TreeMap<>();
154       for (Map.Entry<String, Object> entry : configCollection.getProperties().entrySet()) {
155         properties.put(entry.getKey(), entry.getValue());
156       }
157       result.setProperties(properties);
158     }
159 
160     List<ConfigItem> items = new ArrayList<>();
161     for (ConfigurationData configData : configCollection.getItems()) {
162       items.add(fromConfig(configData, configData.isInherited(), fullConfigName));
163     }
164     result.setItems(items);
165 
166     result.setNewItem(fromConfig(newItem, null, fullConfigName));
167 
168     return result;
169   }

171   private ConfigItem fromConfig(ConfigurationData config, Boolean inherited, String fullConfigName) {
172     ConfigItem result = new ConfigItem();
173 
174     result.setConfigName(config.getConfigName());
175     result.setCollectionItemName(config.getCollectionItemName());
176     result.setOverridden(config.isOverridden());
177     result.setInherited(inherited);
178 
179     List<PropertyItem> props = new ArrayList<>();
180     for (String propertyName : config.getPropertyNames()) {
181       ValueInfo<?> item = config.getValueInfo(propertyName);
182       if (item == null) {
183         continue;
184       }
185       PropertyMetadata<?> itemMetadata = item.getPropertyMetadata();
186 
187       PropertyItem prop = new PropertyItem();
188       prop.setName(item.getName());
189 
190       // special handling for nested configs and nested config collections
191       if (itemMetadata != null && itemMetadata.isNestedConfiguration()) {
192         PropertyItemMetadata metadata = new PropertyItemMetadata();
193         metadata.setLabel(itemMetadata.getLabel());
194         metadata.setDescription(itemMetadata.getDescription());
195         metadata.setProperties(toJsonWithValueConversion(itemMetadata.getProperties()));
196         prop.setMetadata(metadata);
197 
198         if (itemMetadata.getType().isArray()) {
199           ConfigurationData[] configDatas = (ConfigurationData[])item.getValue();
200           if (configDatas != null) {
201             ConfigCollectionItem nestedConfigCollection = new ConfigCollectionItem();
202             StringBuilder collectionConfigName = new StringBuilder();
203             if (config.getCollectionItemName() != null) {
204               collectionConfigName.append(configurationPersistenceStrategy.getCollectionItemConfigName(fullConfigName
205                       + "/" + config.getCollectionItemName(), config.getResourcePath()));
206             }
207             else {
208               collectionConfigName.append(configurationPersistenceStrategy.getConfigName(fullConfigName, config.getResourcePath()));
209             }
210             collectionConfigName.append("/").append(itemMetadata.getConfigurationMetadata().getName());
211             nestedConfigCollection.setConfigName(collectionConfigName.toString());
212             List<ConfigItem> items = new ArrayList<>();
213             for (ConfigurationData configData : configDatas) {
214               items.add(fromConfig(configData, false, collectionConfigName.toString()));
215             }
216             nestedConfigCollection.setItems(items);
217             prop.setNestedConfigCollection(nestedConfigCollection);
218           }
219         }
220         else {
221           ConfigurationData configData = (ConfigurationData)item.getValue();
222           if (configData != null) {
223             prop.setNestedConfig(fromConfig(configData, null, fullConfigName
224                 + "/" + itemMetadata.getConfigurationMetadata().getName()));
225           }
226         }
227       }
228 
229       // property data and metadata
230       else {
231         prop.setValue(item.getValue());
232         prop.setEffectiveValue(item.getEffectiveValue());
233         prop.setConfigSourcePath(item.getConfigSourcePath());
234         prop.setIsDefault(item.isDefault());
235         prop.setInherited(item.isInherited());
236         prop.setOverridden(item.isOverridden());
237 
238         if (itemMetadata != null) {
239           PropertyItemMetadata metadata = new PropertyItemMetadata();
240           if (itemMetadata.getType().isArray()) {
241             metadata.setType(ClassUtils.primitiveToWrapper(itemMetadata.getType().getComponentType()).getSimpleName());
242             metadata.setMultivalue(true);
243           }
244           else {
245             metadata.setType(ClassUtils.primitiveToWrapper(itemMetadata.getType()).getSimpleName());
246           }
247           metadata.setDefaultValue(itemMetadata.getDefaultValue());
248           metadata.setLabel(itemMetadata.getLabel());
249           metadata.setDescription(itemMetadata.getDescription());
250           metadata.setProperties(toJsonWithValueConversion(itemMetadata.getProperties()));
251           prop.setMetadata(metadata);
252         }
253       }
254       props.add(prop);
255     }
256     result.setProperties(props);
257 
258     return result;
259   }

261   /**
262    * Converts the given map to JSON. Each map value is checked for a valid JSON string - if this is the case it's
263    * inserted as JSON objects and not as string.
264    * @param properties Map
265    * @return JSON object
266    */
267   private @Nullable Map<String, Object> toJsonWithValueConversion(@Nullable Map<String, String> properties) {
268     if (properties == null || properties.isEmpty()) {
269       return null;
270     }
271     else {
272       Map<String, Object> metadataProps = new TreeMap<>();
273       for (Map.Entry<String, String> entry : properties.entrySet()) {
274         metadataProps.put(entry.getKey(), tryConvertJsonString(entry.getValue()));
275       }
276       return metadataProps;
277     }
278   }
279 
280   private @Nullable Object tryConvertJsonString(@Nullable String value) {
281     if (value == null) {
282       return null;
283     }
284     if (JSON_STRING_ARRAY_PATTERN.matcher(value).matches()) {
285       try {
286          return OBJECT_MAPPER.readValue(value, List.class);
287       }
288       catch (IOException ex) {
289         // no valid json - ignore
290         log.trace("Conversion to JSON arary value failed for: {}", value, ex);
291       }
292     }
293     if (JSON_STRING_OBJECT_PATTERN.matcher(value).matches()) {
294       try {
295         return OBJECT_MAPPER.readValue(value, Map.class);
296       }
297       catch (IOException ex) {
298         // no valid json - ignore
299         log.trace("Conversion to JSON object value failed for: {}", value, ex);
300       }
301     }
302     return value;
303   }
